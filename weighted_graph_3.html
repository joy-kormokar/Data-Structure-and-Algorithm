01
Introduction to Weighted Graphs
Foundations · What, Why & How
📖
Theory
A graph is a collection of vertices (nodes) connected by edges (links).

A weighted graph adds a numeric value — the weight — to every edge. This weight can represent distance, cost, time, or bandwidth.

Formally: G = (V, E, W) — V = vertices, E = edges, W: E→ℝ assigns a weight to each edge.

An Incidence List (adjacency list) stores, for each vertex, a linked list of neighbors plus edge weights. This is memory-efficient for sparse graphs: O(V + E) vs O(V²) for a matrix.
🌍 Real-Life Analogy
Think of a road map: cities are vertices, roads are edges, and distances (in km) are weights. Finding the shortest route from your city to another is exactly a weighted graph problem!

C Code — Graph Structure (Incidence List)
C — Core Data Structures
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX 100
#define INF 99999

/* Node in the adjacency (incidence) list */
struct AdjNode {
    int dest;            /* index of neighbor vertex */
    int weight;          /* edge weight             */
    struct AdjNode *next; /* pointer to next entry   */
};

/* One vertex: label + its incidence list */
struct Vertex {
    char name[50];
    int  active;          /* 1=exists, 0=deleted */
    struct AdjNode *head; /* incidence list head  */
};

/* The graph itself */
struct Graph {
    struct Vertex v[MAX];
    int n;                /* number of vertices   */
};

struct Graph* createGraph() {
    struct Graph *g = (struct Graph*)malloc(sizeof(struct Graph));
    g->n = 0;
    for(int i=0;i<MAX;i++){ g->v[i].active=0; g->v[i].head=NULL; }
    return g;
}
/* Helper: allocate adjacency node */
struct AdjNode* newNode(int dest, int w){
    struct AdjNode *nd=(struct AdjNode*)malloc(sizeof(struct AdjNode));
    nd->dest=dest; nd->weight=w; nd->next=NULL;
    return nd;
}
02
Weighted Incidence Matrix
Representation · How the graph lives in memory
📖
Theory
An Incidence List stores each vertex's edges as a linked list: [dest | weight | next→]. We compare this against the adjacency matrix:

• Adjacency List: O(V + E) memory — great for sparse graphs
• Adjacency Matrix: O(V²) memory — always a full grid

For 5 vertices and 6 edges: List needs ~11 nodes; Matrix needs 25 cells.
📱 Analogy
A phone contact list vs. a grid of "do you know everyone on Earth". The list only stores your actual connections — that's the incidence list advantage.

    for(int i=0;i<g->n;i++){

Insert Vertex
Vertex Operations · Adding a new node to the graph
📖
Theory
Inserting a vertex adds a new node with an empty incidence list. We store its name, set active=1, and set head=NULL. No edges yet.

Time: O(1) — just write into the next free slot.
🏙️ Analogy
Adding a new city to a map. It exists on the map, but no roads connect to it yet.

C — insertVertex()
int insertVertex(struct Graph *g, const char *name){
    if(g->n >= MAX){ printf("Graph full!\n"); return -1; }
    int id = g->n;
    strcpy(g->v[id].name, name);
    g->v[id].active = 1;
    g->v[id].head   = NULL;  /* empty incidence list */

    return id;
}

    insertVertex(g, "A");  /* index 0 */

    insertVertex(g, "C");  /* index 2 */
    return 0;
}
04
Delete Vertex
Vertex Operations · Removing a node and all its edges
📖
Theory
Deleting a vertex requires 3 steps:
1. Free its own incidence list (outgoing edges)
2. Remove all references to it from other vertices' lists (incoming edges)
3. Mark it inactive (active=0)

Time: O(V + E) — must scan all lists to remove incoming references.
🏙️ Analogy
Demolishing a city — all roads that connected TO it must also be removed from the map.
💻
C Code
C — deleteVertex()
/* Remove a single adjacency node pointing to 'target' */
struct AdjNode* removeFromList(struct AdjNode *head, int target){
    struct AdjNode *cur=head, *prev=NULL;
    while(cur){
        if(cur->dest==target){
            if(prev) prev->next=cur->next;
            else     head=cur->next;
            free(cur); return head;
        }
        prev=cur; cur=cur->next;
    }
    return head;
}

void deleteVertex(struct Graph *g, int vid){
    if(!g->v[vid].active){ printf("Not found\n"); return; }
    /* Step 1: free its own list */
    struct AdjNode *c=g->v[vid].head;
    while(c){ struct AdjNode *t=c->next; free(c); c=t; }
    g->v[vid].head=NULL;
    /* Step 2: remove incoming edges from others */
    for(int i=0;i<g->n;i++)
        if(i!=vid && g->v[i].active)
            g->v[i].head = removeFromList(g->v[i].head, vid);
    /* Step 3: mark deleted */
    g->v[vid].active=0;
    printf("Vertex %d deleted\n", vid);
}

Insert Weighted Edge
Edge Operations · Adding a connection with weight
📖
Theory
Adding edge (src→dest, weight): create an AdjNode(dest, weight) and prepend it to src's incidence list.

For undirected graphs, also add (dest→src, weight).
Time: O(1) — prepend to head of list.
🛣️ Analogy
Building a road between two cities with a known distance. Two-way road = undirected edge (added to both cities' lists).

C Code
C — insertEdge() / insertUndirectedEdge()
/* Directed edge: src -> dest */
void insertEdge(struct Graph *g, int src, int dest, int w){
    if(!g->v[src].active||!g->v[dest].active){
        printf("Invalid vertex\n"); return;
    }
    struct AdjNode *nd = newNode(dest, w);
    nd->next = g->v[src].head;  /* prepend */
    g->v[src].head = nd;
    printf("Edge %s->%s (w=%d) added\n",
           g->v[src].name, g->v[dest].name, w);
}

/* Undirected: add both directions */
void insertUndirectedEdge(struct Graph *g, int u, int v, int w){
    insertEdge(g, u, v, w);
    insertEdge(g, v, u, w);
}

void deleteEdge(struct Graph *g, int src, int dest){
    struct AdjNode *cur=g->v[src].head, *prev=NULL;
    while(cur){
        if(cur->dest==dest){
            if(prev) prev->next=cur->next;
            else     g->v[src].head=cur->next;
            free(cur);
            printf("Edge %d->%d deleted\n", src, dest);
            return;
        }
        prev=cur; cur=cur->next;
    }
    printf("Edge not found\n");
}

BFS — Breadth-First Search
Traversal · Level-by-level exploration using a queue
📖
Theory
BFS explores vertices level by level using a queue. Visit start, then all its neighbors, then their neighbors, etc.

In weighted graphs BFS finds the path with fewest edges (not lowest weight). Use Dijkstra's for weight-based shortest paths.
Time: O(V + E).
🌊 Analogy
Drop a stone in water — ripples spread outward in rings, visiting nearer points before farther ones. BFS visits nearby nodes first.
🎨

💻
C Code
C — BFS()
void BFS(struct Graph *g, int start){
    int visited[MAX]={0};
    int queue[MAX], front=0, rear=0;
    visited[start]=1;
    queue[rear++]=start;
    printf("BFS: ");
    while(front<rear){
        int v=queue[front++];          /* dequeue */
        printf("%s ", g->v[v].name);   /* visit   */
        struct AdjNode *c=g->v[v].head;
        while(c){
            if(!visited[c->dest]){
                visited[c->dest]=1;
                queue[rear++]=c->dest; /* enqueue */
            }
            c=c->next;
        }
    }
    printf("\n");
}

DFS — Depth-First Search
Traversal · Dive deep then backtrack
📖
Theory
DFS goes as deep as possible before backtracking. Uses recursion (or explicit stack).

Used for: cycle detection, topological sort, connected components, strongly connected components.
Time: O(V + E).
🗺️ Analogy
Exploring a maze — always go forward until stuck, then backtrack and try another branch.

Weighted Traversal Analysis
Traversal · Tracking cumulative weight during DFS
📖
Theory
During DFS we accumulate edge weights to compute the cumulative cost of the traversal path. This shows how "expensive" each visited vertex is to reach via the DFS route.

Use Case: Delivery route analysis — find total distance traveled during a full sweep of all locations.
💻
C Code
C — weightedDFS() with cumulative cost
void weightedDFS(struct Graph *g, int v, int *vis, int cost){
    vis[v]=1;
    printf("%s(cum=%d) ", g->v[v].name, cost);
    struct AdjNode *c=g->v[v].head;
    while(c){
        if(!vis[c->dest])
            weightedDFS(g, c->dest, vis, cost+c->weight);
        c=c->next;
    }
}
void runWeightedTraversal(struct Graph *g, int start){
    int vis[MAX]={0};
    printf("Weighted DFS from %s:\n", g->v[start].name);
    weightedDFS(g, start, vis, 0);
}
/* Output: A(cum=0) B(cum=4) C(cum=7) D(cum=9) E(cum=14) */
15
Find Path
Path Operations · Does a path exist between two vertices?
📖
Theory
Run DFS from src. If dest is visited, a path exists. Use a parent array to reconstruct the actual path.
Time: O(V + E).
🗺️ Analogy
Checking if you can drive from City A to City B using any sequence of available roads.
💻
C Code
C — findPath()
int fpDFS(struct Graph *g, int cur, int dest,
           int *vis, int *path, int pLen){
    vis[cur]=1; path[pLen++]=cur;
    if(cur==dest){
        printf("Path: ");
        for(int i=0;i<pLen;i++) printf("%d%s",path[i],i<pLen-1?"->":"\n");
        return 1;
    }
    struct AdjNode *c=g->v[cur].head;
    while(c){
        if(!vis[c->dest] && fpDFS(g,c->dest,dest,vis,path,pLen))
            return 1;
        c=c->next;
    }
    return 0;
}
void findPath(struct Graph *g, int src, int dst){
    int vis[MAX]={0}, path[MAX];
    if(!fpDFS(g,src,dst,vis,path,0)) printf("No path\n");
}

Shortest Path (DFS-based)
Path Operations · Minimum weight path via exhaustive search
📖
Theory
Explore all paths via DFS and keep track of the minimum total weight found. Practical only for small graphs — use Dijkstra's (section 20) for production.
Time: O(V!) worst case.
💻
C Code
C — shortestPathDFS()
int minCost;
void spDFS(struct Graph *g, int cur, int dst, int *vis, int cost){
    if(cur==dst){ if(cost<minCost) minCost=cost; return; }
    vis[cur]=1;
    struct AdjNode *c=g->v[cur].head;
    while(c){
        if(!vis[c->dest]) spDFS(g,c->dest,dst,vis,cost+c->weight);
        c=c->next;
    }
    vis[cur]=0; /* unmark for other paths */
}
int shortestPath(struct Graph *g, int src, int dst){
    int vis[MAX]={0}; minCost=INF;
    spDFS(g,src,dst,vis,0);
    printf("Shortest cost: %d\n", minCost);
    return minCost;
}

Longest Path (Specific Cases)
Path Operations · Maximum weight path (works on DAGs)
📖
Theory
The longest path is NP-hard in general, but solvable for DAGs (Directed Acyclic Graphs). For DAGs use topological sort + DP. For small graphs use DFS tracking max instead of min.

Use Case: Critical Path Method (CPM) in project scheduling — the longest dependency chain defines project duration.
🏗️ Analogy
Project management: which chain of tasks takes the longest? You can't finish before that critical chain completes.
💻
C Code
C — longestPath()
int maxCost;
void lpDFS(struct Graph *g, int cur, int dst, int *vis, int cost){
    if(cur==dst){ if(cost>maxCost) maxCost=cost; return; }
    vis[cur]=1;
    struct AdjNode *c=g->v[cur].head;
    while(c){
        if(!vis[c->dest]) lpDFS(g,c->dest,dst,vis,cost+c->weight);
        c=c->next;
    }
    vis[cur]=0; /* allow other paths through this node */
}
int longestPath(struct Graph *g, int src, int dst){
    int vis[MAX]={0}; maxCost=-INF;
    lpDFS(g,src,dst,vis,0);
    printf("Longest cost: %d\n", maxCost);
    return maxCost;
}

All Paths
Path Operations · Enumerate every route between two vertices
📖
Theory
DFS with backtracking (unmark visited after each branch) to explore every possible route from src to dest.
Time: O(V!) — exponential. Only practical for small graphs.

Use Case: Network reliability — how many independent routes exist between two routers?
💻
C Code
C — printAllPaths()
void apDFS(struct Graph *g, int cur, int dst,
           int *vis, int *path, int pLen, int cost){
    vis[cur]=1; path[pLen++]=cur;
    if(cur==dst){
        printf("Path(cost=%d): ", cost);
        for(int i=0;i<pLen;i++) printf("%d%s",path[i],i<pLen-1?"->":"\n");
    } else {
        struct AdjNode *c=g->v[cur].head;
        while(c){
            if(!vis[c->dest])
                apDFS(g,c->dest,dst,vis,path,pLen,cost+c->weight);
            c=c->next;
        }
    }
    vis[cur]=0; /* backtrack */
}
void printAllPaths(struct Graph *g, int src, int dst){
    int vis[MAX]={0}, path[MAX];
    apDFS(g,src,dst,vis,path,0,0);
}

Minimum Cost Path
Path Operations · Path with lowest total weight + full route
📖
Theory
Extends shortest path to also record the exact vertex sequence forming the cheapest route. For production use Dijkstra's.
✈️ Analogy
Cheapest flight itinerary — not just the total price, but which specific sequence of flights achieves it.
💻
C Code
C — minCostPath() with path recording
int bestCost, bestPath[MAX], bestLen;
void mcpDFS(struct Graph *g, int cur, int dst,
            int *vis, int *path, int pLen, int cost){
    path[pLen++]=cur;
    if(cur==dst){
        if(cost<bestCost){
            bestCost=cost; bestLen=pLen;
            for(int i=0;i<pLen;i++) bestPath[i]=path[i];
        }
        return;
    }
    vis[cur]=1;
    struct AdjNode *c=g->v[cur].head;
    while(c){
        if(!vis[c->dest]) mcpDFS(g,c->dest,dst,vis,path,pLen,cost+c->weight);
        c=c->next;
    }
    vis[cur]=0;
}
void minCostPath(struct Graph *g, int src, int dst){
    int vis[MAX]={0}, path[MAX];
    bestCost=INF; bestLen=0;
    mcpDFS(g,src,dst,vis,path,0,0);
    printf("Min cost=%d Path: ", bestCost);
    for(int i=0;i<bestLen;i++) printf("%d%s",bestPath[i],i<bestLen-1?"->":"\n");
}
20
Dijkstra's Algorithm
Shortest Path Algorithms · Greedy single-source, non-negative weights
📖
Theory
Algorithm steps:
1. Set dist[src]=0, all others=∞
2. Pick unvisited vertex u with smallest dist
3. Relax all neighbors: if dist[u]+w < dist[v], update dist[v]
4. Mark u visited; repeat until all visited

Constraint: Only works with non-negative edge weights.
Time: O(V²) with array; O((V+E) log V) with min-heap.
🗺️ Analogy
GPS navigation — always extends the currently shortest known path, guaranteeing optimal routes to every destination.
🎨
Dijkstra's Step-by-Step Animation
Click Run to animate Dijkstra's algorithm step by step.
💻
C Code
C — dijkstra()
int minDist(int *d, int *vis, int n){
    int m=-1;
    for(int i=0;i<n;i++) if(!vis[i]&&(m==-1||d[i]<d[m])) m=i;
    return m;
}
void dijkstra(struct Graph *g, int src){
    int n=g->n, dist[MAX], vis[MAX]={0}, par[MAX];
    for(int i=0;i<n;i++){ dist[i]=INF; par[i]=-1; }
    dist[src]=0;
    for(int it=0;it<n;it++){
        int u=minDist(dist,vis,n);
        if(u==-1||dist[u]==INF) break;
        vis[u]=1;
        struct AdjNode *c=g->v[u].head;
        while(c){
            int v=c->dest, w=c->weight;
            if(!vis[v] && dist[u]+w<dist[v]){ dist[v]=dist[u]+w; par[v]=u; }
            c=c->next;
        }
    }
    printf("Dijkstra from %s:\n", g->v[src].name);
    for(int i=0;i<n;i++)
        printf("  %s: dist=%d\n", g->v[i].name, dist[i]);
}
21
Bellman-Ford Algorithm
Shortest Path · Handles negative weights & detects negative cycles
📖
Theory
Relax ALL edges V-1 times. A V-th relaxation means a negative cycle exists.

Advantage over Dijkstra: Works with negative edge weights.
Time: O(V × E) — slower but more general.
💸 Analogy
Currency arbitrage — negative edges = profit on exchange. If you can loop forever gaining money, Bellman-Ford detects that negative cycle.
💻
C Code
C — bellmanFord()
void bellmanFord(struct Graph *g, int src){
    int n=g->n, dist[MAX];
    for(int i=0;i<n;i++) dist[i]=INF;
    dist[src]=0;
    /* Relax all edges n-1 times */
    for(int iter=0;iter<n-1;iter++){
        for(int u=0;u<n;u++){
            if(!g->v[u].active||dist[u]==INF) continue;
            struct AdjNode *c=g->v[u].head;
            while(c){
                if(dist[u]+c->weight<dist[c->dest])
                    dist[c->dest]=dist[u]+c->weight;
                c=c->next;
            }
        }
    }
    /* Check for negative cycles (Vth relaxation) */
    for(int u=0;u<n;u++){
        struct AdjNode *c=g->v[u].head;
        while(c){
            if(dist[u]+c->weight<dist[c->dest]){
                printf("Negative cycle detected!\n"); return;
            }
            c=c->next;
        }
    }
    printf("Bellman-Ford from %d:\n",src);
    for(int i=0;i<n;i++) printf("  %s: %d\n",g->v[i].name,dist[i]);
}

Floyd-Warshall Algorithm
Shortest Path · All-pairs shortest paths in O(V³)
📖
Theory
Uses DP to compute shortest paths between all pairs of vertices.

Key recurrence: dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]) for each intermediate vertex k.
Time: O(V³).
📡 Analogy
Phone network routing table — for every pair of phones, find the cheapest relay path through any intermediate exchanges.
🎨

void floydWarshall(struct Graph *g){
    int n=g->n, d[MAX][MAX];
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) d[i][j]=(i==j)?0:INF;
    for(int u=0;u<n;u++){
        struct AdjNode *c=g->v[u].head;
        while(c){ d[u][c->dest]=c->weight; c=c->next; }
    }
    for(int k=0;k<n;k++)
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                if(d[i][k]!=INF&&d[k][j]!=INF&&d[i][k]+d[k][j]<d[i][j])
                    d[i][j]=d[i][k]+d[k][j];
    printf("All-pairs distances:\n");
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            d[i][j]==INF?printf("INF "):printf("%3d ",d[i][j]);
        printf("\n");
    }
}

A* Search (Advanced)
Shortest Path · Heuristic-guided search, faster in practice
📖
Theory
A* extends Dijkstra's with a heuristic h(v) estimating remaining cost to the goal.

Priority: f(v) = g(v) + h(v) where g(v)=actual cost from source, h(v)=estimated cost to goal.

With an admissible heuristic (never overestimates), A* is both complete and optimal. Much faster in practice for pathfinding in spatial graphs.
🧭 Analogy
A smarter GPS — instead of exploring all roads equally, it prioritizes roads that head toward your destination. Same optimal result, but far fewer nodes checked.
💻
C Code
C — aStar()
/* heuristic[v] = estimated cost from v to goal */
void aStar(struct Graph *g, int src, int goal, int *h){
    int n=g->n, gCost[MAX], fCost[MAX], vis[MAX]={0}, par[MAX];
    for(int i=0;i<n;i++){ gCost[i]=INF; fCost[i]=INF; par[i]=-1; }
    gCost[src]=0; fCost[src]=h[src];
    for(int it=0;it<n;it++){
        int u=-1;
        for(int i=0;i<n;i++) if(!vis[i]&&(u==-1||fCost[i]<fCost[u])) u=i;
        if(u==-1||u==goal) break;
        vis[u]=1;
        struct AdjNode *c=g->v[u].head;
        while(c){
            int v=c->dest, ng=gCost[u]+c->weight;
            if(ng<gCost[v]){ gCost[v]=ng; fCost[v]=ng+h[v]; par[v]=u; }
            c=c->next;
        }
    }
    printf("A* cost to goal: %d\n", gCost[goal]);
}

Theory

Kruskal's Algorithm
MST · Sort edges, pick cheapest non-cycle ones
📖
Theory
Kruskal's: Sort all edges by weight. Greedily add each edge unless it creates a cycle (use Union-Find to check).
Time: O(E log E) for sorting.
🌐 Analogy
Building a network: list all cables by cost, install cheapest first, skip any cable that creates a redundant loop.
💻
C Code
C — kruskalMST() with Union-Find
struct Edge { int src,dst,w; };
int find(int *p,int x){ if(p[x]!=x) p[x]=find(p,p[x]); return p[x]; }
void unite(int *p,int *r,int x,int y){
    int rx=find(p,x),ry=find(p,y);
    if(r[rx]<r[ry]) p[rx]=ry;
    else if(r[rx]>r[ry]) p[ry]=rx;
    else{p[ry]=rx;r[rx]++;}
}
int cmpE(const void*a,const void*b){ return ((struct Edge*)a)->w-((struct Edge*)b)->w; }
void kruskalMST(struct Graph *g){
    int n=g->n,ec=0; struct Edge edges[MAX*MAX];
    for(int i=0;i<n;i++){
        struct AdjNode*c=g->v[i].head;
        while(c){edges[ec].src=i;edges[ec].dst=c->dest;edges[ec++].w=c->weight;c=c->next;}
    }
    qsort(edges,ec,sizeof(struct Edge),cmpE);
    int par[MAX],rnk[MAX]={0},total=0;
    for(int i=0;i<n;i++) par[i]=i;
    printf("Kruskal MST:\n");
    for(int i=0;i<ec;i++){
        int u=edges[i].src,v=edges[i].dst;
        if(find(par,u)!=find(par,v)){
            unite(par,rnk,u,v);
            printf("  %s-%s w=%d\n",g->v[u].name,g->v[v].name,edges[i].w);
            total+=edges[i].w;
        }
    }
    printf("Total: %d\n",total);
}

Network Flow
Advanced · Maximum flow from source to sink (Ford-Fulkerson)
📖
Theory
Models how much "material" (data, water, cars) can flow from source to sink through edges with capacities.

Ford-Fulkerson / Edmonds-Karp:
1. Find augmenting path (BFS)
2. Push flow = bottleneck (min capacity on path)
3. Update residual capacities; repeat
Time: O(V × E²).
🚰 Analogy
Water reservoir (source) → city (sink) through pipes of limited capacity. Maximum flow = maximum water deliverable per second.
💻
C Code
C — maxFlow() Edmonds-Karp
int cap[MAX][MAX]; /* capacity matrix */

int bfsFlow(int src,int sink,int n,int *par){
    int vis[MAX]={0},q[MAX],f=0,r=0;
    vis[src]=1;par[src]=-1;q[r++]=src;
    while(f<r){ int u=q[f++];
        for(int v=0;v<n;v++) if(!vis[v]&&cap[u][v]>0){par[v]=u;vis[v]=1;if(v==sink)return 1;q[r++]=v;}
    } return 0;
}
int maxFlow(int src,int sink,int n){
    int par[MAX],flow=0;
    while(bfsFlow(src,sink,n,par)){
        int pf=INF,v=sink;
        while(v!=src){ int u=par[v]; if(cap[u][v]<pf)pf=cap[u][v]; v=u; }
        v=sink;
        while(v!=src){ int u=par[v]; cap[u][v]-=pf; cap[v][u]+=pf; v=u; }
        flow+=pf; printf("Aug path flow: %d\n",pf);
    }
    printf("Max Flow: %d\n",flow); return flow;
}

    for(int u=0;u<n;u++){

        while(c){insertEdge(gT,c->dest,u,c->weight);c=c->next;}
    }
    int vis2[MAX]={0};
    printf("SCCs:\n");
    while(top>=0){ int v=stk[top--]; if(!vis2[v]){printf("{ ");dfs2(gT,v,vis2);printf("}\n");} }

void printCostMatrix(struct Graph *g){
    int n=g->n, d[MAX][MAX];
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) d[i][j]=(i==j)?0:INF;
    for(int u=0;u<n;u++){ struct AdjNode*c=g->v[u].head; while(c){d[u][c->dest]=c->weight;c=c->next;} }
    for(int k=0;k<n;k++) for(int i=0;i<n;i++) for(int j=0;j<n;j++)
        if(d[i][k]!=INF&&d[k][j]!=INF&&d[i][k]+d[k][j]<d[i][j]) d[i][j]=d[i][k]+d[k][j];
    printf("    ");
    for(int j=0;j<n;j++) printf("%5s",g->v[j].name);
    printf("\n");
    for(int i=0;i<n;i++){
        printf("%4s",g->v[i].name);
        for(int j=0;j<n;j++) d[i][j]==INF?printf("    ∞"):printf("%5d",d[i][j]);
        printf("\n");
    }
}

void routeSummary(struct Graph *g, int src){
    int n=g->n,dist[MAX],vis[MAX]={0},par[MAX];
    for(int i=0;i<n;i++){dist[i]=INF;par[i]=-1;}
    dist[src]=0;
    for(int it=0;it<n;it++){
        int u=minDist(dist,vis,n); if(u==-1) break;
        vis[u]=1; struct AdjNode*c=g->v[u].head;
        while(c){ if(!vis[c->dest]&&dist[u]+c->weight<dist[c->dest]){dist[c->dest]=dist[u]+c->weight;par[c->dest]=u;} c=c->next; }
    }
    printf("╔═══ ROUTE SUMMARY from %s ═══╗\n",g->v[src].name);
    printf("║ %-8s %-6s %-15s ║\n","Dest","Cost","Path");
    printf("╠════════════════════════════╣\n");
    for(int i=0;i<n;i++){
        if(i==src||!g->v[i].active) continue;
        printf("║ %-8s %-6d ",g->v[i].name,dist[i]);
        printRoute(g,par,i); printf(" ║\n");
    }
    printf("╚════════════════════════════╝\n");
}
