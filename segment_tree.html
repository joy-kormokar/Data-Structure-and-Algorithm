// What is a Segment Tree?

A Segment Tree is a binary tree where each node stores information about a segment (range) of an array. The root covers the entire array, and each leaf represents a single element.

For an array of size n, the segment tree is stored in an array of size 4×n to safely accommodate all nodes.

  • Root is at index 1 (1-indexed)
  • Left child of node i → 2i
  • Right child of node i → 2i+1
  • Parent of node i → i/2

Ccreate_tree.c
#include <stdio.h>
#include <stdlib.h>

/* A segment tree for array of size n needs 4*n space */
#define MAXN 100005

int tree[4 * MAXN]; /* The segment tree array */
int arr[MAXN];     /* The original input array */
int n;             /* Size of input array */

/* Create (allocate) the segment tree */
void create_tree(int size) {
    n = size;
    /* Initialize all tree nodes to 0 */
    for (int i = 0; i < 4 * n; i++) {
        tree[i] = 0;
    }
    printf("Segment tree created for n=%d (array size: %d)\n", n, 4*n);
}

int main() {
    int data[] = {1, 3, 5, 7, 9, 11};
    int size = sizeof(data) / sizeof(data[0]);

    /* Copy data to our global array */
    for (int i = 0; i < size; i++) arr[i] = data[i];

    create_tree(size); /* Step 1: Create */

    return 0;
}

// Four build strategies

Recursive Build (Top-Down): Split the range in half, recursively build left and right subtrees, then merge. Most intuitive approach.

Iterative Build (Bottom-Up): Fill leaf nodes first (positions n to 2n-1), then compute internal nodes from right to left. Faster in practice.

Both approaches produce the same tree. Recursive is easier to understand; iterative is more cache-friendly.

  • Recursive: build(node, start, end) → split at mid
  • Iterative: leaves at [n..2n-1], parents at [1..n-1]
  • Both: O(n) time complexity

Click "Build Step" to start building the tree from arr=[2,4,6,8]
Cbuild_tree.c
#include <stdio.h>
#define MAXN 100005

int arr[MAXN], tree[4*MAXN];
int n;

/* ─── Approach 1: Recursive (Top-Down) Build ─── */
void build_recursive(int node, int start, int end) {
    if (start == end) {
        /* Leaf node: store the element */
        tree[node] = arr[start];
        return;
    }
    int mid = (start + end) / 2;
    int left  = 2 * node;
    int right = 2 * node + 1;

    build_recursive(left,  start, mid);   /* Build left subtree */
    build_recursive(right, mid+1, end);  /* Build right subtree */

    /* Internal node = sum of children */
    tree[node] = tree[left] + tree[right];
}

/* ─── Approach 2: Iterative (Bottom-Up) Build ─── */
void build_iterative() {
    /* Copy arr to leaves (positions n to 2n-1) */
    for (int i = 0; i < n; i++)
        tree[n + i] = arr[i];

    /* Build internal nodes bottom-up */
    for (int i = n - 1; i >= 1; i--)
        tree[i] = tree[2*i] + tree[2*i+1];
}

int main() {
    int data[] = {2, 4, 6, 8};
    n = 4;
    for (int i=0; i<n; i++) arr[i] = data[i];

    /* Method 1: recursive, root=1, range [0,n-1] */
    build_recursive(1, 0, n-1);
    printf("Recursive: tree[1]=%d\n", tree[1]); /* 20 */

    /* Method 2: iterative */
    build_iterative();
    printf("Iterative: tree[1]=%d\n", tree[1]); /* 20 */
    return 0;
}

// Visiting all nodes

Preorder (Root→Left→Right): Visit node, then left subtree, then right subtree.

Inorder (Left→Root→Right): Visit left subtree, then node, then right subtree. For segment trees this gives sorted leaf order.

Postorder (Left→Right→Root): Used to build and delete trees (process children before parent).

Level-Order (BFS): Visit level by level using a queue. In an array-based tree, this is simply tree[1], tree[2], tree[3], ...

Ctraversal.c
#include <stdio.h>
#define MAXN 100005

int tree[4*MAXN], n;

/* Preorder: Root → Left → Right */
void preorder(int node, int start, int end) {
    if (start > end) return;
    printf("%d ", tree[node]);          /* Visit root first */
    if (start == end) return;            /* Leaf node */
    int mid = (start + end) / 2;
    preorder(2*node,   start, mid);    /* Left  */
    preorder(2*node+1, mid+1, end); /* Right */
}

/* Inorder: Left → Root → Right */
void inorder(int node, int start, int end) {
    if (start == end) { printf("%d ", tree[node]); return; }
    int mid = (start + end) / 2;
    inorder(2*node,   start, mid);
    printf("%d ", tree[node]);          /* Visit root */
    inorder(2*node+1, mid+1, end);
}

/* Postorder: Left → Right → Root */
void postorder(int node, int start, int end) {
    if (start == end) { printf("%d ", tree[node]); return; }
    int mid = (start + end) / 2;
    postorder(2*node,   start, mid);
    postorder(2*node+1, mid+1, end);
    printf("%d ", tree[node]);          /* Visit root last */
}

/* Level-Order: array index order is already level-order! */
void level_order(int total_nodes) {
    for (int i = 1; i <= total_nodes; i++)
        printf("%d ", tree[i]);
}

Real-Life Analogy: Reading a Book

Preorder: Read the chapter title, then sub-sections left to right (table of contents style).
Inorder: Read deepest subsection first, then the chapter title (rare but useful for sorted data).
Postorder: Read all subsections before the chapter summary (like writing a book review — understand parts before summarizing).
Level-Order: Read all chapter titles first, then all section titles, then all subsections (breadth-first).

// Array index formulas (1-indexed)

  • Root: tree[1] — always index 1
  • Left Child of node i: tree[2*i]
  • Right Child of node i: tree[2*i+1]

These index relationships are why the array representation is so efficient — no pointers needed!

#include <stdio.h>

int tree[] = {0, 20, 6, 14, 2, 4, 6, 8}; /* index 0 unused */

int  get_root()              { return tree[1]; }
int  get_left_child(int i)  { return tree[2 * i]; }
int  get_right_child(int i) { return tree[2 * i + 1]; }

    printf("Right child[%d] : %d\n", 2*i+1, get_right_child(i));
    printf("Parent of [%d]  : %d\n", i,     get_parent(i));
    printf("Is leaf [%d]?   : %s\n", i,     is_leaf(i, n) ? "Yes" : "No");

// Query logic — three cases

When querying range [l, r] at a node covering [start, end]:

  • No Overlap: [start,end] and [l,r] don't intersect → return identity value (0 for sum, INT_MAX for min)
  • Full Overlap: [start,end] is completely within [l,r] → return tree[node]
  • Partial Overlap: Split and recurse on both children, merge results

Time complexity: O(log n) per query

Cquery.c
#include <stdio.h>
#include <limits.h>
#define MAXN 100005

int tree[4*MAXN], n;

/* ─── Range Sum Query ─── */
int query_sum(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return 0;          /* No overlap */
    if (l <= start && end <= r) return tree[node]; /* Full overlap */
    int mid = (start + end) / 2;
    int left  = query_sum(2*node,   start, mid,   l, r);
    int right = query_sum(2*node+1, mid+1, end, l, r);
    return left + right;                             /* Partial */
}

/* ─── Range Min Query ─── */
int query_min(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return INT_MAX;
    if (l <= start && end <= r) return tree[node];
    int mid = (start + end) / 2;
    int lv = query_min(2*node,   start, mid,   l, r);
    int rv = query_min(2*node+1, mid+1, end, l, r);
    return (lv < rv) ? lv : rv;
}

/* ─── Range Max Query ─── */
int query_max(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return INT_MIN;
    if (l <= start && end <= r) return tree[node];
    int mid = (start + end) / 2;
    int lv = query_max(2*node,   start, mid,   l, r);
    int rv = query_max(2*node+1, mid+1, end, l, r);
    return (lv > rv) ? lv : rv;
}

/* ─── Range XOR Query ─── */
int query_xor(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return 0;
    if (l <= start && end <= r) return tree[node];
    int mid = (start + end) / 2;
    return query_xor(2*node, start, mid, l, r) ^
           query_xor(2*node+1, mid+1, end, l, r);
}

/* ─── Point Query (single element) ─── */
int point_query(int node, int start, int end, int idx) {
    if (start == end) return tree[node];
    int mid = (start + end) / 2;
    if (idx <= mid) return point_query(2*node,   start, mid,   idx);
    else            return point_query(2*node+1, mid+1, end, idx);
}

/* ─── Full Range Query ─── */
int full_range_sum() { return tree[1]; } /* Root always has full sum */

int main() {
    /* Assume tree is already built for arr=[1,3,5,7,9,11] */
    printf("Sum [1..4]  = %d\n", query_sum(1,0,5,1,4));  /* 3+5+7+9=24 */
    printf("Min [0..3]  = %d\n", query_min(1,0,5,0,3));  /* 1 */
    printf("Max [2..5]  = %d\n", query_max(1,0,5,2,5));  /* 11 */
    printf("Point [3]   = %d\n", point_query(1,0,5,3)); /* 7 */
    printf("Full sum    = %d\n", full_range_sum());      /* 36 */
    return 0;
}

// How updates propagate

Point Update: Change one element at index idx. Traverse from root to that leaf, updating all ancestor nodes on the way. O(log n).

Range Update (naive): Update each point in range [l,r] individually. O(n log n) — slow. Use lazy propagation for O(log n).

Increment: Add delta to a position. Same as point update with arr[idx] += delta.
Decrement: Subtract delta. Same logic.
Replace: Set a value to exactly new_val. The difference (new_val - old_val) propagates up.

// point update demo — arr=[1,3,5,7,9,11]
Index: New Value:
Cupdate.c
#include <stdio.h>
#define MAXN 100005
int arr[MAXN], tree[4*MAXN], n;

/* ─── Point Update (replace value) ─── */
void update_point(int node, int start, int end, int idx, int new_val) {
    if (start == end) {
        arr[idx]  = new_val;      /* Update original array */
        tree[node] = new_val;     /* Update leaf */
        return;
    }
    int mid = (start + end) / 2;
    if (idx <= mid)
        update_point(2*node,   start, mid,   idx, new_val);
    else
        update_point(2*node+1, mid+1, end, idx, new_val);
    /* Recalculate parent from updated children */
    tree[node] = tree[2*node] + tree[2*node+1];
}

/* ─── Increment at index ─── */
void increment(int idx, int delta) {
    update_point(1, 0, n-1, idx, arr[idx] + delta);
}

/* ─── Decrement at index ─── */
void decrement(int idx, int delta) {
    update_point(1, 0, n-1, idx, arr[idx] - delta);
}

/* ─── Replace at index ─── */
void replace(int idx, int new_val) {
    update_point(1, 0, n-1, idx, new_val);
}

int main() {
    /* After building tree from arr=[1,3,5,7,9,11] */
    increment(2, 5);   /* arr[2]: 5 → 10 */
    decrement(4, 3);   /* arr[4]: 9 → 6  */
    replace(0, 100);   /* arr[0]: 1 → 100 */
    return 0;
}

// The problem with naive range updates

Updating every element in [l,r] by +5 naively takes O(n log n). Lazy propagation brings this down to O(log n) by deferring updates.

lazy[] array: parallel to tree[], stores pending updates not yet pushed to children.

Push Down: Before accessing children, flush the lazy value to them.

Lazy Range Update: Update the node itself + store the delta in lazy[node]. Don't touch children yet.

Lazy Range Query: Push lazy before recursing to ensure children have up-to-date values.

// lazy propagation — add +5 to range [1..3] of arr=[1,3,5,7,9]
tree[] (sum values)
lazy[] (pending adds)
Clazy_propagation.c
#include <stdio.h>
#include <string.h>
#define MAXN 100005

int tree[4*MAXN], lazy[4*MAXN], n;

/* Step 1: Initialize lazy array to 0 */
void init_lazy() {
    memset(lazy, 0, sizeof(lazy));
}

/* Step 2: Push lazy value to children (propagate) */
void push_down(int node, int start, int end) {
    if (lazy[node] != 0) {
        int mid = (start + end) / 2;
        int left_len  = mid - start + 1;
        int right_len = end - mid;

        /* Apply to left child */
        tree[2*node]    += lazy[node] * left_len;
        lazy[2*node]    += lazy[node];

        /* Apply to right child */
        tree[2*node+1] += lazy[node] * right_len;
        lazy[2*node+1] += lazy[node];

        lazy[node] = 0; /* Clear current node's lazy */
    }
}

/* Step 3: Lazy Range Update — add val to all [l..r] */
void lazy_update(int node, int start, int end, int l, int r, int val) {
    if (r < start || end < l) return;         /* No overlap */
    if (l <= start && end <= r) {              /* Full overlap */
        tree[node] += val * (end - start + 1); /* Update sum */
        lazy[node] += val;                      /* Mark for children */
        return;
    }
    push_down(node, start, end);               /* Partial: push first */
    int mid = (start + end) / 2;
    lazy_update(2*node,   start, mid,   l, r, val);
    lazy_update(2*node+1, mid+1, end, l, r, val);
    tree[node] = tree[2*node] + tree[2*node+1];
}

/* Step 4: Lazy Range Query */
int lazy_query(int node, int start, int end, int l, int r) {
    if (r < start || end < l) return 0;
    if (l <= start && end <= r) return tree[node];
    push_down(node, start, end); /* Must push before querying children */
    int mid = (start + end) / 2;
    return lazy_query(2*node, start, mid, l, r) +
           lazy_query(2*node+1, mid+1, end, l, r);
}

int main() {
    /* After building for arr=[1,3,5,7,9] */
    lazy_update(1, 0, 4, 1, 3, 5);  /* Add 5 to [1..3] */
    printf("Sum [0..4] = %d\n", lazy_query(1,0,4,0,4)); /* 1+8+10+12+9=40 */
    return 0;
}

// Searching within a segment tree

Search Element: Find if a value exists in a range. Traverse; compare leaves.

Find First/Last Occurrence: Walk left-first (for first) or right-first (for last) to find earliest/latest position of a target.

Find K-th Element: Walk down the tree; if left subtree sum ≥ k, go left. Else subtract left sum from k, go right.

Binary Search on Tree: Binary search over answers using the tree's range sum/min/max as a monotonic predicate.

// find K-th smallest element — arr as frequency array
K:
Csearch.c
#include <stdio.h>
#define MAXN 100005
int tree[4*MAXN], n;

/* ─── Search Element (returns index, -1 if not found) ─── */
int search_element(int node, int start, int end, int target) {
    if (start == end)
        return (tree[node] == target) ? start : -1;
    int mid = (start + end) / 2;
    int l = search_element(2*node,   start, mid,   target);
    if (l != -1) return l;
    return search_element(2*node+1, mid+1, end, target);
}

/* ─── Find K-th element (frequency segment tree) ─── */
/* Tree stores frequencies: tree[node] = count of elements in range */
int find_kth(int node, int start, int end, int k) {
    if (start == end) return start; /* Found the k-th element */
    int mid = (start + end) / 2;
    if (tree[2*node] >= k)
        return find_kth(2*node,   start, mid,   k);
    else
        return find_kth(2*node+1, mid+1, end, k - tree[2*node]);
}

/* ─── Find First Occurrence of value >= target ─── */
int find_first(int node, int start, int end, int l, int r, int target) {
    if (r < start || end < l || tree[node] < target) return -1;
    if (start == end) return start;
    int mid = (start + end) / 2;
    int left = find_first(2*node, start, mid, l, r, target);
    if (left != -1) return left;
    return find_first(2*node+1, mid+1, end, l, r, target);
}

/* ─── Binary Search on Tree ─── */
/* Find smallest index where prefix sum >= k */
int binary_search_tree(int node, int start, int end, int k) {
    if (start == end) return start;
    int mid = (start + end) / 2;
    if (tree[2*node] >= k)
        return binary_search_tree(2*node,   start, mid,   k);
    return binary_search_tree(2*node+1, mid+1, end, k - tree[2*node]);
}

A segment tree is defined by its merge function — how two children combine to produce their parent's value.

Merge Nodes: Combine two adjacent segment tree node values.
Merge Segments: Combine results from [l,mid] and [mid+1,r].
Combine Child Values: tree[node] = merge(tree[2*node], tree[2*node+1]).
Custom Merge: For complex queries (max subarray sum, etc.) you define a struct with multiple fields and a custom combiner.

/* ─── Simple merge functions ─── */
int merge_sum(int a, int b) { return a + b; }
int merge_min(int a, int b) { return (a < b) ? a : b; }
int merge_max(int a, int b) { return (a > b) ? a : b; }
int merge_xor(int a, int b) { return a ^ b; }
int merge_gcd(int a, int b) { return b ? merge_gcd(b, a%b) : a; }

/* ─── Custom merge: max subarray sum ─── */
typedef struct {
    int total;   /* total sum of segment */
    int prefix;  /* max prefix sum */
    int suffix;  /* max suffix sum */
    int answer;  /* max subarray sum in segment */
} Node;

Node custom_merge(Node L, Node R) {
    Node res;
    res.total   = L.total + R.total;
    res.prefix  = (L.prefix > L.total + R.prefix) ? L.prefix : L.total + R.prefix;
    res.suffix  = (R.suffix > R.total + L.suffix) ? R.suffix : R.total + L.suffix;
    int cross   = L.suffix + R.prefix;
    res.answer  = L.answer;
    if (R.answer  > res.answer) res.answer = R.answer;
    if (cross     > res.answer) res.answer = cross;
    return res;
}

/* Generic build using a function pointer */
typedef int (*MergeFn)(int, int);

void build_generic(int* tree, int* arr, int node,
                   int start, int end, MergeFn merge) {
    if (start == end) { tree[node] = arr[start]; return; }
    int mid = (start + end) / 2;
    build_generic(tree, arr, 2*node,   start, mid,   merge);
    build_generic(tree, arr, 2*node+1, mid+1, end, merge);
    tree[node] = merge(tree[2*node], tree[2*node+1]);
}

// How ranges are divided

Every segment tree operation internally splits ranges at the midpoint. Understanding this is key to understanding all recursive operations.

Split Segment: Divide [start, end] into [start, mid] and [mid+1, end] where mid = (start+end)/2.

Divide Range: Decompose a query range [l,r] into O(log n) canonical segments stored in the tree.

Recursive Split: The recursive descent pattern used by build, query, and update — always splitting until we reach a base case (start == end).

// visualize how range [0,7] is split — binary division
Csplit.c
#include <stdio.h>

/* ─── Divide a range into canonical segments ─── */
/* This shows how [l,r] is decomposed into O(log n) pieces */
void divide_range(int node, int start, int end, int l, int r) {
    if (r < start || end < l) {
        printf("  [%d..%d] SKIP (no overlap)\n", start, end);
        return;
    }
    if (l <= start && end <= r) {
        printf("  [%d..%d] USE (full overlap) → node[%d]\n", start, end, node);
        return;
    }
    int mid = (start + end) / 2;
    printf("  [%d..%d] SPLIT at mid=%d\n", start, end, mid);
    divide_range(2*node,   start, mid,   l, r);
    divide_range(2*node+1, mid+1, end, l, r);
}

int main() {
    printf("Dividing query [2..5] over tree [0..7]:\n");
    divide_range(1, 0, 7, 2, 5);
    return 0;
}

// Advanced query patterns

Prefix Sum Query: sum(0, r) = query from index 0 to r. Useful for cumulative values.

Suffix Sum Query: sum(l, n-1) = query from l to end.

Maximum Subarray Query: Uses a custom node struct storing {total, prefix, suffix, answer} and a custom merge. Returns max-sum contiguous subarray in O(log n).

Frequency Query: Build tree over values (not indices). tree[v] = count of v. Query count of values in [a,b].

Majority Element Query: Returns element with count > half the range length using frequency segment tree.

// max subarray sum — arr=[-2,1,-3,4,-1,2,1,-5,4]
Cadvanced.c
#include <stdio.h>
#define MAXN 100005

/* ─── Prefix Sum: sum[0..r] ─── */
int prefix_sum(int* tree, int r, int n) {
    /* Using iterative bottom-up tree */
    int sum = 0;
    for (r += n; r > 0; r >>= 1)
        sum += tree[r];
    return sum;
}

/* ─── Max Subarray (Kadane's on segment tree) ─── */
typedef struct { int total, pre, suf, ans; } MS;

int max2(int a, int b) { return a>b?a:b; }

MS make_leaf(int v) { return (MS){v, v, v, v}; }

MS combine(MS L, MS R) {
    MS res;
    res.total = L.total + R.total;
    res.pre   = max2(L.pre, L.total + R.pre);
    res.suf   = max2(R.suf, R.total + L.suf);
    res.ans   = max2(max2(L.ans, R.ans), L.suf + R.pre);
    return res;
}

MS seg[4*MAXN];

void build_ms(int* arr, int node, int s, int e) {
    if (s == e) { seg[node] = make_leaf(arr[s]); return; }
    int mid = (s+e)/2;
    build_ms(arr, 2*node, s, mid);
    build_ms(arr, 2*node+1, mid+1, e);
    seg[node] = combine(seg[2*node], seg[2*node+1]);
}

/* ─── Frequency Query ─── */
/* freq[v]++ for each v in arr; tree[v] = count of v */
int freq_tree[MAXN]; /* indexed by value */

void add_element(int v, int max_val) {
    for (v++; v <= max_val; v += v & (-v))
        freq_tree[v]++;  /* BIT update (simpler for freq queries) */
}

int main() {
    int arr[] = {-2,1,-3,4,-1,2,1,-5,4};
    int n = 9;
    build_ms(arr, 1, 0, n-1);
    printf("Max subarray sum = %d\n", seg[1].ans); /* 6: [4,-1,2,1] */
    return 0;
}

// Five powerful segment tree variants

Persistent Segment Tree: After each update, create a new version that shares unchanged nodes with the old version. O(log n) space per version. Allows answering historical queries.

Dynamic Segment Tree: Only creates nodes when actually accessed. Useful when value range is huge (e.g., 1 to 10⁹) but only few updates occur. Uses linked nodes with pointers.

2D Segment Tree: A segment tree over rows, where each node is itself a segment tree over columns. Supports 2D range queries in O(log² n).

Merge Sort Tree: Each node stores a sorted list of its segment's elements. Supports count-of-elements-less-than-x queries in O(log² n).

Segment Tree Beats (Ji Driver Segmentation): Efficiently handles range updates like "replace each a[i] with min(a[i], x)". Achieves amortized O(n log² n) instead of O(n log n) per operation.

Cvariants.c
#include <stdio.h>
#include <stdlib.h>
#define MAXNODES 4000000

/* ═══ 1. PERSISTENT SEGMENT TREE ═══ */
typedef struct {
    int left, right, val;  /* child indices in node pool */
} PNode;

PNode pool[MAXNODES];
int pool_cnt = 0;
int roots[MAXNODES]; /* root of each version */

int new_node() { return ++pool_cnt; }

int persistent_update(int prev, int s, int e, int idx, int val) {
    int cur = new_node();
    pool[cur] = pool[prev]; /* copy old node */
    if (s == e) { pool[cur].val += val; return cur; }
    int mid = (s+e)/2;
    if (idx <= mid)
        pool[cur].left  = persistent_update(pool[prev].left,  s, mid,   idx, val);
    else
        pool[cur].right = persistent_update(pool[prev].right, mid+1, e, idx, val);
    pool[cur].val = pool[pool[cur].left].val + pool[pool[cur].right].val;
    return cur;
}

/* ═══ 2. DYNAMIC SEGMENT TREE ═══ */
typedef struct DNode {
    int val, left, right; /* 0 = null */
} DNode;

DNode dpool[MAXNODES];
int dcnt = 0;

int dnew() { dcnt++; dpool[dcnt] = (DNode){0,0,0}; return dcnt; }

void dynamic_update(int node, long s, long e, long idx, int val) {
    if (s == e) { dpool[node].val += val; return; }
    long mid = (s+e)/2;
    if (idx <= mid) {
        if (!dpool[node].left) dpool[node].left = dnew();
        dynamic_update(dpool[node].left,  s,   mid, idx, val);
    } else {
        if (!dpool[node].right) dpool[node].right = dnew();
        dynamic_update(dpool[node].right, mid+1, e, idx, val);
    }
    dpool[node].val = dpool[dpool[node].left].val +
                      dpool[dpool[node].right].val;
}

/* ═══ 3. SEGMENT TREE BEATS (range chmin) ═══ */
typedef struct {
    int max1;  /* maximum value */
    int max2;  /* second maximum */
    int cntmax; /* count of max */
    long long sum; /* sum of segment */
} BNode;

/* Beats: if x >= max1, no change. if x > max2, update max1. */
void beats_chmin(BNode* b, int x) {
    if (x >= b->max1) return;
    b->sum   -= (long long)(b->max1 - x) * b->cntmax;
    b->max1   = x;
}

    printf("Segment tree variants initialized.\n");

    return 0;
}
