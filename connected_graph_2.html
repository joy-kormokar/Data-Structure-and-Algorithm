What is a Connected Graph?

A graph is a set of nodes (vertices) connected by edges. A connected graph means every vertex can reach every other vertex through some path.

Real-life analogy

Think of cities (vertices) connected by roads (edges). A connected graph means you can drive from any city to any other city — no city is isolated. Facebook friends, city maps, computer networks — all graphs!

Key concepts

  • Vertex (Node): A point in the graph. Example: a city, a person.
  • Edge: A connection between two vertices. Example: a road, a friendship.
  • Adjacency: Two vertices are adjacent if there is an edge between them.
  • Connected Graph: There is a path between every pair of vertices.
  • Directed vs Undirected: Directed edges have direction (one-way). Undirected are two-way.

Beginner Level

Adjacency List Representation

An adjacency list stores each vertex with a list of its neighbors. It's memory-efficient, especially for sparse graphs.

Real-life analogy

Imagine a phone book where each person has a list of their friends' numbers. Each person is a vertex; their friend list is the adjacency list.

How it works in C

In C, we use an array of linked lists. The array index represents the vertex. Each list node stores a neighbor vertex.

Adjacency list builder

C — Adjacency List Structure
/* Node in linked list */
struct Node {
    int vertex;
    struct Node* next;
};

/* Graph with N vertices */
struct Graph {
    int numVertices;
    struct Node** adjList; /* array of list heads */
};

/* Create a new graph */
struct Graph* createGraph(int V) {
    struct Graph* g = malloc(sizeof(struct Graph));
    g->numVertices = V;
    g->adjList = malloc(V * sizeof(struct Node*));
    for (int i = 0; i < V; i++)
        g->adjList[i] = NULL; /* empty lists */
    return g;
}

void insertEdge(struct Graph* g, int src, int dest) {

    struct Node* n = malloc(sizeof(struct Node));
    n->vertex = dest;

Delete Edge

C — Delete Edge
void deleteEdge(struct Graph* g, int src, int dest) {
    removeFromList(&g->adjList[src], dest);
    removeFromList(&g->adjList[dest], src); /* undirected */
}

void removeFromList(struct Node** head, int val) {
    struct Node *curr = *head, *prev = NULL;
    while (curr) {
        if (curr->vertex == val) {
            if (prev) prev->next = curr->next;
            else *head = curr->next;
            free(curr); return;
        }
        prev = curr; curr = curr->next;
    }
}

int countEdges(struct Graph* g) {
    int count = 0;
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* curr = g->adjList[i];
        while (curr) { count++; curr = curr->next; }
    }
    return count / 2; /* undirected: each edge counted twice */
}
Traversal

Breadth First Search (BFS)

BFS visits all neighbors of a vertex before moving deeper. It uses a queue and explores level by level.

Analogy

Like ripples in water — you visit the closest nodes first, then their neighbors, expanding outward ring by ring. Used in GPS shortest path, social network friend suggestions.

BFS animation

BFS Algorithm Steps

  • 1. Initialize a queue. Mark start vertex as visited. Enqueue it.
  • 2. While queue is not empty: dequeue a vertex, print it.
  • 3. For each unvisited neighbor: mark visited, enqueue it.
  • 4. Repeat until queue is empty.
C — BFS using Queue
void BFS(struct Graph* g, int start) {
    int visited[MAX] = {0};
    int queue[MAX], front = 0, rear = 0;

    visited[start] = 1;
    queue[rear++] = start;

    printf("BFS: ");
    while (front < rear) {
        int v = queue[front++]; /* dequeue */
        printf("%d ", v);

        struct Node* curr = g->adjList[v];
        while (curr) {
            if (!visited[curr->vertex]) {
                visited[curr->vertex] = 1;
                queue[rear++] = curr->vertex;
            }
            curr = curr->next;
        }
    }
}

Traversal

Depth First Search (DFS)

DFS dives as deep as possible along each branch before backtracking. Can be recursive or iterative using a stack.

Analogy

Like exploring a maze — go as far as possible, hit a dead end, backtrack, try another path. Used for topological sorting, detecting cycles, maze solving.

DFS animation

Recursive DFS

C — Recursive DFS
int visited[MAX];

void DFS(struct Graph* g, int v) {
    visited[v] = 1;
    printf("%d ", v);
    struct Node* curr = g->adjList[v];
    while (curr) {
        if (!visited[curr->vertex])
            DFS(g, curr->vertex); /* recurse */
        curr = curr->next;
    }
}

Iterative DFS (Stack)

C — Stack-based DFS
void DFS_Stack(struct Graph* g, int start) {
    int visited[MAX] = {0};
    int stack[MAX], top = -1;
    stack[++top] = start;

    while (top >= 0) {
        int v = stack[top--]; /* pop */
        if (!visited[v]) {
            visited[v] = 1;
            printf("%d ", v);
            struct Node* curr = g->adjList[v];
            while (curr) {
                if (!visited[curr->vertex])
                    stack[++top] = curr->vertex;
                curr = curr->next;
            }
        }
    }
}

int isConnected(struct Graph* g) {

Find Path (DFS-based)

C — Find Path
int findPath(struct Graph* g, int src, int dest,
             int* visited, int* path, int pLen) {
    visited[src] = 1;
    path[pLen++] = src;
    if (src == dest) {
        printPath(path, pLen); return 1;
    }
    struct Node* curr = g->adjList[src];
    while (curr) {
        if (!visited[curr->vertex])
            if (findPath(g,curr->vertex,dest,visited,path,pLen))
                return 1;
        curr = curr->next;
    }
    return 0; /* backtrack */
}

Shortest Path (BFS)

C — Shortest Path using BFS
int shortestPath(struct Graph* g, int src, int dest) {
    int dist[MAX], visited[MAX];
    int queue[MAX], front=0, rear=0;
    for(int i=0;i<MAX;i++) dist[i]=-1, visited[i]=0;
    visited[src]=1; dist[src]=0;
    queue[rear++]=src;
    while(front<rear) {
        int v=queue[front++];
        struct Node* c=g->adjList[v];
        while(c){
            if(!visited[c->vertex]){
                visited[c->vertex]=1;
                dist[c->vertex]=dist[v]+1;
                queue[rear++]=c->vertex;
            }
            c=c->next;
        }
    }
    return dist[dest]; /* -1 if unreachable */
}

Cycle Operations

Cycle Operations

Detect cycles, count them, check cyclic structure, and remove cycles from the graph.

Analogy

A cycle is like a circular road loop — you can keep driving and end up where you started. Detecting cycles is critical in dependency resolution and deadlock detection in operating systems.

Cycle detector

Detect Cycle (Undirected Graph)

C — Cycle Detection (DFS)
int hasCycleDFS(struct Graph* g, int v,
                int* visited, int parent) {
    visited[v] = 1;
    struct Node* curr = g->adjList[v];
    while (curr) {
        int u = curr->vertex;
        if (!visited[u]) {
            if (hasCycleDFS(g, u, visited, v))
                return 1;
        } else if (u != parent) {
            return 1; /* back edge found = cycle */
        }
        curr = curr->next;
    }
    return 0;
}

Cycle Detection (Directed Graph — using color)

C — Directed Cycle Detection
/* 0=white(unvisited) 1=gray(in progress) 2=black(done) */
int color[MAX];

int hasCycleDirected(struct Graph* g, int v) {
    color[v] = 1; /* gray */
    struct Node* curr = g->adjList[v];
    while (curr) {
        if (color[curr->vertex] == 1)
            return 1; /* back edge */
        if (color[curr->vertex] == 0)
            if (hasCycleDirected(g, curr->vertex))
                return 1;
        curr = curr->next;
    }
    color[v] = 2; /* black */
    return 0;
}

/* Undirected: count neighbors */
int findDegree(struct Graph* g, int v) {
    int deg = 0;
    struct Node* curr = g->adjList[v];
    while (curr) { deg++; curr = curr->next; }
    return deg;

/* Out-degree (directed): length of v's list */
int outDegree(struct Graph* g, int v) {
    return findDegree(g, v);
}

/* In-degree (directed): count v in all lists */
int inDegree(struct Graph* g, int v) {
    int deg = 0;
    for(int i=0;i<g->numVertices;i++) {
        struct Node* c = g->adjList[i];
        while(c){
            if(c->vertex==v) deg++;
            c=c->next;
        }
    }
    return deg;
}

Tree Validation

A tree is a connected graph with no cycles. For V vertices and E edges: a tree has exactly E = V - 1 edges.

C — Is Tree?
int isTree(struct Graph* g) {
    int V = g->numVertices;
    int E = countEdges(g);
    /* Tree: connected AND E == V-1 */
    if (E != V - 1) return 0;
    return isConnected(g);
}

Minimum Spanning Tree (MST)

Analogy

Connecting all cities with minimum total road length. MST gives the cheapest way to keep the entire graph connected — used in network cable routing, water pipes, electrical grids.

MST visualizer (Kruskal / Prim)

C — Kruskal's Algorithm (Union-Find)
int parent[MAX];
int find(int i) {
    if (parent[i]!=i) parent[i]=find(parent[i]);
    return parent[i];
}
void unite(int a, int b){parent[find(a)]=find(b);}

void kruskal(struct Edge edges[], int E, int V) {
    sortEdgesByWeight(edges, E);
    for(int i=0;i<V;i++) parent[i]=i;
    int mstCost=0, mstEdges=0;
    for(int i=0;i<E && mstEdges<V-1;i++) {
        if(find(edges[i].src)!=find(edges[i].dest)){
            unite(edges[i].src, edges[i].dest);
            mstCost += edges[i].weight;
            mstEdges++;
            printf("%d-%d (w=%d)\n",
                   edges[i].src,edges[i].dest,edges[i].weight);
        }
    }
}
C — Prim's Algorithm
void prim(int graph[MAX][MAX], int V) {
    int key[MAX], parent[MAX], inMST[MAX];
    for(int i=0;i<V;i++){key[i]=INT_MAX;inMST[i]=0;}
    key[0]=0; parent[0]=-1;
    for(int cnt=0;cnt<V-1;cnt++) {
        int u = minKey(key,inMST,V);
        inMST[u]=1;
        for(int v=0;v<V;v++)
            if(graph[u][v] && !inMST[v] && graph[u][v]<key[v]){
                parent[v]=u; key[v]=graph[u][v];
            }
    }
    /* print parent[] for MST edges */
}

Analogy

/* Step 1: DFS + push finish order to stack */
void dfsFinish(struct Graph* g, int v, int* vis, int* stk, int* top){
    vis[v]=1;
    struct Node* c=g->adjList[v];
    while(c){if(!vis[c->vertex])dfsFinish(g,c->vertex,vis,stk,top);c=c->next;}
    stk[++(*top)]=v; /* push on finish */
}

/* Step 2: Transpose graph */
/* Step 3: DFS on transposed in reverse finish order */
void findSCCs(struct Graph* g) {
    int vis[MAX]={0}, stk[MAX], top=-1;
    for(int i=0;i<g->numVertices;i++)
        if(!vis[i]) dfsFinish(g,i,vis,stk,&top);
    struct Graph* t=transpose(g);
    memset(vis,0,sizeof(vis));

        int v=stk[top--];

}

Graph Coloring

Analogy

Color a map so no two neighboring countries share the same color. Or schedule exams so no student has two exams at the same time. Used in register allocation in compilers, scheduling.

Graph coloring demo

C — Greedy Graph Coloring
void greedyColor(struct Graph* g) {
    int color[MAX], V=g->numVertices;
    for(int i=0;i<V;i++) color[i]=-1;
    color[0]=0;
    int used[MAX];
    for(int v=1;v<V;v++){
        for(int i=0;i<V;i++) used[i]=0;
        struct Node* c=g->adjList[v];
        while(c){
            if(color[c->vertex]!=-1)
                used[color[c->vertex]]=1;
            c=c->next;
        }
        int col=0;
        while(used[col]) col++;
        color[v]=col;
    }
    for(int i=0;i<V;i++)
        printf("Vertex %d → Color %d\n",i,color[i]);
}
