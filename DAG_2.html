Introduction to DAG
What is a Directed Acyclic Graph and why does it matter?

📖 Theory

A Directed Acyclic Graph (DAG) is a graph with three key properties:

  • Vertices (nodes) — data points labeled A, B, C …
  • Directed edges — each edge A→B means "A points to B" (one-way)
  • No cycles — you can NEVER start at a node and follow edges back to the same node

DAGs model any situation where things must happen in a specific order with no circular dependencies.

🔵 DAG Structure — Source → Middle → Sink

✅ DAG Properties

  • Directed edges (one-way only)
  • No cycles ever
  • Has at least one topological order
  • Source nodes: in-degree = 0
  • Sink nodes: out-degree = 0

Representation — Directed Adjacency List
How to store a DAG in memory using C structures

📖 Theory

An Adjacency List stores for each vertex a linked list of all vertices it points to. Space: O(V+E) — much better than a matrix for sparse graphs.

Each vertex stores: its ID, name, and a pointer to its neighbor linked list.

📋 Adjacency List — Graph: A→B, A→C, B→D, C→D
dag_structure.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX_V 100
#define MAX_NAME 50

/* ─── Node in the adjacency linked list ─── */
typedef struct AdjNode {
    int dest;              /* index of destination vertex */
    int weight;            /* edge weight (0 if unweighted) */
    struct AdjNode* next;  /* pointer to next neighbor */
} AdjNode;

/* ─── Each vertex in the graph ─── */
typedef struct {
    char     name[MAX_NAME]; /* label */
    AdjNode* head;           /* head of neighbor list */
    int      active;         /* 1=exists, 0=deleted */
} Vertex;

/* ─── The Graph ─── */
typedef struct {
    Vertex v[MAX_V];  /* array of vertices */
    int    count;     /* total vertices */
} DAG;

/* Create empty DAG */
DAG* createDAG() {
    DAG* dag = (DAG*)malloc(sizeof(DAG));
    dag->count = 0;
    return dag;
}

/* Create new adjacency node */
AdjNode* newNode(int dest, int w) {
    AdjNode* n = (AdjNode*)malloc(sizeof(AdjNode));
    n->dest = dest; n->weight = w; n->next = NULL;
    return n;
}

Adjacency List ✓O(V+E)O(deg)Sparse graphs
Adjacency MatrixO(V²)O(1)Dense graphs

🌍 Real-Life Analogy

Phone contacts: Each person (vertex) has a list of people they call. Calls only go one way. Adjacency list = each person's outgoing call list.

Insert Vertex
Add a new node to the graph — O(1)

📖 Theory

Inserting a vertex adds a new entry to the vertices array with an empty neighbor list (head=NULL). Just increment the count. Time: O(1), Space: O(1).

▶ Insert Vertex Visualization
insert_vertex.c
/* Insert vertex — returns its index, or -1 if full */
int insertVertex(DAG* dag, const char* name) {
    if (dag->count >= MAX_V) { printf("Graph full!\n"); return -1; }
    int idx = dag->count;
    strcpy(dag->v[idx].name, name);
    dag->v[idx].head   = NULL;  /* empty neighbor list */
    dag->v[idx].active = 1;
    dag->count++;
    printf("Inserted '%s' at index %d\n", name, idx);
    return idx;
}
/* Time: O(1)  Space: O(1) */

🌍 Real-Life Analogy

Delete Vertex
Remove a node and all its connected edges — O(V+E)

📖 Theory

Deleting a vertex requires: (1) free its outgoing neighbor list, (2) scan all other vertices and remove any edge pointing TO it, (3) mark it inactive. This is O(V+E).

delete_vertex.c
/* Free a linked list of AdjNodes */
void freeList(AdjNode* h) {
    while(h){ AdjNode* t=h; h=h->next; free(t); }
}

/* Remove all edges pointing INTO vertex idx */
void removeIncoming(DAG* dag, int idx) {
    for(int i=0;i<dag->count;i++) {
        AdjNode** cur = &dag->v[i].head;
        while(*cur) {
            if((*cur)->dest==idx){ AdjNode* d=*cur; *cur=d->next; free(d); }
            else cur=&(*cur)->next;
        }
    }
}

/* Delete vertex idx */
void deleteVertex(DAG* dag, int idx) {
    if(idx<0||idx>=dag->count||!dag->v[idx].active){
        printf("Vertex not found!\n"); return;
    }
    freeList(dag->v[idx].head);
    dag->v[idx].head   = NULL;
    dag->v[idx].active = 0;         /* mark deleted */
    removeIncoming(dag, idx);
    printf("Vertex %d deleted\n", idx);
}
/* Time: O(V+E) */

🌍 Real-Life Analogy

Dropping a course: Remove the course AND remove it from every other course's prerequisites list.

🔗
Insert Directed Edge
Add a one-way connection from src → dest — O(1)

📖 Theory

Inserting an edge adds a new AdjNode at the front of src's neighbor list. We check for self-loops, duplicates, and cycle creation before inserting.

▶ Insert Edge Demo
insert_edge.c
int insertEdge(DAG* dag, int src, int dest, int w) {
    if(src==dest){ printf("No self-loops!\n"); return -1; }
    if(!dag->v[src].active||!dag->v[dest].active){
        printf("Invalid vertex!\n"); return -1;
    }
    /* Check duplicate */
    AdjNode* c=dag->v[src].head;
    while(c){ if(c->dest==dest){ printf("Edge exists!\n"); return -1; } c=c->next; }
    /* Insert at front — O(1) */
    AdjNode* node  = newNode(dest, w);
    node->next     = dag->v[src].head;
    dag->v[src].head = node;
    printf("Edge %d→%d (w=%d) inserted\n",src,dest,w);
    return 0;
}
/* Time: O(degree) for duplicate check, O(1) for actual insert */

🌍 Real-Life Analogy

One-way street: Building a road from City A to City B. Traffic flows only A→B. You can't use it to drive from B back to A.

Validate Acyclic Condition
Ensure adding an edge won't create a cycle — O(V+E)

📖 Theory

Before inserting edge src→dest, check: "Can I reach src starting from dest?" If YES, adding this edge creates a cycle — REJECT it.

🚫 Cycle Prevention — C→A would create cycle X→Y→Z→X
validate_acyclic.c
/* Can we reach 'target' from 'start'? */
int reaches(DAG* dag,int s,int t,int vis[]){
    if(s==t) return 1;
    vis[s]=1;
    AdjNode* c=dag->v[s].head;
    while(c){ if(!vis[c->dest]&&reaches(dag,c->dest,t,vis)) return 1; c=c->next; }
    return 0;
}

/* Safe insert — checks acyclic before inserting */
int safeInsert(DAG* dag,int src,int dest,int w){
    int vis[MAX_V]={0};
    if(reaches(dag,dest,src,vis)){
        printf("BLOCKED: %d→%d creates cycle!\n",src,dest);
        return -1;
    }
    return insertEdge(dag,src,dest,w);
}
🌊
BFS — Breadth-First Search
Explore level by level using a queue — O(V+E)

📖 Theory

BFS visits all neighbors at distance 1 first, then distance 2, etc. Uses a queue (FIFO). Good for shortest hop-count paths and level-order traversal of the DAG.

▶ BFS Animation — A→B,A→C,B→D,C→D,D→E
bfs.c
void BFS(DAG* dag, int src) {
    int queue[MAX_V], front=0, rear=0;
    int vis[MAX_V]; memset(vis,0,sizeof(vis));
    queue[rear++]=src; vis[src]=1;
    printf("BFS: ");
    while(front<rear){
        int v=queue[front++];             /* dequeue */
        printf("%s ",dag->v[v].name);
        AdjNode* c=dag->v[v].head;
        while(c){
            if(!vis[c->dest]){ vis[c->dest]=1; queue[rear++]=c->dest; }
            c=c->next;
        }
    }
    printf("\n");
}
/* Output: BFS: A B C D E F */

🌍 Real-Life Analogy

Spreading news: You tell 2 friends (Level 1), each tells 2 more (Level 2). BFS = how information spreads level by level.

DFS — Depth-First Search
Explore as deep as possible before backtracking — O(V+E)

📖 Theory

DFS dives as deep as possible down one path before backing up. Uses a stack or recursion. Foundation for topological sort, cycle detection, and path finding.

🌍 Real-Life Analogy

Exploring a maze: Walk one corridor to its end. Dead end? Backtrack to last junction and try another. That's DFS.

Topological Traversal
Visit vertices in dependency order — O(V+E)

📖 Theory

Topological traversal ensures every vertex is visited ONLY AFTER all its predecessors. Only possible in a DAG (no cycles). Uses post-order DFS pushed onto a stack.

topo_traversal.c
void topoHelper(DAG* dag,int v,int vis[],int stk[],int* top){
    vis[v]=1;
    AdjNode* c=dag->v[v].head;
    while(c){ if(!vis[c->dest]) topoHelper(dag,c->dest,vis,stk,top); c=c->next; }
    stk[(*top)++]=v;   /* push AFTER all children processed */
}
void topoTraversal(DAG* dag){
    int vis[MAX_V]={0},stk[MAX_V],top=0;
    for(int i=0;i<dag->count;i++) if(dag->v[i].active&&!vis[i]) topoHelper(dag,i,vis,stk,&top);
    printf("Topo order: ");
    for(int i=top-1;i>=0;i--) printf("%s ",dag->v[stk[i]].name);
    printf("\n");
}

🌍 Real-Life Analogy

Getting dressed: Socks before shoes, underwear before pants. Topological traversal finds the correct order where no rule is violated.

Shortest Path in DAG
Topological sort + edge relaxation — O(V+E)

📖 Theory

Faster than Dijkstra for DAGs! Steps: (1) topologically sort, (2) relax edges in topo order. Works with negative weights too (no negative cycles in a DAG).

Relaxation: if dist[dest] > dist[u] + weight, update dist[dest].

⚡ Shortest Path — Gold = optimal path
shortest_path_dag.c
#define INF 999999
void topoSort(DAG* dag,int v,int vis[],int stk[],int* top){
    vis[v]=1;
    AdjNode* c=dag->v[v].head;
    while(c){if(!vis[c->dest])topoSort(dag,c->dest,vis,stk,top);c=c->next;}
    stk[(*top)++]=v;
}
void shortestPath(DAG* dag,int src){
    int vis[MAX_V]={0},stk[MAX_V],top=0,dist[MAX_V];
    for(int i=0;i<dag->count;i++) dist[i]=INF;
    dist[src]=0;
    for(int i=0;i<dag->count;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    while(top>0){
        int u=stk[--top];
        if(dist[u]==INF) continue;
        AdjNode* c=dag->v[u].head;
        while(c){
            if(dist[u]+c->weight < dist[c->dest]) dist[c->dest]=dist[u]+c->weight;
            c=c->next;
        }
    }
    for(int i=0;i<dag->count;i++)
        printf("Dist to %s = %d\n",dag->v[i].name,dist[i]==INF?-1:dist[i]);
}
🏔️
Longest Path in DAG
Maximize instead of minimize — O(V+E)

📖 Theory

NP-hard in general graphs, but O(V+E) in DAGs! Initialize distances to -INF and maximize during relaxation (instead of minimize). Used in project scheduling to find the critical path.

longest_path_dag.c
#define NEG_INF -999999
void longestPath(DAG* dag,int src){
    int vis[MAX_V]={0},stk[MAX_V],top=0,dist[MAX_V];
    for(int i=0;i<dag->count;i++) dist[i]=NEG_INF;
    dist[src]=0;
    for(int i=0;i<dag->count;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    while(top>0){
        int u=stk[--top];
        if(dist[u]==NEG_INF) continue;
        AdjNode* c=dag->v[u].head;
        while(c){
            if(dist[u]+c->weight > dist[c->dest])  /* MAXIMIZE */
                dist[c->dest]=dist[u]+c->weight;
            c=c->next;
        }
    }
    for(int i=0;i<dag->count;i++)
        printf("Longest to %s = %d\n",dag->v[i].name,dist[i]==NEG_INF?-1:dist[i]);
}

🌍 Real-Life Analogy

Construction timeline: The longest sequence of tasks determines minimum project duration — you can't finish before the longest chain completes.

🧮
Topological Sort — Kahn's Algorithm
BFS-based using in-degree counts — O(V+E)

📖 Theory

Repeatedly remove vertices with in-degree 0. After removing one, decrement neighbors' in-degrees. If a neighbor's in-degree hits 0, enqueue it. Bonus: detects cycles if not all vertices are processed.

🧮 Kahn's Algorithm Step-by-Step
Queue: [ A ]
In-degrees: A=0 B=1 C=1 D=2 E=1 F=2
Result: [ ]
Step 0: Add all in-degree=0 vertices to queue
topo_sort_kahn.c
void topoSortKahn(DAG* dag){
    int inDeg[MAX_V]={0},q[MAX_V],f=0,r=0,res[MAX_V],rc=0;
    /* Step 1: Compute in-degrees */
    for(int i=0;i<dag->count;i++){ AdjNode* c=dag->v[i].head; while(c){inDeg[c->dest]++;c=c->next;} }
    /* Step 2: Enqueue in-degree 0 vertices */
    for(int i=0;i<dag->count;i++) if(dag->v[i].active&&inDeg[i]==0) q[r++]=i;
    /* Step 3: Process */
    while(f<r){
        int u=q[f++]; res[rc++]=u;
        AdjNode* c=dag->v[u].head;
        while(c){ if(--inDeg[c->dest]==0) q[r++]=c->dest; c=c->next; }
    }
    /* Step 4: Cycle check */
    int active=0; for(int i=0;i<dag->count;i++) if(dag->v[i].active) active++;
    if(rc!=active){ printf("CYCLE DETECTED!\n"); return; }
    printf("Kahn Topo: ");
    for(int i=0;i<rc;i++) printf("%s ",dag->v[res[i]].name);
    printf("\n");
}

🌍 Real-Life Analogy

Cafeteria queue: Only students with no unpaid fees (in-degree 0) go first. After they leave, some others' requirements clear — they join the queue next.

Task Scheduling Order

void scheduleTasks(DAG* dag){
    int inDeg[MAX_V]={0},q[MAX_V],f=0,r=0,step=1;
    for(int i=0;i<dag->count;i++){ AdjNode* c=dag->v[i].head; while(c){inDeg[c->dest]++;c=c->next;} }
    for(int i=0;i<dag->count;i++) if(dag->v[i].active&&inDeg[i]==0) q[r++]=i;
    printf("=== Task Execution Schedule ===\n");
    while(f<r){
        int u=q[f++];
        printf("Step %d: Execute '%s'\n",step++,dag->v[u].name);
        AdjNode* c=dag->v[u].head;
        while(c){ if(--inDeg[c->dest]==0) q[r++]=c->dest; c=c->next; }
    }
}

   Step 1: Execute 'CompileLib'
   Step 2: Execute 'CompileApp'
   Step 3: Execute 'LinkBinary'  */

Cycle Detection
3-color DFS: WHITE / GRAY / BLACK — O(V+E)

📖 Theory — 3-Color DFS

  • WHITE unvisited
  • GRAY in current DFS path (being processed)
  • BLACK fully processed

If DFS reaches a GRAY vertex → we found a back-edge → CYCLE detected!

cycle_detection.c
#define WHITE 0
#define GRAY  1  /* in current path */
#define BLACK 2  /* fully done */

int cycleHelper(DAG* dag,int v,int color[]){
    color[v]=GRAY;
    AdjNode* c=dag->v[v].head;
    while(c){
        int d=c->dest;
        if(color[d]==GRAY){ printf("Cycle: %s→%s (back edge)\n",dag->v[v].name,dag->v[d].name); return 1; }
        if(color[d]==WHITE&&cycleHelper(dag,d,color)) return 1;
        c=c->next;
    }
    color[v]=BLACK; return 0;
}
int detectCycle(DAG* dag){
    int color[MAX_V]; memset(color,WHITE,sizeof(color));
    for(int i=0;i<dag->count;i++)
        if(dag->v[i].active&&color[i]==WHITE&&cycleHelper(dag,i,color)) return 1;
    printf("No cycle. Valid DAG.\n"); return 0;
}

/* Transitive closure — all pairs O(V³) */
void transitiveClosure(DAG* dag,int R[][MAX_V]){
    int n=dag->count;
    for(int i=0;i<n;i++){R[i][i]=1; AdjNode* c=dag->v[i].head; while(c){R[i][c->dest]=1;c=c->next;}}
    for(int k=0;k<n;k++) for(int i=0;i<n;i++) for(int j=0;j<n;j++)
        if(R[i][k]&&R[k][j]) R[i][j]=1;
}
/* R[i][j]==1 means j is reachable from i */

Critical Path Analysis
Find the bottleneck path determining project duration

📖 Theory

The critical path is the longest duration path. Tasks on it have zero float — any delay delays the entire project. Computed with a forward pass (earliest start times).

critical_path.c
void criticalPath(DAG* dag,int dur[]){
    int vis[MAX_V]={0},stk[MAX_V],top=0;
    int es[MAX_V]; memset(es,0,sizeof(es));
    for(int i=0;i<dag->count;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    /* Forward pass */
    while(top>0){
        int u=stk[--top]; int ef=es[u]+dur[u];
        AdjNode* c=dag->v[u].head;
        while(c){ if(ef>es[c->dest]) es[c->dest]=ef; c=c->next; }
    }
    int maxT=0;
    for(int i=0;i<dag->count;i++) if(es[i]+dur[i]>maxT) maxT=es[i]+dur[i];
    printf("Project duration: %d\n",maxT);
    printf("Critical tasks: ");
    for(int i=0;i<dag->count;i++) if(es[i]+dur[i]==maxT) printf("%s ",dag->v[i].name);
    printf("\n");
}

🌍 Real-Life Analogy

Construction: Foundation→Framing→Roofing is the critical path. Even if painting finishes early, the house isn't done until roofing completes.

🧠
Dynamic Programming on DAG
Apply DP in topological order — no subproblem computed twice

📖 Theory

DAGs are perfect for DP: no cycles means each subproblem is solved once. Process in topological order — each vertex's DP value depends only on already-computed predecessors.

  • Count paths: dp[v] += dp[u] for each edge u→v
  • Max/min path: dp[v] = max/min(dp[u] + weight)
dp_dag.c
/* Count number of paths from src to every vertex */
void countPaths(DAG* dag,int src){
    int vis[MAX_V]={0},stk[MAX_V],top=0;
    long dp[MAX_V]; memset(dp,0,sizeof(dp));
    dp[src]=1;   /* 1 way to reach source */
    for(int i=0;i<dag->count;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    while(top>0){
        int u=stk[--top];
        AdjNode* c=dag->v[u].head;
        while(c){
            dp[c->dest]+=dp[u];  /* DP recurrence */
            c=c->next;
        }
    }
    printf("Paths from %s:\n",dag->v[src].name);
    for(int i=0;i<dag->count;i++) printf("  To %s: %ld\n",dag->v[i].name,dp[i]);
}
/* Output: To A:1 To B:1 To C:1 To D:2 To E:1 To F:3 */

🌍 Real-Life Analogy

Counting routes: "In how many ways can a package travel from warehouse A to store F?" DP counts all routes without re-exploring.

Shortest Path — Advanced
Handles negative weights, O(V+E) vs Dijkstra's O((V+E)logV)

📖 Theory

DAG shortest path beats Dijkstra: it handles negative weights (no negative cycles in a DAG) and runs in O(V+E). Also reconstructs the actual path using a predecessor array.

adv_shortest.c
void shortestPathAdv(DAG* dag,int src){
    int vis[MAX_V]={0},stk[MAX_V],top=0;
    int dist[MAX_V],prev[MAX_V];
    for(int i=0;i<dag->count;i++){dist[i]=INF;prev[i]=-1;}
    dist[src]=0;
    for(int i=0;i<dag->count;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    while(top>0){
        int u=stk[--top]; if(dist[u]==INF) continue;
        AdjNode* c=dag->v[u].head;
        while(c){
            if(dist[u]+c->weight<dist[c->dest]){
                dist[c->dest]=dist[u]+c->weight;
                prev[c->dest]=u;  /* track path */
            }
            c=c->next;
        }
    }
    for(int i=0;i<dag->count;i++) printf("  %s: dist=%d\n",dag->v[i].name,dist[i]==INF?-1:dist[i]);
}

📊
Critical Path Method (CPM)
Calculate ES, EF, LS, LF and Float for project tasks

📖 Theory — CPM Values

  • ES Earliest Start = max(EF of all predecessors)
  • EF Earliest Finish = ES + duration
  • LF Latest Finish = min(LS of all successors)
  • LS Latest Start = LF − duration
  • Float = LS − ES (zero float = critical task)
cpm.c
void computeCPM(DAG* dag,int dur[]){
    int n=dag->count,vis[MAX_V]={0},stk[MAX_V],tmp[MAX_V],top=0;
    int es[MAX_V]={0},ef[MAX_V],ls[MAX_V],lf[MAX_V];
    for(int i=0;i<n;i++){ls[i]=lf[i]=INF;}
    for(int i=0;i<n;i++) if(!vis[i]) topoSort(dag,i,vis,stk,&top);
    for(int i=0;i<top;i++) tmp[i]=stk[i];
    /* Forward pass */
    while(top>0){
        int u=stk[--top]; ef[u]=es[u]+dur[u];
        AdjNode* c=dag->v[u].head;
        while(c){if(ef[u]>es[c->dest])es[c->dest]=ef[u];c=c->next;}
    }
    int proj=0; for(int i=0;i<n;i++) if(ef[i]>proj)proj=ef[i];
    for(int i=0;i<n;i++) if(ef[i]==proj)lf[i]=proj;
    /* Backward pass */
    for(int i=0;i<top;i++){
        int u=tmp[i]; if(lf[u]==INF)lf[u]=proj; ls[u]=lf[u]-dur[u];
        AdjNode* c=dag->v[u].head;
        while(c){if(ls[u]<lf[c->dest])lf[c->dest]=ls[u];c=c->next;}
    }
    printf("Task     ES  EF  LS  LF Float  Critical\n");
    for(int i=0;i<n;i++){
        int fl=ls[i]-es[i];
        printf("%-8s %3d %3d %3d %3d   %3d  %s\n",
               dag->v[i].name,es[i],ef[i],ls[i]==INF?-1:ls[i],lf[i]==INF?-1:lf[i],
               fl<0?0:fl,(fl<=0)?"*** YES ***":"no");
    }
    printf("Project Duration: %d\n",proj);
}

            printf("  [t=%d..%d] %-12s",time,time+durations[u],dag->v[u].name);
            for(int d=0;d<durations[u];d++) printf("█");
            printf("\n");
            if(durations[u]>maxDur) maxDur=durations[u];
            AdjNode* c=dag->v[u].head;
            while(c){ if(--inDeg[c->dest]==0) q[r++]=c->dest; c=c->next; }
        }
        time+=maxDur;
    }
    printf("Total time: %d\n",time);
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX_V 10
#define INF   99999

typedef struct Node { int dest,w; struct Node* next; } Node;
typedef struct { char name[20]; Node* head; } Vtx;
typedef struct { Vtx v[MAX_V]; int cnt; } DAG;

DAG dag;

void addV(const char* n){ strcpy(dag.v[dag.cnt].name,n); dag.v[dag.cnt].head=NULL; dag.cnt++; }
void addE(int s,int d,int w){ Node* n=(Node*)malloc(sizeof(Node)); n->dest=d;n->w=w;n->next=dag.v[s].head;dag.v[s].head=n; }

void topo(int v,int vis[],int stk[],int* top){
    vis[v]=1; Node* c=dag.v[v].head;
    while(c){if(!vis[c->dest])topo(c->dest,vis,stk,top);c=c->next;} stk[(*top)++]=v;
}

void display(){
    printf("\n=== DAG ===\n");
    for(int i=0;i<dag.cnt;i++){
        printf("%s: ",dag.v[i].name); Node* c=dag.v[i].head;
        while(c){printf("%s(%d) ",dag.v[c->dest].name,c->w);c=c->next;} printf("\n");
    }
}

void bfs(int src){
    int q[MAX_V],f=0,r=0,vis[MAX_V]={0};
    q[r++]=src; vis[src]=1; printf("BFS: ");
    while(f<r){ int v=q[f++]; printf("%s ",dag.v[v].name); Node* c=dag.v[v].head; while(c){if(!vis[c->dest]){vis[c->dest]=1;q[r++]=c->dest;}c=c->next;} } printf("\n");
}

void shortestPath(int src){
    int vis[MAX_V]={0},stk[MAX_V],top=0,dist[MAX_V];
    for(int i=0;i<dag.cnt;i++)dist[i]=INF; dist[src]=0;
    for(int i=0;i<dag.cnt;i++)if(!vis[i])topo(i,vis,stk,&top);
    while(top>0){ int u=stk[--top]; if(dist[u]==INF)continue; Node* c=dag.v[u].head; while(c){if(dist[u]+c->w<dist[c->dest])dist[c->dest]=dist[u]+c->w;c=c->next;} }
    printf("Shortest from %s:\n",dag.v[src].name);
    for(int i=0;i<dag.cnt;i++) printf("  %s=%d\n",dag.v[i].name,dist[i]==INF?-1:dist[i]);
}

int main(){
    memset(&dag,0,sizeof(dag));
    addV("A"); addV("B"); addV("C"); addV("D"); addV("E");
    addE(0,1,4); addE(0,2,2); addE(1,3,1); addE(2,3,5); addE(3,4,2);
    display();
    bfs(0);
    shortestPath(0);
    return 0;
}
