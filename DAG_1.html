What is a DAG?

A Directed Acyclic Graph (DAG) has two rules:
1. Every edge has a direction — A→B means A points to B, not the other way around.
2. No cycles — following edges, you can never return to where you started.

INTRO — DAG OVERVIEW
Click ▶ Animate to see how a DAG works!
Directed

Edges have arrows. A→B ≠ B→A. Dependencies flow one way only.

Acyclic

No loops. Starting at any node, following edges never brings you back.

Graph

Vertices (nodes) connected by edges. The foundation of networks.

🌍 Real-Life Analogy

Think of a cooking recipe: boil water → add pasta → drain → serve. Each step depends on the previous one, but you never go backwards. That's a DAG — ordered tasks with no circular waiting!

We store a DAG in a 2D array. matrix[i][j] = 1 means a directed edge from vertex i to vertex j exists.

C — DAG STRUCT + INIT + PRINT
#include <stdio.h>
#include <string.h>
#define MAX 10

/* DAG using adjacency matrix */
struct DAG {
    int matrix[MAX][MAX]; /* 0 = no edge, 1 = edge */
    int V;               /* number of vertices    */
};

/* Initialize: zero out the matrix */
void initDAG(struct DAG *g, int v) {
    g->V = v;
    for (int i = 0; i < MAX; i++)
        for (int j = 0; j < MAX; j++)
            g->matrix[i][j] = 0;
}

/* Print the adjacency matrix */
void printMatrix(struct DAG *g) {
    int v = g->V;
    printf("   ");
    for (int i = 0; i < v; i++) printf("%2d ", i);
    printf("\n");
    for (int i = 0; i < v; i++) {
        printf("%2d ", i);
        for (int j = 0; j < v; j++)
            printf("%2d ", g->matrix[i][j]);
        printf("\n");
    }
}

int main() {
    struct DAG g;
    initDAG(&g, 5);
    /* Add edges: 0→1, 0→2, 1→3, 2→3, 3→4 */
    g.matrix[0][1] = 1;  g.matrix[0][2] = 1;
    g.matrix[1][3] = 1;  g.matrix[2][3] = 1;
    g.matrix[3][4] = 1;
    printMatrix(&g);
    return 0;
}
C — WEIGHTED DAG (weights in matrix)
#define NO_EDGE -1   /* -1 means no edge */

void initWeighted(struct DAG *g, int v) {
    g->V = v;
    for (int i = 0; i < v; i++)
        for (int j = 0; j < v; j++)
            g->matrix[i][j] = NO_EDGE;
}

/* Add directed weighted edge u→v with weight w */
void addWeightedEdge(struct DAG *g, int u, int v, int w) {
    g->matrix[u][v] = w;   /* directed: only [u][v] */
    /* matrix[v][u] remains NO_EDGE */
}

🔵 Module 02

Vertex Operations

Insert, Delete, Search, and Update vertices — the nodes of the graph.

VERTEX OPERATIONS DEMO
Use the controls to perform vertex operations on the DAG.

Insert: Increment vertex count. The new row/column in the matrix is already 0 from initialization. No edges connect to the new vertex yet. Time: O(1).

C — INSERT VERTEX
/* Returns new vertex ID, or -1 if graph is full */
int insertVertex(struct DAG *g) {
    if (g->V >= MAX) {
        printf("Graph full! Cannot add vertex.\n");
        return -1;
    }
    int newID = g->V;
    /* New row & column already 0 from initDAG() */
    g->V++;
    printf("Inserted vertex %d\n", newID);
    return newID; /* O(1) */
}

Delete: Zero out the vertex's entire row AND column (removes all edges). Then shift remaining rows/columns. Time: O(V²).

C — DELETE VERTEX
void deleteVertex(struct DAG *g, int v) {
    int n = g->V;
    if (v < 0 || v >= n) { printf("Invalid!\n"); return; }
    /* Remove row v (all outgoing edges) */
    for (int j = 0; j < n; j++) g->matrix[v][j] = 0;
    /* Remove col v (all incoming edges) */
    for (int i = 0; i < n; i++) g->matrix[i][v] = 0;
    /* Shift rows up */
    for (int i = v; i < n-1; i++)
        for (int j = 0; j < n; j++)
            g->matrix[i][j] = g->matrix[i+1][j];
    /* Shift cols left */
    for (int j = v; j < n-1; j++)
        for (int i = 0; i < n-1; i++)
            g->matrix[i][j] = g->matrix[i][j+1];
    g->V--;
    printf("Deleted vertex %d, edges removed.\n", v);
}

Search: Check if vertex ID is in range [0, V). Instant O(1) check — just compare against vertex count.

C — SEARCH VERTEX
/* Returns 1 if vertex exists, 0 otherwise */
int searchVertex(struct DAG *g, int v) {
    if (v >= 0 && v < g->V) {
        printf("Vertex %d EXISTS\n", v);
        return 1; /* O(1) */
    }
    printf("Vertex %d NOT FOUND\n", v);
    return 0;
}

Update: Vertex data (label/value) lives in a separate array. Update that array. The matrix structure doesn't change. Time: O(1).

C — UPDATE VERTEX LABEL
#define NLEN 20
struct DAG {

void updateVertex(struct DAG *g, int v,

Edge Operations

Insert, Delete, Search, Update edges — always enforcing the directed and acyclic properties.

C — INSERT DIRECTED EDGE (safe, cycle check)
/* Check reachability from src to dest via DFS */
int canReach(struct DAG *g, int src, int dest) {
    if (src == dest) return 1;
    int vis[MAX] = {0}, stk[MAX], top = -1;
    stk[++top] = src;
    while (top >= 0) {
        int v = stk[top--];
        if (vis[v]) continue;
        vis[v] = 1;
        if (v == dest) return 1;
        for (int j = 0; j < g->V; j++)
            if (g->matrix[v][j] && !vis[j])
                stk[++top] = j;
    }
    return 0;
}

/* Safe insert: rejects if it would create cycle */
int insertEdge(struct DAG *g, int u, int v, int w) {
    if (u == v) { printf("No self-loops!\n"); return 0; }
    if (canReach(g, v, u)) { /* v already reaches u? */
        printf("Cycle! Edge %d→%d rejected.\n", u, v);
        return 0;
    }
    g->matrix[u][v] = w; /* Only [u][v], not [v][u] */
    printf("Edge %d→%d (w=%d) added.\n", u, v, w);
    return 1;
}
C — DELETE EDGE
void deleteEdge(struct DAG *g, int u, int v) {
    if (!g->matrix[u][v]) {
        printf("Edge %d→%d doesn't exist!\n", u, v);
        return;
    }
    g->matrix[u][v] = 0; /* Just set to zero — O(1) */
    printf("Edge %d→%d deleted.\n", u, v);
}
C — SEARCH EDGE — O(1)!
/* Adjacency matrix gives O(1) edge lookup */
int searchEdge(struct DAG *g, int u, int v) {
    if (g->matrix[u][v] != 0) {
        printf("Edge %d→%d EXISTS, weight=%d\n",
               u, v, g->matrix[u][v]);
        return 1;
    }
    printf("Edge %d→%d does NOT exist\n", u, v);
    return 0;
}
C — UPDATE EDGE WEIGHT
void updateEdgeWeight(struct DAG *g,
                       int u, int v, int newW) {
    if (!g->matrix[u][v]) {
        printf("Edge %d→%d doesn't exist!\n", u, v);
        return;
    }
    printf("Weight %d→%d: %d → %d\n",
           u, v, g->matrix[u][v], newW);
    g->matrix[u][v] = newW; /* O(1) update */
}

Validate Acyclic: DFS coloring — WHITE=unvisited, GRAY=in progress, BLACK=done. A back edge to a GRAY node = cycle found!

C — FULL ACYCLIC VALIDATION (DFS Coloring)
#define WHITE 0
#define GRAY  1
#define BLACK 2

int dfsCheck(struct DAG *g, int v, int col[]) {
    col[v] = GRAY;
    for (int j = 0; j < g->V; j++) {
        if (!g->matrix[v][j]) continue;
        if (col[j] == GRAY)  return 1; /* CYCLE! */
        if (col[j] == WHITE && dfsCheck(g, j, col))
            return 1;
    }
    col[v] = BLACK;
    return 0;
}

int isValidDAG(struct DAG *g) {
    int col[MAX] = {WHITE};
    for (int i = 0; i < g->V; i++)
        if (col[i] == WHITE && dfsCheck(g, i, col))
            return 0; /* has cycle */
    return 1; /* valid DAG */
}
🌍 Real-Life Analogy

Project dependencies: "Task A must finish before Task B." Adding an edge = creating a dependency. The cycle check prevents deadlock: "A waits for B, B waits for A" — that's a cycle and must be rejected!

🔍 Module 04

Traversal Operations

BFS, DFS, Topological Traversal, and Dependency Traversal — four ways to visit every vertex systematically.

TRAVERSAL VISUALIZER
Choose a traversal method. Watch nodes get visited step by step!

BFS (Breadth-First): Visit all neighbors at distance 1, then distance 2, etc. Uses a queue. Good for shortest unweighted paths.

C — BFS ON DAG
void BFS(struct DAG *g, int start) {
    int vis[MAX]={0}, Q[MAX], f=0, r=0;
    vis[start]=1; Q[r++]=start;
    printf("BFS: ");
    while (f < r) {
        int v = Q[f++];
        printf("%d ", v);
        for (int j=0; j < g->V; j++)
            if (g->matrix[v][j] && !vis[j]) {
                vis[j]=1; Q[r++]=j;
            }
    }
    printf("\n");
}

DFS (Depth-First): Go as deep as possible before backtracking. Uses recursion (or a stack). Naturally explores dependency chains.

C — DFS ON DAG (Recursive)
void dfsUtil(struct DAG *g, int v, int vis[]) {
    vis[v]=1;
    printf("%d ", v);
    for (int j=0; j < g->V; j++)
        if (g->matrix[v][j] && !vis[j])
            dfsUtil(g, j, vis);
}

void DFS(struct DAG *g, int start) {
    int vis[MAX]={0};
    printf("DFS: ");
    dfsUtil(g, start, vis);
    printf("\n");
}

Topological Traversal: Visit a vertex only after ALL its predecessors are visited. Post-order DFS + stack. The stack pop gives topological order.

C — TOPOLOGICAL TRAVERSAL (DFS Post-Order)
int stk[MAX], top=-1;

void topoUtil(struct DAG *g, int v, int vis[]) {
    vis[v]=1;
    for (int j=0; j < g->V; j++)
        if (g->matrix[v][j] && !vis[j])
            topoUtil(g, j, vis);
    stk[++top]=v; /* push AFTER all descendants */
}

void topologicalTraversal(struct DAG *g) {
    int vis[MAX]={0};  top=-1;
    for (int i=0; i < g->V; i++)
        if (!vis[i]) topoUtil(g, i, vis);
    printf("Topological: ");
    while (top>=0) printf("%d ", stk[top--]);
}

Dependency Traversal: Start from a target vertex, traverse backwards (follow incoming edges). Reveals all ancestors — everything that must complete before the target.

C — DEPENDENCY TRAVERSAL (Reverse DFS)
/* Find all prerequisites (ancestors) of target */
void depTraversal(struct DAG *g, int target) {
    int vis[MAX]={0}, stk[MAX], top=-1;
    stk[++top]=target;
    printf("Deps of %d: ", target);
    while (top>=0) {
        int v=stk[top--];
        if (vis[v]) continue;
        vis[v]=1;
        if (v!=target) printf("%d ", v);
        /* Walk BACKWARDS: check who points TO v */
        for (int i=0; i < g->V; i++)
            if (g->matrix[i][v] && !vis[i])
                stk[++top]=i;
    }
    printf("\n");
}
🌍 Real-Life Analogy

BFS=broadcasting floor-by-floor in a building. DFS=exploring a maze, going deep before backtracking. Topological=following a study plan in prerequisite order. Dependency=finding all ingredients a dish needs by working backwards.

🗺️ Module 05

Path Operations

Find any path, the shortest, the longest, or all paths. DAGs allow O(V+E) shortest/longest path — much better than general graphs!

PATH FINDER — WEIGHTED DAG
Select source/dest and click Shortest, Longest, or All Paths.
C — FIND PATH (existence check, DFS)
int findPath(struct DAG *g, int src, int dst,
              int vis[]) {
    if (src==dst) return 1;
    vis[src]=1;
    for (int j=0; j < g->V; j++) {
        if (g->matrix[src][j] && !vis[j])
            if (findPath(g,j,dst,vis)) return 1;
    }
    return 0; /* no path found */
}

Shortest Path in DAG: Topological sort + edge relaxation (same idea as Bellman-Ford but O(V+E) because we use topological order). Works with negative weights too!

C — SHORTEST PATH IN DAG (Topo + Relax)
#define INF 99999

void shortestPathDAG(struct DAG *g, int src) {
    int dist[MAX], par[MAX], ord[MAX], cnt=0;
    for (int i=0; i < g->V; i++) {
        dist[i]=INF; par[i]=-1;
    }
    dist[src]=0;

    kahnOrder(g, ord, &cnt); /* topological order */

    for (int i=0; i < cnt; i++) {
        int u=ord[i];
        if (dist[u]==INF) continue;
        for (int v=0; v < g->V; v++) {
            if (!g->matrix[u][v]) continue;
            int nd = dist[u] + g->matrix[u][v];
            if (nd < dist[v]) {   /* RELAX (min) */
                dist[v]=nd; par[v]=u;
            }
        }
    }
    /* dist[] now has shortest distances from src */
}

Longest Path in DAG: Same algorithm, use max instead of min. Start with -∞. Critical for project scheduling — NP-Hard in general graphs, but O(V+E) in DAGs!

C — LONGEST PATH IN DAG (Topo + Max-Relax)
#define NEGINF -99999

void longestPathDAG(struct DAG *g, int src) {
    int dist[MAX], ord[MAX], cnt=0;
    for (int i=0; i < g->V; i++) dist[i]=NEGINF;
    dist[src]=0;

    kahnOrder(g, ord, &cnt);

    for (int i=0; i < cnt; i++) {
        int u=ord[i];
        if (dist[u]==NEGINF) continue;
        for (int v=0; v < g->V; v++) {
            if (!g->matrix[u][v]) continue;
            int nd = dist[u] + g->matrix[u][v];
            if (nd > dist[v])  /* MAX relax! */
                dist[v]=nd;
        }
    }
}
C — ALL PATHS (DFS + Backtracking)
int path[MAX], pLen=0;

void allPaths(struct DAG *g, int src,
               int dst, int vis[]) {
    vis[src]=1; path[pLen++]=src;

    if (src==dst) {
        for (int i=0; i < pLen; i++)
            printf("%d%s",path[i],i<pLen-1?"→":"\n");
    } else {
        for (int j=0; j < g->V; j++)
            if (g->matrix[src][j] && !vis[j])
                allPaths(g,j,dst,vis);
    }
    pLen--;  vis[src]=0; /* BACKTRACK */
}
C — DEPENDENCY CHAIN ANALYSIS
/* Print full chain from source to target */
void printChain(int par[], int v) {
    if (par[v]==-1) { printf("%d",v); return; }
    printChain(par, par[v]);
    printf("→%d", v);
}

GPS Navigation: Shortest = fastest route. Longest = scenic route (or critical project timeline). All Paths = every possible route. Dep Chain = "to reach airport: pack → drive to station → metro → bus → airport terminal".

📋 Module 06

Ordering Operations

Topological Sort — arrange all vertices so every directed edge goes left→right. Two algorithms: DFS-based and Kahn's BFS-based.

TOPOLOGICAL SORT VISUALIZER
Click an algorithm to see topological ordering step by step.
  1. Mark all vertices as unvisited (WHITE).
  2. For each unvisited vertex, run DFS.
  3. After ALL neighbors are visited, push current vertex to a stack.
  4. Pop the entire stack — that's the topological order!
C — DFS TOPOLOGICAL SORT
int result[MAX], rTop=-1;

void topoSortUtil(struct DAG *g, int v,
                   int vis[], int stk[], int *top) {
    vis[v]=1;
    for (int j=0; j < g->V; j++)
        if (g->matrix[v][j] && !vis[j])
            topoSortUtil(g,j,vis,stk,top);
    stk[++(*top)]=v; /* post-order push */
}

void topoSortDFS(struct DAG *g) {
    int vis[MAX]={0}, stk[MAX], top=-1;
    for (int i=0; i < g->V; i++)
        if (!vis[i]) topoSortUtil(g,i,vis,stk,&top);
    printf("DFS Topo: ");
    while (top>=0) printf("%d ", stk[top--]);
}
  1. Compute in-degree of every vertex.
  2. Push all vertices with in-degree 0 into queue.
  3. Dequeue vertex u → add to result.
  4. For each neighbor v of u, decrement in-degree[v]. If it becomes 0, enqueue v.
  5. If result size < V → cycle exists. Else result = topological order.
C — KAHN'S ALGORITHM (BFS-based)
void kahnOrder(struct DAG *g, int ord[], int *cnt) {
    int n=g->V, inD[MAX]={0};
    int Q[MAX], f=0, r=0;
    *cnt=0;
    /* Step 1: compute in-degrees */
    for (int i=0; i < n; i++)
        for (int j=0; j < n; j++)
            if (g->matrix[i][j]) inD[j]++;
    /* Step 2: enqueue zero in-degree */
    for (int i=0; i < n; i++)
        if (inD[i]==0) Q[r++]=i;
    /* Step 3: process */
    while (f < r) {
        int u=Q[f++];
        ord[(*cnt)++]=u;
        for (int v=0; v < n; v++) {
            if (!g->matrix[u][v]) continue;
            if (--inD[v]==0) Q[r++]=v;
        }
    }
    if (*cnt != n) printf("Cycle detected!\n");
}
C — TASK SCHEDULING ORDER
char tasks[MAX][30] = {
    "Design UI", "Write API", "Setup DB",
    "Frontend",  "Testing",   "Deploy"
};

void printSchedule(struct DAG *g) {
    int ord[MAX], cnt=0;
    kahnOrder(g, ord, &cnt);
    printf("=== Task Schedule ===\n");
    for (int i=0; i < cnt; i++)
        printf("Step %d: %s\n", i+1, tasks[ord[i]]);
}
C — PACKAGE DEPENDENCY RESOLUTION (like npm/pip)
char pkgs[MAX][20] = {
    "libc","openssl","curl","node","npm","react"
};
/* Edges: libc→openssl, openssl→curl, libc→node */
/*        curl→npm, node→npm, npm→react          */

void installPackages(struct DAG *g) {
    int ord[MAX], cnt=0;
    kahnOrder(g, ord, &cnt);
    printf("Installing in order:\n");
    for (int i=0; i < cnt; i++)
        printf("  [OK] installing %s\n", pkgs[ord[i]]);
}
🌍 Real-Life Analogy

Getting dressed in the morning: Topological sort gives the order — underwear before pants, socks before shoes. Kahn's starts from items with no prerequisites (underwear!) and works outward. Package managers (npm, pip, apt) do exactly this!

🔄 Module 07

Cycle Operations

The defining property of a DAG is no cycles. We must detect cycles, prevent their creation, and validate the DAG property.

CYCLE DETECTOR (DFS Coloring)
Test cycle detection on a valid DAG vs a graph with a cycle.

DFS Coloring: WHITE=unvisited, GRAY=currently in DFS stack, BLACK=completely done. If we visit a GRAY node → back edge → cycle detected!

C — CYCLE DETECTION (DFS 3-Color)
#define WHITE 0
#define GRAY  1   /* currently in DFS stack */
#define BLACK 2   /* fully processed        */

int detectCycle(struct DAG *g, int v, int col[]) {
    col[v]=GRAY;
    for (int j=0; j < g->V; j++) {
        if (!g->matrix[v][j]) continue;
        if (col[j]==GRAY) {
            printf("CYCLE: back edge %d→%d!\n",v,j);
            return 1; /* cycle found */
        }
        if (col[j]==WHITE)
            if (detectCycle(g,j,col)) return 1;
    }
    col[v]=BLACK;
    return 0; /* no cycle through v */
}

int hasCycle(struct DAG *g) {
    int col[MAX]={WHITE};
    for (int i=0; i < g->V; i++)
        if (col[i]==WHITE && detectCycle(g,i,col))
            return 1;
    return 0;
}
C — PREVENT CYCLE ON EDGE INSERT
/* Before adding u→v, check if v can reach u */
/* If yes, adding u→v closes a cycle → reject */
int safeAddEdge(struct DAG *g, int u, int v) {
    if (u==v) {
        printf("Self-loop rejected!\n");
        return 0;
    }
    int vis[MAX]={0};
    if (canReach(g, v, u, vis)) {
        printf("Edge %d→%d creates cycle! Rejected.\n",u,v);
        return 0;
    }
    g->matrix[u][v]=1;
    printf("Edge %d→%d added safely.\n", u, v);
    return 1;
}
C — FULL DAG VALIDATION
int validateDAG(struct DAG *g) {
    int n=g->V;
    /* Check 1: no self-loops (diagonal = 0) */
    for (int i=0; i < n; i++) {
        if (g->matrix[i][i]) {
            printf("FAIL: self-loop at %d\n", i);
            return 0;
        }
    }
    /* Check 2: no cycles (Kahn's: if count<V) */
    int ord[MAX], cnt=0;
    kahnOrder(g, ord, &cnt);
    if (cnt != n) {
        printf("FAIL: cycle exists!\n");
        return 0;
    }
    printf("VALID DAG!\n");
    return 1;
}
🌍 Real-Life Analogy

University course prerequisites: A cycle would mean "Course A requires B, B requires C, C requires A" — you could never enroll in any! The registrar's system uses cycle detection to prevent this deadlock situation.

🔗 Module 08

Connectivity Operations

Reachability check, source node detection (in-degree=0), sink node detection (out-degree=0), and component analysis.

CONNECTIVITY ANALYZER
Click Sources, Sinks, or Reachability to analyze the graph.
C — REACHABILITY + TRANSITIVE CLOSURE
/* Can src reach dest? DFS-based O(V+E) */
int isReachable(struct DAG *g, int src, int dest) {
    int vis[MAX]={0};
    return findPath(g, src, dest, vis);
}

/* Transitive closure: reach[i][j]=1 if i can reach j */
void transitiveClosure(struct DAG *g,
                         int reach[MAX][MAX]) {
    int n=g->V;
    for (int i=0; i < n; i++)
        for (int j=0; j < n; j++)
            reach[i][j]=g->matrix[i][j];
    for (int i=0; i < n; i++) reach[i][i]=1;
    /* Floyd-Warshall transitive closure */
    for (int k=0; k < n; k++)
      for (int i=0; i < n; i++)
        for (int j=0; j < n; j++)
          if (reach[i][k] && reach[k][j])
            reach[i][j]=1;
}

C — WEAKLY CONNECTED COMPONENTS
/* Ignore edge direction, find connected groups */
void dfsComp(struct DAG *g, int v,
              int vis[], int comp[], int id) {
    vis[v]=1; comp[v]=id;
    for (int j=0; j < g->V; j++) {
        int connected = g->matrix[v][j] || g->matrix[j][v];
        if (connected && !vis[j])
            dfsComp(g,j,vis,comp,id);
    }
}
int countComponents(struct DAG *g) {
    int vis[MAX]={0}, comp[MAX], id=0;
    for (int i=0; i < g->V; i++)
        if (!vis[i]) dfsComp(g,i,vis,comp,id++);
    printf("%d component(s)\n", id);
    return id;
}

/* In-degree of vertex v: count column v */
int inDegree(struct DAG *g, int v) {
    int cnt=0;
    for (int i=0; i < g->V; i++) cnt+=g->matrix[i][v];
    return cnt; /* O(V) */

int outDegree(struct DAG *g, int v) {
    int cnt=0;
    for (int j=0; j < g->V; j++) cnt+=g->matrix[v][j];
    return cnt; /* O(V) */
}

int isSource(struct DAG *g, int v) { return inDegree(g,v)==0; }
int isSink  (struct DAG *g, int v) { return outDegree(g,v)==0; }

}

C — PARALLEL WORKFLOW SCHEDULING (Wave levels)
/* Group tasks by level — same level runs in parallel */
void parallelSchedule(struct DAG *g) {
    int n=g->V, inD[MAX]={0}, lv[MAX]={0};
    int Q[MAX], f=0, r=0;
    for (int i=0; i < n; i++)
        for (int j=0; j < n; j++)
            if (g->matrix[i][j]) inD[j]++;
    for (int i=0; i < n; i++) if (!inD[i]) Q[r++]=i;
    while (f < r) {
        int u=Q[f++];
        for (int v=0; v < n; v++) {
            if (!g->matrix[u][v]) continue;
            if (lv[u]+1 > lv[v]) lv[v]=lv[u]+1;
            if (!--inD[v]) Q[r++]=v;
        }
    }
    /* Tasks with same lv[] value run in parallel */
}
C — CRITICAL PATH ANALYSIS (CPM)
/* Forward pass → Earliest Start (ES)           */
/* Backward pass → Latest Start (LS)            */
/* Slack = LS-ES. Slack==0 → Critical Path      */
void criticalPathAnalysis(struct DAG *g, int dur[]) {
    int n=g->V, ord[MAX], cnt=0;
    int ES[MAX]={0}, LS[MAX], EF[MAX];
    kahnOrder(g, ord, &cnt);
    /* Forward pass */
    for (int i=0; i < cnt; i++) {
        int u=ord[i]; EF[u]=ES[u]+dur[u];
        for (int v=0; v < n; v++)
            if (g->matrix[u][v] && EF[u]>ES[v]) ES[v]=EF[u];
    }
    int T=0;
    for (int i=0; i < n; i++) if (EF[i]>T) T=EF[i];
    for (int i=0; i < n; i++) LS[i]=T-dur[i];
    /* Backward pass */
    for (int i=cnt-1; i>=0; i--) {
        int u=ord[i];
        for (int v=0; v < n; v++)
            if (g->matrix[u][v] && LS[v]-dur[u]<LS[u])
                LS[u]=LS[v]-dur[u];
    }
    printf("Critical Path (slack=0): ");
    for (int i=0; i < n; i++)
        if (LS[i]-ES[i]==0) printf("%d ", i);
}

void shortestFull(struct DAG *g, int src, int dst) {
    int n=g->V, dist[MAX], par[MAX], ord[MAX], cnt=0;
    for (int i=0; i < n; i++) { dist[i]=INF; par[i]=-1; }
    dist[src]=0;
    kahnOrder(g, ord, &cnt);
    for (int i=0; i < cnt; i++) {
        int u=ord[i];
        if (dist[u]==INF) continue;
        for (int v=0; v < n; v++) {
            if (!g->matrix[u][v]) continue;
            int nd=dist[u]+g->matrix[u][v];
            if (nd < dist[v]) { dist[v]=nd; par[v]=u; }
        }
    }
    printf("Dist %d→%d = %d  Path: ",src,dst,dist[dst]);
    printChain(par, dst); printf("\n");
}
C — LONGEST PATH (Full)
void longestFull(struct DAG *g, int src) {
    int n=g->V, dist[MAX], ord[MAX], cnt=0;
    for (int i=0; i < n; i++) dist[i]=NEGINF;
    dist[src]=0;
    kahnOrder(g, ord, &cnt);
    for (int i=0; i < cnt; i++) {
        int u=ord[i];
        if (dist[u]==NEGINF) continue;
        for (int v=0; v < n; v++)
            if (g->matrix[u][v]) {
                int nd=dist[u]+g->matrix[u][v];
                if (nd > dist[v]) dist[v]=nd; /* MAX */
            }
    }
    int mx=NEGINF;
    for (int i=0; i < n; i++) if (dist[i]>mx) mx=dist[i];
    printf("Longest path length from %d = %d\n",src,mx);
}
C — FULL CPM (ES, EF, LS, LF, Slack, Critical)
struct Task { char name[20]; int dur,ES,EF,LS,LF,slack,crit; };

void computeCPM(struct DAG *g, struct Task T[]) {
    int n=g->V, ord[MAX], cnt=0;
    kahnOrder(g, ord, &cnt);
    for (int i=0; i < n; i++) T[i].ES=0;
    /* Forward pass: compute ES and EF */
    for (int i=0; i < cnt; i++) {
        int u=ord[i]; T[u].EF=T[u].ES+T[u].dur;
        for (int v=0; v < n; v++)
            if (g->matrix[u][v] && T[u].EF>T[v].ES)
                T[v].ES=T[u].EF;
    }
    int end=0;
    for (int i=0; i < n; i++) if (T[i].EF>end) end=T[i].EF;
    for (int i=0; i < n; i++) T[i].LF=end;
    /* Backward pass: compute LS and LF */
    for (int i=cnt-1; i>=0; i--) {
        int u=ord[i]; T[u].LS=T[u].LF-T[u].dur;
        for (int p=0; p < n; p++)
            if (g->matrix[p][u] && T[u].LS<T[p].LF)
                T[p].LF=T[u].LS;
    }
    for (int i=0; i < n; i++) {
        T[i].slack=T[i].LS-T[i].ES;
        T[i].crit =(T[i].slack==0);
    }
}
