01
Initialization Operations
Creating and setting up the Indexable Skip List from scratch
📖 Theory
🎬 Visualization
💻 C Code
💡 Analogy

An Indexable Skip List is a probabilistic data structure that extends a regular linked list with multiple "express lanes" (levels). Unlike a regular skip list, it also tracks width values (spans) at each pointer, enabling O(log n) access by index — just like an array, but with dynamic insertion/deletion.

Initialization sets up the skeleton: a special header node that acts as the entry point, forward pointer arrays, width/span arrays, probability factor (usually 0.5), and the maximum level cap.

Main Operations

  • Create Indexable Skip List
  • Initialize Header Node
  • Set Maximum Level
  • Define Probability Factor

Suboperations

  • Allocate Header Node
  • Initialize Forward Pointers
  • Initialize Width/Span Arrays
  • Set Current Level
  • Assign NULL References

💡 Key Insight: The header node is a dummy node that never holds real data. All levels of the skip list start from the header. This simplifies insertion and deletion logic considerably.

The maximum level is typically set to log₂(expected_n). For 1 million elements, max_level ≈ 20. The probability factor p = 0.5 means each node has a 50% chance of being promoted to the next level.

C Language
/* ===== Indexable Skip List — Initialization ===== */
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define MAX_LEVEL  16   /* maximum number of levels */
#define PROBABILITY 0.5 /* coin-flip probability   */

/* ── Node Structure ── */
typedef struct Node {
    int key;                        /* the stored value        */
    struct Node *forward[MAX_LEVEL]; /* forward pointer array   */
    int width[MAX_LEVEL];            /* span/width per level    */
} Node;

/* ── Skip List Structure ── */
typedef struct {
    Node  *header;      /* dummy head node  */
    int    level;       /* current max used level */
    int    size;        /* total element count    */
    float  probability; /* promotion probability  */
} SkipList;

/* ── Allocate a New Node ── */
Node* createNode(int key) {
    Node *n = (Node*)malloc(sizeof(Node));
    if (!n) { printf("Memory error!\n"); exit(1); }
    n->key = key;
    for (int i = 0; i < MAX_LEVEL; i++) {
        n->forward[i] = NULL;   /* no next node yet */
        n->width[i]   = 0;     /* width is 0       */
    }
    return n;
}

/* ── Create the Skip List ── */
SkipList* createSkipList() {
    SkipList *sl = (SkipList*)malloc(sizeof(SkipList));
    sl->header      = createNode(-1); /* dummy header key = -1 */
    sl->level       = 0;             /* list starts empty     */
    sl->size        = 0;
    sl->probability = PROBABILITY;
    srand(time(NULL));               /* seed random levels    */
    printf("Skip list created! Header at %p\n", sl->header);
    return sl;
}

/* ── Main ── */
int main() {
    SkipList *sl = createSkipList();
    printf("Level: %d, Size: %d\n", sl->level, sl->size);
    return 0;
}

🏢 Real-Life Analogy: A Multi-Floor Library Index

Imagine a huge library. On the ground floor, every single book is in order (this is Level 0 — the base linked list). On the 2nd floor, only every other bookshelf is accessible (Level 1 — fewer pointers, faster jumps). On the 3rd floor, only every 4th bookshelf is reachable (Level 2). And so on.

When you initialize the library, you first build the reception desk (header node) with express-elevator cards for every floor. No books are shelved yet — but the structure is ready to receive them efficiently.

The width/span on each level tells you: "If I take this express elevator, I will skip over exactly N books." This is what makes it indexable — you can jump to the 500th book without reading the first 499!

Insertion is the most complex operation. It must: find the correct position at every level, update the forward pointer chain, and recalculate all width/span values so positional indexing remains accurate.

  • Insert Element
  • Insert at Position

Suboperations

The update[] array stores the last node at each level that falls to the left of the new node. The rank[] array tracks how many nodes each update node is from the header — this is critical for computing new widths.

⚠️ Width Update Rule: When inserting at level i, the new node's width at level i = (width of the update[i] node at level i) − (rank[new] − rank[update[i]]). All bypassed upper-level pointers get their width increased by 1.

C Language
/* ===== Insertion with Width/Span Update ===== */

/* Generate a random level using coin flips */
int randomLevel(SkipList *sl) {
    int lvl = 1;
    while ((float)rand() / RAND_MAX < sl->probability
           && lvl < MAX_LEVEL)
        lvl++;
    return lvl;
}

/* Insert a key into the Indexable Skip List */
void insert(SkipList *sl, int key) {
    Node *update[MAX_LEVEL]; /* predecessor at each level */
    int   rank[MAX_LEVEL];   /* rank of each predecessor  */
    Node *cur = sl->header;
    int   pos = 0;           /* current position count    */

    /* Step 1: Traverse from top level down, recording update nodes */
    for (int i = sl->level - 1; i >= 0; i--) {
        rank[i] = (i == sl->level - 1) ? 0 : rank[i + 1];
        while (cur->forward[i] && cur->forward[i]->key < key) {
            rank[i] += cur->width[i]; /* count skipped nodes */
            cur = cur->forward[i];
        }
        update[i] = cur; /* save the predecessor node */
    }

    /* Step 2: Pick a random height for the new node */
    int newLevel = randomLevel(sl);

    /* Step 3: If new level exceeds current, extend update array */
    if (newLevel > sl->level) {
        for (int i = sl->level; i < newLevel; i++) {
            rank[i]    = 0;
            update[i]  = sl->header;
            update[i]->width[i] = sl->size; /* entire list width */
        }
        sl->level = newLevel;
    }

    /* Step 4: Create the new node */
    Node *newNode = createNode(key);

    /* Step 5: Splice new node into each level and fix widths */
    for (int i = 0; i < newLevel; i++) {
        newNode->forward[i]    = update[i]->forward[i];
        update[i]->forward[i]  = newNode;

        /* Width of new node = predecessor's old width - rank gap */
        newNode->width[i]      = update[i]->width[i]
                                 - (rank[0] - rank[i]);
        update[i]->width[i]    = rank[0] - rank[i] + 1;
    }

    /* Step 6: Update skipped levels' widths (add 1 for new node) */
    for (int i = newLevel; i < sl->level; i++)
        update[i]->width[i]++;

    sl->size++; /* increment total count */
    printf("Inserted %d at level %d\n", key, newLevel);
}

🚂 Real-Life Analogy: Adding a New Train Station

Imagine a railway line where every station is a node. There are local trains (level 0, stop everywhere), express trains (level 1, stop every 2nd station), and bullet trains (level 2, stop every 4th). Each train has a ticket that says "my next stop is X stations away" — this is the width/span.

When you add a new station, you must update the tickets: the local train now shows an extra stop, the express train's skip count changes, and if the new station gets a bullet-train platform (random coin flip), you create a new bullet-train route through it and update all related tickets.

C Language
/* ===== Deletion with Width Merge ===== */

void deleteKey(SkipList *sl, int key) {
    Node *update[MAX_LEVEL];
    Node *cur = sl->header;

    /* Step 1: Find predecessors at each level */
    for (int i = sl->level - 1; i >= 0; i--) {
        while (cur->forward[i] && cur->forward[i]->key < key)
            cur = cur->forward[i];
        update[i] = cur;
    }

    /* Step 2: Check if the key actually exists at level 0 */
    Node *target = update[0]->forward[0];
    if (!target || target->key != key) {
        printf("Key %d not found!\n", key);
        return;
    }

    /* Step 3: Remove target from each level and merge widths */
    for (int i = 0; i < sl->level; i++) {
        if (update[i]->forward[i] != target) {
            /* target doesn't exist at this level — just decrease width */
            update[i]->width[i]--;
        } else {
            /* merge: predecessor's width absorbs target's width minus 1 */
            update[i]->width[i]  += target->width[i] - 1;
            update[i]->forward[i] = target->forward[i];
        }
    }

    /* Step 4: Remove empty top levels */
    while (sl->level > 1 &&
           sl->header->forward[sl->level - 1] == NULL)
        sl->level--;

    /* Step 5: Free memory and update count */
    free(target);
    sl->size--;
    printf("Deleted key %d. New size: %d\n", key, sl->size);
}

🧱 Real-Life Analogy: Removing a Floor from a Building

Imagine floors of a building are connected by elevators. Removing a floor means you have to reconnect the elevators that previously went through that floor. For each elevator (level), you redirect its cable from the floor above to the floor below the removed one, and update the "distance" reading on each elevator so passengers still know how many floors they'll skip.

If after removal the top floor of the building is empty (no one up there), you demolish that floor entirely — just like reducing sl→level after deletion when top levels become empty.

Search in a skip list begins at the highest level (express lane) and moves forward as long as the next key is ≤ the target. When no more forward progress is possible, it drops one level down and continues. It's like a greedy staircase descent.

/* ===== Search Operations ===== */

/* Search by key — returns the node or NULL */
Node* searchKey(SkipList *sl, int key) {
    Node *cur = sl->header;
    for (int i = sl->level - 1; i >= 0; i--) {
        /* move forward while next key is smaller */
        while (cur->forward[i] && cur->forward[i]->key < key)
            cur = cur->forward[i];
    }
    /* after dropping to level 0, check the next node */
    cur = cur->forward[0];
    if (cur && cur->key == key) {
        printf("Found: %d\n", cur->key);
        return cur;
    }
    printf("Key %d not found\n", key);
    return NULL;
}

/* Search by index (1-based) — O(log n) using widths */
Node* searchByIndex(SkipList *sl, int index) {
    if (index < 1 || index > sl->size) {
        printf("Index out of range\n");
        return NULL;
    }
    Node *cur = sl->header;
    int   remaining = index; /* steps left to count */

    for (int i = sl->level - 1; i >= 0; i--) {
        while (cur->forward[i] &&
               cur->width[i] <= remaining) {
            remaining -= cur->width[i]; /* subtract span */
            cur = cur->forward[i];
        }
    }
    printf("Index %d = key %d\n", index, cur->key);
    return cur;
}

/* Detect if a key is missing (returns 1 if absent) */
int isMissing(SkipList *sl, int key) {
    return searchKey(sl, key) == NULL;
}

Searching by index is like saying "I want the 50th city on the map" — you use the flight's distance gauge (width) to know how many cities each hop covers, and stop precisely when you've counted 50.

Positional Access Operations
O(log n) random access by numeric index — the superpower of indexable skip lists
📖 Theory
🎬 Visualization
💻 C Code
💡 Analogy

The "indexable" in indexable skip list comes entirely from this section. By storing a width (span) on every forward pointer — representing how many base-level nodes that pointer skips — we can compute positional rank queries in O(log n).

Main Operations

  • Access by Index
  • Find Position of Key

Suboperations

  • Traverse Using Width Values
  • Count Node Offsets
  • Locate Exact Position
  • Return Indexed Element
  • Compute Rank

How width works: If a level-2 pointer skips over 4 nodes at level 0, its width = 4. To get the 7th element: start with remaining=7, take a level-2 hop of width 4 → remaining=3, take a level-1 hop of width 2 → remaining=1, take a level-0 hop → you've arrived at index 7.

💡 Finding Rank of a Key: Accumulate widths while traversing to the key. The total accumulated width is the rank (1-based position) of that key in the sorted list.

Use Access by Index or Get Rank to see width traversal in action.
C Language
/* ===== Positional Access & Rank Computation ===== */

/* Access element at 1-based index using accumulated widths */
Node* accessByIndex(SkipList *sl, int index) {
    if (index < 1 || index > sl->size) return NULL;

    Node *cur       = sl->header;
    int   remaining = index;   /* positions left to count */

    for (int i = sl->level - 1; i >= 0; i--) {
        /* take the hop if it doesn't overshoot */
        while (cur->forward[i] &&
               cur->width[i] <= remaining) {
            remaining -= cur->width[i];
            cur = cur->forward[i];
            printf("  Hop at level %d → node %d (rem=%d)\n",
                   i, cur->key, remaining);
        }
    }
    return cur; /* remaining == 0 means exact node found */
}

/* Find 1-based rank of a given key */
int getRank(SkipList *sl, int key) {
    Node *cur  = sl->header;
    int   rank = 0;

    for (int i = sl->level - 1; i >= 0; i--) {
        while (cur->forward[i] &&
               cur->forward[i]->key <= key) {
            rank += cur->width[i];  /* accumulate skipped nodes */
            cur   = cur->forward[i];
        }
        if (cur->key == key) {
            printf("Rank of %d is %d\n", key, rank);
            return rank;
        }
    }
    printf("Key %d not found\n", key);
    return -1; /* not found */
}

Update modifies the value stored in an existing node. Since the skip list is keyed and sorted, we only update the value (satellite data), not the key — changing a key would break sorted order. If you need to change the key, delete and re-insert.

⚠️ Key vs Value: In a key-value skip list, keys define ordering and are immutable after insertion. Only the associated value (satellite data) can be safely updated in-place.

C Language
/* ===== Update Operations (Key-Value Version) ===== */

typedef struct KVNode {
    int   key;
    char  value[64];          /* satellite data */
    struct KVNode *forward[MAX_LEVEL];
    int   width[MAX_LEVEL];
} KVNode;

/* Search for a node by key, then modify its value */
int updateValue(SkipList *sl, int key, const char *newVal) {
    KVNode *cur = (KVNode*)sl->header;

    for (int i = sl->level - 1; i >= 0; i--) {
        while (cur->forward[i] &&
               cur->forward[i]->key < key)
            cur = cur->forward[i];
    }
    cur = cur->forward[0];

    if (!cur || cur->key != key) {
        printf("Key %d not found for update\n", key);
        return 0; /* failure */
    }

    /* Replace old value with new value in-place */
    printf("Old value: %s\n", cur->value);
    snprintf(cur->value, sizeof(cur->value), "%s", newVal);
    printf("Updated key %d → \"%s\"\n", key, cur->value);
    return 1; /* success */
}

/* Full traversal — always on level 0 (complete list) */
void traverseAll(SkipList *sl) {
    Node *cur = sl->header->forward[0]; /* skip dummy header */
    printf("Full list: ");

/* Traverse a specific level and show widths */
void traverseLevel(SkipList *sl, int level) {
    if (level < 0 || level >= sl->level) {
        printf("Invalid level\n"); return;
    }
    Node *cur = sl->header;
    printf("Level %d: [HDR]", level);
    while (cur->forward[level]) {
        printf(" -(%d)→ [%d]",
               cur->width[level],
               cur->forward[level]->key);
        cur = cur->forward[level];
    }
    printf("\n");
}

/* Show the entire skip list structure (all levels) */
void displaySkipList(SkipList *sl) {
    printf("\n=== Skip List (size=%d, level=%d) ===\n",
           sl->size, sl->level);
    for (int i = sl->level - 1; i >= 0; i--)
        traverseLevel(sl, i);
    printf("=====================================\n");
}

Level Management Operations
Growing and shrinking the skip list's height dynamically
📖 Theory
💻 C Code
💡 Analogy

The current level of a skip list is dynamic — it grows when a new node gets promoted to a level higher than current, and shrinks when the topmost level(s) become empty after deletion. Proper level management avoids wasted traversal through empty levels.

Main Operations

  • Manage Levels
  • Adjust Maximum Level

Suboperations

  • Increase Current Level
  • Decrease Current Level
  • Remove Empty Levels
  • Maintain Balanced Levels

💡 The ideal current level is ⌈log₂(n)⌉. With probability p=0.5, around 50% of nodes are at level 1, 25% at level 2, and so on — naturally balancing itself without any extra logic.

⚠️ Changing max_level after creation requires rebuilding or carefully extending all arrays — typically avoided in practice. Set max_level generously at creation time.

C Language
/* ===== Level Management ===== */

/* Called during insertion when new node exceeds current level */
void increaseLevel(SkipList *sl, int newLevel,
                    Node *update[]) {
    for (int i = sl->level; i < newLevel; i++) {
        update[i]          = sl->header;
        update[i]->width[i] = sl->size; /* full list width */
    }
    sl->level = newLevel;
    printf("Level increased to %d\n", sl->level);
}

/* Called during deletion to remove empty top levels */
void removeEmptyLevels(SkipList *sl) {
    while (sl->level > 1 &&
           sl->header->forward[sl->level - 1] == NULL) {
        sl->level--;
        printf("Removed empty top level. Now: %d\n", sl->level);
    }
}

/* Check and maintain balanced level distribution */
void maintainLevels(SkipList *sl) {
    int ideal = 1;
    int n = sl->size;
    while (n > 1) { n >>= 1; ideal++; } /* log2(size) */
    if (sl->level > ideal)
        printf("Warning: level %d > ideal %d\n",
               sl->level, ideal);
}

🏗️ Real-Life Analogy: A Growing Skyscraper

When a city is small, it needs only a 2-floor building (2 skip list levels). As the city grows, you add more floors (increase level). When the population shrinks and the top floors empty out, you demolish them (decrease level) to save maintenance costs.

The target height is always log₂(population) — perfectly balanced for efficient travel between any two floors.

Width / Span Management Operations
The backbone of O(log n) positional queries
📖 Theory
💻 C Code
💡 Analogy

Width (also called span) is the number of base-level (level 0) nodes a forward pointer skips over, plus 1 for the destination node itself. Maintaining widths correctly during every insert and delete is what separates an indexable skip list from a plain skip list.

Main Operations

  • Maintain Width Values
  • Maintain Span Information

Suboperations

  • Update Width During Insert
  • Update Width During Delete
  • Adjust Span Counts
  • Recalculate Positional Offsets

💡 Width Invariant: The sum of all width values along the path from header to a node at level 0 must always equal that node's 1-based position in the list.

During Insert: For each level ≤ newNode's level, split the predecessor's width. For each level > newNode's level, increment the predecessor's width by 1 (new node added below).

During Delete: For each level ≤ node's level, merge widths (pred.width = pred.width + node.width - 1). For each level > node's level, decrement by 1.

C Language
/* ===== Width / Span Management ===== */

/* Update widths during insertion (called from insert()) */
void updateWidthInsert(SkipList *sl, Node *newNode,
                        Node *update[], int rank[],
                        int newLevel) {
    int insertRank = rank[0]; /* rank of insertion point */

    /* Split widths for levels the new node participates in */
    for (int i = 0; i < newLevel; i++) {
        int gap = insertRank - rank[i];
        newNode->width[i]     = update[i]->width[i] - gap;
        update[i]->width[i]   = gap + 1;
    }

    /* Increment widths for levels above new node */
    for (int i = newLevel; i < sl->level; i++)
        update[i]->width[i]++;
}

/* Update widths during deletion (called from deleteKey()) */
void updateWidthDelete(SkipList *sl, Node *target,
                        Node *update[]) {
    for (int i = 0; i < sl->level; i++) {
        if (update[i]->forward[i] == target) {
            /* merge: absorb target's width, subtract 1 */
            update[i]->width[i] += target->width[i] - 1;
        } else {
            /* target not at this level — just decrement */
            update[i]->width[i]--;
        }
    }
}

/* Verify width invariant (for debugging) */
void verifyWidths(SkipList *sl) {
    Node *cur = sl->header->forward[0];
    int   pos = 1;
    while (cur) {
        int rank = getRank(sl, cur->key);
        if (rank != pos)
            printf("WIDTH ERROR at key %d: expected %d, got %d\n",
                   cur->key, pos, rank);
        pos++; cur = cur->forward[0];
    }
    printf("Width verification complete.\n");
}

Pointer Management Operations
Maintaining the network of forward references correctly
📖 Theory
💻 C Code
💡 Analogy

Every node holds an array of forward pointers — one per level it participates in. Pointer management ensures these references remain consistent during every structural change. Incorrect pointers cause infinite loops, skipped nodes, or segmentation faults.

Main Operations

  • Maintain Forward Pointers

Suboperations

  • Update Pointer Links
  • Redirect References
  • Remove Invalid Links
  • Initialize Pointer Arrays

⚠️ Always set unused forward pointer slots to NULL during node creation. Dangling pointers (pointing to freed memory) are the most common bug in skip list implementations.

C Language
/* ===== Pointer Management ===== */

/* Initialize pointer array for a newly created node */
void initPointers(Node *n) {
    for (int i = 0; i < MAX_LEVEL; i++) {
        n->forward[i] = NULL;  /* safe NULL initialization */
        n->width[i]   = 0;
    }
}

/* Update pointer links when splicing new node at level i */
void updatePointerLinks(Node *newNode, Node *pred, int level) {
    newNode->forward[level] = pred->forward[level]; /* inherit */
    pred->forward[level]    = newNode;              /* insert   */
}

/* Remove invalid link during deletion */
void removeLink(Node *pred, Node *target, int level) {
    if (pred->forward[level] == target) {
        pred->forward[level] = target->forward[level];
        target->forward[level] = NULL; /* prevent dangling pointer */
    }
}

/* Redirect a pointer (e.g., during split or merge) */
void redirectPointer(Node *node, Node *newTarget, int level) {
    node->forward[level] = newTarget;
    printf("Redirected level %d pointer of node %d → %d\n",
           level, node->key,
           newTarget ? newTarget->key : -1);
}

When an exchange is removed, you reconnect the cables directly from predecessor to successor — no dangling wires left behind.

Memory Management Operations
Safe allocation and deallocation — preventing leaks and corruption

💻 C Code
💡 Analogy

C requires explicit memory management. Every malloc() must be paired with a free(). A skip list can accumulate many nodes, and forgetting to free even one when the structure is destroyed causes a memory leak. Systematic destruction is critical in long-running systems.

  • Deallocate Memory

Suboperations

  • Create Node
  • Free Single Node
  • Free Entire Skip List
  • Prevent Memory Leak

⚠️ After calling free(ptr), immediately set ptr = NULL to prevent use-after-free bugs. Never access a freed node's members.

C Language
/* ===== Memory Management ===== */

/* Allocate memory for a new node */
Node* allocateNode(int key) {
    Node *n = (Node*)malloc(sizeof(Node));
    if (!n) {
        fprintf(stderr, "Out of memory!\n");
        exit(EXIT_FAILURE);
    }
    n->key = key;
    for (int i = 0; i < MAX_LEVEL; i++) {
        n->forward[i] = NULL; /* safe initialization */
        n->width[i]   = 0;
    }
    return n;
}

/* Free a single node safely */
void freeNode(Node **n) {
    if (*n) {
        free(*n);
        *n = NULL; /* prevent dangling pointer */
    }
}

/* Destroy entire skip list — free every node and struct */
void freeSkipList(SkipList **sl) {
    if (!*sl) return;

    Node *cur = (*sl)->header;  /* start from header */
    while (cur) {
        Node *next = cur->forward[0]; /* save next before free */
        free(cur);
        cur = next;
    }
    free(*sl);   /* free the SkipList struct itself */
    *sl = NULL;  /* caller's pointer goes NULL       */
    printf("Skip list fully freed.\n");
}

/* Safe usage pattern */
int main() {
    SkipList *sl = createSkipList();
    /* ... use the skip list ... */
    freeSkipList(&sl);   /* sl is NULL after this */
    return 0;
}

🏨 Real-Life Analogy: Hotel Check-in / Check-out

When a guest arrives, you allocate a room (malloc). When they leave, you free the room (free) and mark it available (set to NULL). If you forget to mark it available, the room appears occupied forever — a memory leak. If you accidentally give the room to a new guest while the old one is still there, that's a dangling pointer bug.

When the hotel closes permanently, you free every room systematically before demolishing the building — just like freeSkipList().

/* Measure search steps for a given key */
int measureSearchSteps(SkipList *sl, int key) {
    Node *cur   = sl->header;
    int   steps = 0;

    for (int i = sl->level - 1; i >= 0; i--) {
        while (cur->forward[i] &&
               cur->forward[i]->key < key) {
            cur = cur->forward[i];
            steps++;
        }
        steps++; /* count the drop too */
    }
    printf("Search for %d took %d steps (log2(%d)=%.1f)\n",
           key, steps, sl->size,
           sl->size > 1 ? log2(sl->size) : 0);

    while (cur) {
        total += sl->header->width[0];
        count++;
        cur = cur->forward[1];
    }
    if (count)
        printf("Avg level-1 width: %.2f (ideal ~2.0)\n",
               (float)total / count);
}

/* ===== Iterator Operations ===== */

/* Iterator struct — holds current position */
typedef struct {
    Node     *current;   /* current node pointer  */
    SkipList *list;      /* reference to the list */
} Iterator;

/* Create and initialize an iterator */
Iterator iteratorCreate(SkipList *sl) {
    Iterator it;
    it.list    = sl;
    it.current = sl->header->forward[0]; /* first real node */
    return it;
}

/* Check if more nodes exist */
int hasNext(Iterator *it) {
    return it->current != NULL;
}

/* Move to next node and return current key */
int iterNext(Iterator *it) {
    if (!it->current) {
        printf("Iterator exhausted\n");
        return -1;
    }
    int key    = it->current->key;
    it->current = it->current->forward[0]; /* advance */
    return key;
}

/* Reset iterator to beginning */
void iterReset(Iterator *it) {
    it->current = it->list->header->forward[0];
}

/* Example usage */
void printAll(SkipList *sl) {
    Iterator it = iteratorCreate(sl);
    printf("Iterate: ");
    while (hasNext(&it))
        printf("%d ", iterNext(&it));
    printf("\n");
}

📖 Real-Life Analogy: A Bookmark in a Book

An iterator is like a physical bookmark. You place it at the start (reset), then move it one page at a time (next), checking if you've reached the end (hasNext). You can always go back to the beginning by moving the bookmark to page 1.

The key benefit: multiple bookmarks can exist simultaneously in the same book — you can have multiple iterators traversing the same skip list at different positions without interfering with each other.

C Language
/* ===== Advanced Structural Operations ===== */

/* Find minimum — O(1) */
Node* findMin(SkipList *sl) {
    return sl->header->forward[0]; /* always first node */
}

/* Find maximum — O(log n) */
Node* findMax(SkipList *sl) {
    Node *cur = sl->header;
    for (int i = sl->level - 1; i >= 0; i--)
        while (cur->forward[i])
            cur = cur->forward[i]; /* move right on highest level */
    return (cur == sl->header) ? NULL : cur;
}

/* Range query: collect all keys in [lo, hi] */
void rangeQuery(SkipList *sl, int lo, int hi) {
    Node *cur = sl->header;

    /* Skip to the first node >= lo */
    for (int i = sl->level - 1; i >= 0; i--)
        while (cur->forward[i] && cur->forward[i]->key < lo)
            cur = cur->forward[i];

    cur = cur->forward[0]; /* first node >= lo */
    printf("Range [%d, %d]: ", lo, hi);
    while (cur && cur->key <= hi) {
        printf("%d ", cur->key);
        cur = cur->forward[0];
    }
    printf("\n");
}

/* Merge: insert all elements from src into dst */
void mergeSkipLists(SkipList *dst, SkipList *src) {
    Node *cur = src->header->forward[0];
    while (cur) {
        insert(dst, cur->key); /* re-insert each element */
        cur = cur->forward[0];
    }
    printf("Merged %d elements into destination.\n", src->size);
}

/* Split: create two lists — keys < pivot and keys >= pivot */
void splitSkipList(SkipList *src,
                   SkipList **left, SkipList **right,
                   int pivot) {
    *left  = createSkipList();
    *right = createSkipList();
    Node *cur = src->header->forward[0];
    while (cur) {
        if (cur->key < pivot) insert(*left, cur->key);
        else                  insert(*right, cur->key);
        cur = cur->forward[0];
    }
    printf("Split at %d: left=%d, right=%d nodes\n",
           pivot, (*left)->size, (*right)->size);
}

Merge is combining two post offices' sorted letter piles into one sorted pile. Split is dividing one office's letters into two piles at a dividing street number.
