What is a Queue?

// the foundation
DEFINITION

A Queue is a linear data structure that follows the FIFO principle — First In, First Out. The element that enters first is also the first one to leave.


Think of it as a one-way pipe: you push data in from one end (the rear) and pull data out from the other end (the front).

PROPERTIES
  • FIFO order — first inserted, first removed
  • Front pointer — tracks the deletion end
  • Rear pointer — tracks the insertion end
  • Fixed or dynamic — array-based or linked list
  • O(1) enqueue & dequeue — very efficient
STRUCTURE IN C — ARRAY-BASED QUEUE (used throughout this guide)
C · queue.h
#define MAX 10          // maximum size of the queue

struct Queue {
    int items[MAX];   // array to hold elements
    int front;        // index of front element
    int rear;         // index of rear element
};

// Initialize queue — always do this first!
void initQueue(struct Queue *q) {
    q->front = -1;
    q->rear  = -1;
}

01

Enqueue

// insert element at rear
THEORY

Enqueue adds a new element to the rear of the queue. Before inserting, we must check if the queue is full (overflow condition).


Steps:

  • Check if queue is full → overflow error
  • If queue is empty, set front = 0
  • Increment rear by 1
  • Insert element at items[rear]
COMPLEXITY
CaseTimeSpace
AlwaysO(1)O(1)

Enqueue is a constant-time operation — it always takes the same time regardless of queue size.

C CODE — ENQUEUE
C · enqueue.c
#include <stdio.h>
#define MAX 10

struct Queue {
    int items[MAX], front, rear;
};

// Enqueue: insert element at rear
void enqueue(struct Queue *q, int value) {
    if (q->rear == MAX - 1) {
        printf("Queue Overflow! Cannot insert %d\n", value);
        return;
    }
    if (q->front == -1)   // first element being inserted
        q->front = 0;

    q->rear++;              // move rear forward
    q->items[q->rear] = value; // store value
    printf("Enqueued: %d\n", value);
}

int main() {
    struct Queue q;
    q.front = q.rear = -1;   // initialize
    enqueue(&q, 10);
    enqueue(&q, 20);
    enqueue(&q, 30);
    return 0;
}

When you join a fast food queue, you stand at the rear of the line. You can't join from the middle or the front — you must join at the back. This is exactly what enqueue does: it always inserts at the rear.

02

Dequeue

// remove element from front
THEORY

Dequeue removes the element at the front of the queue. Before removing, we check for underflow (empty queue).


Steps:

  • Check if queue is empty → underflow error
  • Store items[front] as the value to return
  • If front == rear → queue is now empty, reset both to -1
  • Otherwise increment front by 1
COMPLEXITY
CaseTimeSpace
AlwaysO(1)O(1)

Dequeue is also constant-time. We simply move the front pointer forward — no shifting needed!

LIVE VISUALIZATION

C CODE — DEQUEUE
C · dequeue.c
// Dequeue: remove element from front
int dequeue(struct Queue *q) {
    if (q->front == -1) {
        printf("Queue Underflow! Queue is empty.\n");
        return -1;
    }
    int value = q->items[q->front]; // save front value

    if (q->front == q->rear) {      // queue had only 1 element
        q->front = q->rear = -1;
    } else {
        q->front++;                  // move front forward
    }
    printf("Dequeued: %d\n", value);
    return value;
}

int front(struct Queue *q) {

#define MAX 10

// isFull: returns 1 if queue is full, 0 otherwise
int isFull(struct Queue *q) {
    return (q->rear == MAX - 1);
}

// Safe enqueue that checks first
void safeEnqueue(struct Queue *q, int val) {
    if (isFull(q)) {
        printf("Cannot insert %d — Queue is FULL!\n", val);

    enqueue(q, val);

    q.front = q.rear = -1;
    for (int i = 1; i <= 11; i++)
        safeEnqueue(&q, i * 10);  // 11th insert fails
    return 0;
}

// Traverse: print all elements front to rear
void display(struct Queue *q) {
    if (q->front == -1) {
        printf("Queue is empty!\n");
        return;
    }
    printf("Queue [FRONT -> REAR]: ");
    for (int i = q->front; i <= q->rear; i++) {
        printf("%d ", q->items[i]);
    }
    printf("\n");
}

int main() {
    struct Queue q;
    q.front = q.rear = -1;
    enqueue(&q, 10); enqueue(&q, 20); enqueue(&q, 30);
    display(&q);
    // Output: Queue [FRONT -> REAR]: 10 20 30
    return 0;
}

// find an element by value

Search scans the queue from front to rear to find a specific element. It returns the position of the element if found, or -1 if not.


Since a queue has no inherent ordering guarantee, we must do a linear search — check each element one by one.

// Search: find element in queue, return position or -1
int search(struct Queue *q, int key) {
    if (q->front == -1) {
        printf("Queue is empty!\n");
        return -1;
    }
    for (int i = q->front; i <= q->rear; i++) {
        if (q->items[i] == key) {
            printf("Found %d at position %d\n", key, i);
            return i;               // return position
        }
    }
    printf("%d not found in queue.\n", key);
    return -1;
}

int main() {
    struct Queue q;
    q.front = q.rear = -1;
    enqueue(&q, 10); enqueue(&q, 25); enqueue(&q, 40);
    search(&q, 25);  // Found 25 at position 1
    search(&q, 99);  // 99 not found
    return 0;
}

Clear

// delete all elements at once
THEORY

Clear empties the queue by resetting both front and rear back to -1.


In array-based queues, we don't need to physically delete the values — just resetting the pointers is enough. The old data becomes inaccessible and is effectively "gone."

COMPLEXITY
CaseTimeSpace
AlwaysO(1)O(1)

Resetting two integers is constant — much faster than deleting one by one!

LIVE VISUALIZATION
Queue is empty.

// Clear: reset queue to empty state
void clear(struct Queue *q) {
    q->front = -1;

Copy / Clone

// duplicate the queue into a new one
THEORY

Copy / Clone creates an exact duplicate of the queue. The original queue remains unchanged.


Steps:

  • Create a new empty queue
  • Copy the front and rear values
  • Loop from front to rear, copying each element
  • Both queues now have identical data
COMPLEXITY
CaseTimeSpace
AlwaysO(n)O(n)

We need to copy each element, and the new queue needs n extra spaces.

C CODE — COPY / CLONE
C · clone.c
// Clone: create an exact copy of the queue
void cloneQueue(struct Queue *original, struct Queue *copy) {
    copy->front = original->front;  // copy front pointer
    copy->rear  = original->rear;   // copy rear pointer

    for (int i = original->front; i <= original->rear; i++) {
        copy->items[i] = original->items[i]; // copy each element
    }
    printf("Queue cloned successfully!\n");
}

int main() {
    struct Queue q1, q2;
    q1.front = q1.rear = -1;
    q2.front = q2.rear = -1;
    enqueue(&q1, 10); enqueue(&q1, 20); enqueue(&q1, 30);
    cloneQueue(&q1, &q2);
    display(&q2);  // Output: 10 20 30
    return 0;
}

Reverse Queue

// flip the order of all elements
THEORY

Reverse Queue rearranges elements so the rear becomes front and vice versa. The element that was last is now first.


Method — using a Stack (or simple loop):

  • Dequeue all elements and push onto a temporary array
  • Enqueue them back in reverse order
  • Or use a two-pointer swap from front and rear inward
COMPLEXITY
CaseTimeSpace
AlwaysO(n)O(n)
LIVE VISUALIZATION
Add elements, then reverse!
C CODE — REVERSE QUEUE
C · reverse.c
// Reverse: flip queue order using a temp array
void reverseQueue(struct Queue *q) {
    if (q->front == -1) return;  // nothing to reverse

    int temp[MAX];
    int count = 0;

    // Step 1: Copy all elements to temp array
    for (int i = q->front; i <= q->rear; i++)
        temp[count++] = q->items[i];

    // Step 2: Reset queue
    int oldFront = q->front;
    q->rear = oldFront - 1;
    q->front = q->rear = -1;

    // Step 3: Enqueue in reverse order
    for (int i = count - 1; i >= 0; i--)
        enqueue(q, temp[i]);

    printf("Queue reversed!\n");
}

int main() {
    struct Queue q;
    q.front = q.rear = -1;
    enqueue(&q, 10); enqueue(&q, 20); enqueue(&q, 30);
    display(&q);       // 10 20 30
    reverseQueue(&q);
    display(&q);       // 30 20 10
    return 0;
}

Real-Life Analogy — Reversing a Queue of Photos

A photographer has 10 photos in order. The client wants them in reverse chronological order — last taken, first shown. The photographer stacks them face-down (like a stack) then picks them off one by one into a new pile. The last photo is now first!

Sort Queue

// arrange elements in ascending order
THEORY

Sort Queue rearranges all elements in ascending (or descending) order while preserving the queue structure.


Strategy:

  • Extract all elements into a temporary array
  • Sort the array using Bubble Sort (beginner-friendly)
  • Re-enqueue elements from the sorted array
COMPLEXITY
CaseTimeSpace
BestO(n)O(n)
Average / WorstO(n²)O(n)

Bubble sort used for clarity — replace with O(n log n) sort for production.

LIVE VISUALIZATION
Add elements in any order, then sort!
C CODE — SORT QUEUE (BUBBLE SORT)
C · sort.c
// Sort Queue using Bubble Sort
void sortQueue(struct Queue *q) {
    if (q->front == -1) return;

    int n = q->rear - q->front + 1;

    // Bubble sort directly on the items array
    for (int i = 0; i < n - 1; i++) {
        for (int j = q->front; j < q->rear - i; j++) {
            if (q->items[j] > q->items[j + 1]) {
                // swap adjacent elements
                int tmp       = q->items[j];
                q->items[j]   = q->items[j + 1];
                q->items[j + 1] = tmp;
            }
        }
    }
    printf("Queue sorted in ascending order!\n");
}

int main() {
    struct Queue q;
    q.front = q.rear = -1;
    enqueue(&q, 40); enqueue(&q, 10); enqueue(&q, 30); enqueue(&q, 20);
    display(&q);     // 40 10 30 20
    sortQueue(&q);
    display(&q);     // 10 20 30 40
    return 0;
}

Merge Queues

// combine two queues into one
THEORY

Merge combines two queues into a single queue. All elements of Queue 2 are appended after all elements of Queue 1.


Steps:

  • Check combined size doesn't exceed MAX
  • Loop through Queue 2 (front to rear)
  • Enqueue each element into Queue 1
  • Optionally clear Queue 2 afterward
COMPLEXITY
CaseTimeSpace
AlwaysO(m)O(1)

Where m is the number of elements in Queue 2. Queue 1 elements aren't moved.

// Merge: append q2 elements to end of q1
void mergeQueues(struct Queue *q1, struct Queue *q2) {
    if (q2->front == -1) {
        printf("Queue 2 is empty, nothing to merge.\n");
        return;
    }
    for (int i = q2->front; i <= q2->rear; i++) {
        if (isFull(q1)) {
            printf("Queue 1 full! Merge stopped.\n");
            break;
        }
        enqueue(q1, q2->items[i]); // add q2 elems to q1
    }
    clear(q2);  // empty q2 after merge (optional)
    printf("Queues merged!\n");
}

int main() {
    struct Queue q1, q2;
    q1.front = q1.rear = q2.front = q2.rear = -1;
    enqueue(&q1, 1); enqueue(&q1, 2);
    enqueue(&q2, 3); enqueue(&q2, 4);
    mergeQueues(&q1, &q2);
    display(&q1);  // 1 2 3 4
    return 0;
}

Split Queue

// divide one queue into two halves
THEORY

Split divides a queue into two separate queues. Typically we split at the midpoint — first half goes to Queue 1, second half to Queue 2.


Steps:

  • Find the total size n
  • Calculate midpoint mid = n / 2
  • Elements 0..mid-1 go to Queue 1
  • Elements mid..n-1 go to Queue 2
COMPLEXITY
CaseTimeSpace
AlwaysO(n)O(n)

We traverse all n elements and create two new queues totalling n spaces.

// Split: divide queue into two halves
void splitQueue(struct Queue *orig,
               struct Queue *q1,
               struct Queue *q2) {
    if (orig->front == -1) {
        printf("Original queue is empty!\n");
        return;
    }

    int n   = orig->rear - orig->front + 1;
    int mid = n / 2;   // midpoint

    q1->front = q1->rear = -1;
    q2->front = q2->rear = -1;

    int i = orig->front;

    // First half → q1
    for (int k = 0; k < mid; k++, i++)
        enqueue(q1, orig->items[i]);

    // Second half → q2
    for (; i <= orig->rear; i++)
        enqueue(q2, orig->items[i]);

    printf("Queue split at midpoint %d!\n", mid);
}

int main() {
    struct Queue orig, q1, q2;
    orig.front = orig.rear = -1;
    enqueue(&orig, 10); enqueue(&orig, 20);
    enqueue(&orig, 30); enqueue(&orig, 40);
    splitQueue(&orig, &q1, &q2);
    display(&q1);  // 10 20
    display(&q2);  // 30 40
    return 0;
}
