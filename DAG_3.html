01 — INTRODUCTION

What is a DAG?

A Directed Acyclic Graph (DAG) is a graph with two key properties:

  • Directed — every edge has a direction (goes from one vertex to another)
  • Acyclic — there are no cycles (you can never return to where you started by following edges)
🍳 Real-Life Analogy

Think of a cooking recipe. You must chop vegetables before you fry them, and fry them before you serve. The steps have a direction, and you can never go back in time — that's a DAG!

🎯
Directed Edges
Each edge A→B means "A comes before B" or "A depends on B being ready first."
🔄
No Cycles
Following edges, you can never return to the starting vertex. No circular dependencies.
🌳
Multiple Paths
Unlike trees, a node can have multiple parents. Two courses can both require the same prerequisite.
⚡
Topological Order
DAGs always have a linear ordering of vertices where every edge points forward.

Representation: Directed Incidence Matrix

The Directed Incidence Matrix is a 2D array where:

  • Rows = Vertices (nodes)
  • Columns = Edges (connections)
  • Cell value +1 = this vertex is the tail (source) of the edge
  • Cell value -1 = this vertex is the head (destination) of the edge
  • Cell value 0 = this vertex is not involved in this edge
📮 Real-Life Analogy

Imagine a spreadsheet of package deliveries. Rows are warehouses, columns are delivery routes. +1 means "this warehouse SENDS on this route", -1 means "this warehouse RECEIVES from this route".

Reading the Matrix

Row A, column e₁ = +1 → A is the SOURCE of edge e₁. Row B, column e₁ = -1 → B is the DESTINATION of edge e₁. So e₁ = A→B ✓

C — DAG Structure (Incidence Matrix)
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_V 20   /* max vertices */
#define MAX_E 50   /* max edges    */

/* Directed Incidence Matrix: 
   +1 = vertex is tail (source) of edge
   -1 = vertex is head (destination) of edge
    0 = not involved                        */
typedef struct {
    int  matrix[MAX_V][MAX_E];  /* incidence matrix */
    int  weight[MAX_E];         /* edge weights     */
    char vname[MAX_V][20];      /* vertex names     */
    int  vcount;               /* number of vertices */
    int  ecount;               /* number of edges    */
} DAG;

/* Initialize an empty DAG */
void dag_init(DAG *g) {
    g->vcount = 0;
    g->ecount = 0;
    memset(g->matrix, 0, sizeof(g->matrix));
    memset(g->weight, 0, sizeof(g->weight));
}

int main() {
    DAG g;
    dag_init(&g);
    printf("DAG initialized! Vertices: %d, Edges: %d\n",
           g.vcount, g.ecount);
    return 0;
}

02 — VERTEX OPERATIONS

Insert Vertex

Adding a new vertex means adding a new row to the incidence matrix. All entries in the new row start as 0 (the new vertex has no edges yet).

🏫 Real-Life Analogy

Adding a new city to a road network map. The city exists but has no roads yet — all cells are 0. Later, we can add roads (edges) connecting it.

Insert Vertex — Interactive Demo

C — Insert Vertex
/* Add a new vertex to the DAG.
   Returns vertex index, or -1 if full. */
int insert_vertex(DAG *g, const char *name) {
    if (g->vcount >= MAX_V) {
        printf("Error: max vertices reached!\n");
        return -1;
    }
    /* Copy the name */
    strncpy(g->vname[g->vcount], name, 19);

    /* New row is all zeros — no edges yet */
    int e;
    for (e = 0; e < g->ecount; e++)
        g->matrix[g->vcount][e] = 0;

    g->vcount++;
    printf("Vertex '%s' inserted at index %d\n",
           name, g->vcount - 1);
    return g->vcount - 1;
}

Delete Vertex

Deleting a vertex requires: (1) removing all edges connected to it, (2) removing its row from the matrix, and (3) shifting all remaining rows up.

🗺️ Real-Life Analogy

Closing a city. You must first close all roads (edges) that connect to it, then remove it from the map. Other cities reconnect directly if needed.

C — Delete Vertex
/* Find vertex index by name */
int find_vertex(DAG *g, const char *name) {
    int i;
    for (i = 0; i < g->vcount; i++)
        if (strcmp(g->vname[i], name) == 0)
            return i;
    return -1;  /* not found */
}

/* Delete a vertex and all its edges */
int delete_vertex(DAG *g, const char *name) {
    int vi = find_vertex(g, name);
    if (vi == -1) { printf("Vertex not found.\n"); return -1; }

    /* Step 1: Remove edges connected to vi
       Scan columns; delete cols where row vi != 0 */
    int e = 0;
    while (e < g->ecount) {
        if (g->matrix[vi][e] != 0) {
            /* Shift all columns left by 1 */
            int i, j;
            for (j = e; j < g->ecount - 1; j++) {
                for (i = 0; i < g->vcount; i++)
                    g->matrix[i][j] = g->matrix[i][j+1];
                g->weight[j] = g->weight[j+1];
            }
            g->ecount--;
        } else e++;
    }

    /* Step 2: Shift rows up (remove vertex row) */
    int i, j;
    for (i = vi; i < g->vcount - 1; i++) {
        for (j = 0; j < g->ecount; j++)
            g->matrix[i][j] = g->matrix[i+1][j];
        strcpy(g->vname[i], g->vname[i+1]);
    }
    g->vcount--;
    printf("Vertex '%s' deleted.\n", name);
    return 0;
}

C — Insert Directed Edge
/* Forward declaration */
int has_cycle(DAG *g);

/* Insert directed edge from 'from' to 'to' with weight w.
   Returns edge index or -1 on failure/cycle detected. */
int insert_edge(DAG *g, int from, int to, int w) {
    if (g->ecount >= MAX_E) {
        printf("Error: max edges reached!\n"); return -1;
    }
    if (from < 0 || from >= g->vcount ||
        to   < 0 || to   >= g->vcount) {
        printf("Error: invalid vertex index.\n"); return -1;
    }

    /* Add column tentatively */
    int e = g->ecount;
    g->matrix[from][e] = +1;  /* source of edge */
    g->matrix[to  ][e] = -1;  /* dest of edge   */
    g->weight[e] = w;
    g->ecount++;

    /* Check acyclic property */
    if (has_cycle(g)) {
        /* Roll back */
        g->matrix[from][e] = 0;
        g->matrix[to  ][e] = 0;
        g->ecount--;
        printf("Cycle detected! Edge rejected.\n");
        return -1;
    }
    printf("Edge %s→%s (w=%d) inserted at col %d\n",
           g->vname[from], g->vname[to], w, e);
    return e;
}

Delete Edge

Deleting an edge removes its column from the matrix and shifts remaining columns left.

C — Delete Edge
/* Find edge from 'from' to 'to'. Returns column index or -1 */
int find_edge(DAG *g, int from, int to) {
    int e;
    for (e = 0; e < g->ecount; e++)
        if (g->matrix[from][e] == +1 &&
            g->matrix[to  ][e] == -1)
            return e;
    return -1;
}

/* Delete edge by shifting columns left */
int delete_edge(DAG *g, int from, int to) {
    int e = find_edge(g, from, to);
    if (e == -1) { printf("Edge not found.\n"); return -1; }

    /* Shift columns left from position e */
    int i, j;
    for (j = e; j < g->ecount - 1; j++) {
        for (i = 0; i < g->vcount; i++)
            g->matrix[i][j] = g->matrix[i][j+1];
        g->weight[j] = g->weight[j+1];
    }
    g->ecount--;
    printf("Edge %s→%s deleted.\n",
           g->vname[from], g->vname[to]);
    return 0;
}

Validate Acyclic Condition

A graph is a valid DAG only if it has no cycles. We validate using DFS — if we encounter a vertex that is already in the current DFS path (gray/visiting state), a cycle exists.

🔍 Real-Life Analogy

Checking if a company's task list has circular dependencies — e.g., "Task A requires Task B, B requires C, C requires A" — which would be impossible to complete!

C — Cycle Detection / Acyclic Validation
/* Build adjacency list from incidence matrix for DFS */
static int visited[MAX_V], in_stack[MAX_V];

/* DFS helper: returns 1 if cycle found */
static int dfs_cycle(DAG *g, int v) {
    visited[v] = 1;
    in_stack[v] = 1;  /* mark as "in current path" */

    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == +1) {  /* v is source */
            /* Find destination of this edge */
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == -1) {
                    if (!visited[u] && dfs_cycle(g, u))
                        return 1;
                    if (in_stack[u])
                        return 1;  /* Back edge = cycle! */
                }
            }
        }
    }
    in_stack[v] = 0;  /* remove from current path */
    return 0;
}

/* Returns 1 if graph has a cycle, 0 if it's a DAG */
int has_cycle(DAG *g) {
    int i;
    memset(visited,  0, sizeof(visited));
    memset(in_stack, 0, sizeof(in_stack));
    for (i = 0; i < g->vcount; i++)
        if (!visited[i] && dfs_cycle(g, i))
            return 1;
    return 0;
}

void validate_dag(DAG *g) {
    if (has_cycle(g))
        printf("INVALID: Graph has a cycle!\n");
    else
        printf("VALID DAG: No cycles detected.\n");
}
04 — TRAVERSAL OPERATIONS

Breadth-First Search (BFS)

BFS explores the graph level by level. Starting from a source vertex, it visits all immediate neighbors first, then their neighbors, using a queue.

💧 Real-Life Analogy

Think of dropping a stone in a pond — ripples spread outward in rings. BFS explores "rings" of neighbors one level at a time.

C — BFS on DAG (Incidence Matrix)
/* BFS from start vertex. Prints visit order. */
void bfs(DAG *g, int start) {
    int queue[MAX_V];   /* simple array as queue */
    int front = 0, rear = 0;
    int visited[MAX_V];
    memset(visited, 0, sizeof(visited));

    queue[rear++] = start;
    visited[start] = 1;

    while (front < rear) {
        int v = queue[front++];
        printf("Visit: %s\n", g->vname[v]);

        /* Find all neighbors of v via incidence matrix */
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[v][e] == +1) {  /* v is source */
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1
                        && !visited[u]) {
                        visited[u] = 1;
                        queue[rear++] = u;
                    }
                }
            }
        }
    }
}

Depth-First Search (DFS)

DFS dives as deep as possible along each branch before backtracking. Uses a recursive call stack (or explicit stack).

🗺️ Real-Life Analogy

Exploring a maze by always turning left: you go as deep as you can, then backtrack when stuck, trying the next path.

C — DFS on DAG
/* DFS recursive helper */
void dfs_helper(DAG *g, int v, int vis[]) {
    vis[v] = 1;
    printf("Visit: %s\n", g->vname[v]);

    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == +1) {
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == -1
                    && !vis[u]) {
                    dfs_helper(g, u, vis);
                }
            }
        }
    }
}

/* DFS from start vertex */
void dfs(DAG *g, int start) {
    int vis[MAX_V];
    memset(vis, 0, sizeof(vis));
    dfs_helper(g, start, vis);
}

Topological Traversal

Visits vertices in topological order — each vertex is visited only after all its predecessors. This is the foundation for scheduling algorithms.

👔 Real-Life Analogy

Getting dressed in the morning: socks before shoes, underwear before pants. There's a valid order that respects all dependencies.

C — Topological Traversal
/* Compute in-degree of each vertex using incidence matrix */
void compute_indegree(DAG *g, int indeg[]) {
    int v, e;
    for (v = 0; v < g->vcount; v++) indeg[v] = 0;
    for (e = 0; e < g->ecount; e++)
        for (v = 0; v < g->vcount; v++)
            if (g->matrix[v][e] == -1) indeg[v]++;
}

/* Topological traversal using Kahn's idea */
void topo_traversal(DAG *g) {
    int indeg[MAX_V], queue[MAX_V];
    compute_indegree(g, indeg);
    int front = 0, rear = 0;
    int v;
    /* Enqueue vertices with in-degree 0 */
    for (v = 0; v < g->vcount; v++)
        if (indeg[v] == 0) queue[rear++] = v;

    printf("Topological order: ");
    while (front < rear) {
        v = queue[front++];
        printf("%s ", g->vname[v]);
        /* Reduce in-degree of successors */
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1) {
                        indeg[u]--;
                        if (indeg[u] == 0)
                            queue[rear++] = u;
                    }
                }
            }
        }
    }
    printf("\n");
}

Dependency Traversal

Given a target vertex, find all vertices it depends on (ancestors) by traversing edges in reverse direction.

C — Dependency Traversal (Reverse DFS)
/* Find all dependencies (ancestors) of vertex v */
void dep_traversal(DAG *g, int v, int vis[]) {
    vis[v] = 1;
    int e, u;
    /* Find edges where v is DESTINATION (-1) */
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == -1) {  /* v is dest */
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == +1
                    && !vis[u]) {
                    printf("%s depends on %s\n",
                           g->vname[v], g->vname[u]);
                    dep_traversal(g, u, vis);
                }
            }
        }
    }
}

C — Find Path (DFS)
/* Returns 1 if path exists from s to t */
int find_path(DAG *g, int s, int t,
              int path[], int *plen,
              int vis[]) {
    vis[s] = 1;
    path[(*plen)++] = s;
    if (s == t) return 1;

    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[s][e] == +1) {
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == -1
                    && !vis[u]) {
                    if (find_path(g, u, t, path,
                                  plen, vis))
                        return 1;
                }
            }
        }
    }
    (*plen)--;  /* backtrack */
    return 0;
}

Shortest Path in DAG

In a weighted DAG, find the shortest path using topological sort + relaxation. This is faster than Dijkstra's for DAGs — O(V+E) vs O((V+E)logV).

🚗 Real-Life Analogy

Finding the cheapest route from headquarters to a delivery location, where each road has a toll. Process intersections in topological order, always keeping the cheapest cumulative cost.

Shortest Path — Weighted DAG Visualization
C — Shortest Path in DAG (Topo + Relaxation)
#define INF 99999

/* Shortest path from src to all vertices */
void shortest_path_dag(DAG *g, int src) {
    int dist[MAX_V], topo[MAX_V], tlen = 0;
    int i;
    for (i = 0; i < g->vcount; i++) dist[i] = INF;
    dist[src] = 0;

    /* Get topological order */
    topo_sort_dfs(g, topo, &tlen); /* see topo sort section */

    /* Process vertices in topological order */
    int ti;
    for (ti = 0; ti < tlen; ti++) {
        int v = topo[ti];
        if (dist[v] == INF) continue;
        /* Relax all outgoing edges of v */
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1) {
                        int nd = dist[v] + g->weight[e];
                        if (nd < dist[u]) dist[u] = nd;
                    }
                }
            }
        }
    }

    /* Print results */
    for (i = 0; i < g->vcount; i++)
        printf("%s→%s = %d\n", g->vname[src],
               g->vname[i],
               dist[i] == INF ? -1 : dist[i]);
}

Longest Path in DAG

The longest path is critical for project scheduling — it gives the minimum time to complete all tasks. Also uses topological sort + relaxation, but maximizes instead of minimizes.

🏗️ Real-Life Analogy

Building a house: the longest sequence of dependent tasks (foundation → walls → roof) determines the minimum project completion time, even if shorter tasks run in parallel.

C — Longest Path in DAG
/* Longest path from src — same idea as shortest,
   but initialize to -INF and maximize */
void longest_path_dag(DAG *g, int src) {
    int dist[MAX_V], topo[MAX_V], tlen = 0;
    int i;
    for (i = 0; i < g->vcount; i++) dist[i] = -INF;
    dist[src] = 0;

    topo_sort_dfs(g, topo, &tlen);

    int ti;
    for (ti = 0; ti < tlen; ti++) {
        int v = topo[ti];
        if (dist[v] == -INF) continue;
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1) {
                        int nd = dist[v] + g->weight[e];
                        if (nd > dist[u])  /* MAX, not min */
                            dist[u] = nd;
                    }
                }
            }
        }
    }
    for (i = 0; i < g->vcount; i++)
        printf("Longest %s→%s = %d\n",
               g->vname[src], g->vname[i], dist[i]);
}

All Paths

Enumerate all paths from source s to destination t using backtracking DFS.

C — Find All Paths
/* Print all paths from s to t */
void all_paths(DAG *g, int s, int t,
               int path[], int plen,
               int vis[]) {
    vis[s] = 1;
    path[plen++] = s;

    if (s == t) {
        /* Print current path */
        int i;
        for (i = 0; i < plen; i++) {
            printf("%s", g->vname[path[i]]);
            if (i < plen-1) printf("→");
        }
        printf("\n");
    } else {
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[s][e] == +1) {
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1
                        && !vis[u]) {
                        all_paths(g, u, t, path,
                                  plen, vis);
                    }
                }
            }
        }
    }
    /* Backtrack */
    vis[s] = 0;
}

Dependency Chain Analysis

Find the complete chain of dependencies for a given vertex — all ancestors that must complete first.

📦 Real-Life Analogy

To install software package X, you need libraries A, B, C. But A also needs D and E. Dependency chain analysis reveals the full list of things to install first.

C — Dependency Chain Analysis
/* Print all ancestors of vertex v (things v depends on) */
void dep_chain(DAG *g, int v,
               int vis[], int depth) {
    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == -1) { /* v is dest */
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == +1
                    && !vis[u]) {
                    vis[u] = 1;
                    /* indent to show depth */
                    int d;
                    for (d=0; d<depth; d++) printf("  ");
                    printf("↳ %s\n", g->vname[u]);
                    dep_chain(g, u, vis, depth+1);
                }
            }
        }
    }
}

Topological Sort (DFS-Based)

DFS-based topological sort: run DFS, pushing each vertex to a stack after all its descendants are processed. Pop the stack for the final order.

📚 Real-Life Analogy

Organizing a study plan: you can't study Chapter 5 before Chapter 2. DFS-based topological sort automatically figures out the correct reading order for any textbook.

Topological Sort — DFS Animation
Click "Run Topo Sort" to see DFS-based topological sort.
Result: —
C — Topological Sort (DFS)
static int stk[MAX_V], stop = 0;

void topo_dfs_helper(DAG *g, int v, int vis[]) {
    vis[v] = 1;
    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == +1) {
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == -1
                    && !vis[u]) {
                    topo_dfs_helper(g, u, vis);
                }
            }
        }
    }
    stk[stop++] = v;  /* push AFTER all descendants */
}

/* Main topological sort — result in topo[] */
void topo_sort_dfs(DAG *g, int topo[], int *tlen) {
    int vis[MAX_V];
    memset(vis, 0, sizeof(vis));
    stop = 0;
    int v;
    for (v = 0; v < g->vcount; v++)
        if (!vis[v]) topo_dfs_helper(g, v, vis);

    /* Stack is in reverse order */
    *tlen = stop;
    int i;
    for (i = 0; i < stop; i++)
        topo[i] = stk[stop - 1 - i];
}

Topological Sort — Kahn's Algorithm

Kahn's algorithm uses in-degree counts. Repeatedly pick a vertex with in-degree 0, output it, and reduce in-degrees of its successors. Also detects cycles!

🏭 Real-Life Analogy

An assembly line: always pick the task that has no pending prerequisites (in-degree 0). Complete it, cross it off all checklists, then see which new tasks became available.

Kahn's Algorithm — Step by Step
Click "Run Kahn's" to see the algorithm step by step.
C — Kahn's Algorithm
void kahns_topo_sort(DAG *g) {
    int indeg[MAX_V], queue[MAX_V];
    int front = 0, rear = 0, count = 0;

    compute_indegree(g, indeg);

    /* Enqueue all zero in-degree vertices */
    int v;
    for (v = 0; v < g->vcount; v++)
        if (indeg[v] == 0) queue[rear++] = v;

    printf("Kahn's Topo Order: ");
    while (front < rear) {
        v = queue[front++];
        printf("%s ", g->vname[v]);
        count++;
        /* Reduce in-degree of all successors */
        int e, u;
        for (e = 0; e < g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u = 0; u < g->vcount; u++) {
                    if (g->matrix[u][e] == -1) {
                        indeg[u]--;
                        if (indeg[u] == 0)
                            queue[rear++] = u;
                    }
                }
            }
        }
    }
    printf("\n");

    /* If count != V, graph has a cycle */
    if (count != g->vcount)
        printf("WARNING: Cycle detected!\n");
}

Cycle Detection

A cycle in a directed graph means you can follow edges and return to the starting vertex. This violates the "Acyclic" property of a DAG. We detect cycles using DFS with a "currently in stack" marker.

/* Color: 0=white(unvisited), 1=gray(in stack), 2=black(done) */
static int color[MAX_V];

static int dfs_color(DAG *g, int v) {
    color[v] = 1;  /* gray: currently visiting */
    int e, u;
    for (e = 0; e < g->ecount; e++) {
        if (g->matrix[v][e] == +1) {
            for (u = 0; u < g->vcount; u++) {
                if (g->matrix[u][e] == -1) {
                    if (color[u] == 1)
                        return 1;  /* back edge = CYCLE */
                    if (color[u] == 0)
                        if (dfs_color(g, u))
                            return 1;
                }
            }
        }
    }
    color[v] = 2;  /* black: fully processed */
    return 0;
}

    memset(color, 0, sizeof(color));
    int v;
    for (v = 0; v < g->vcount; v++)
        if (color[v] == 0)
            if (dfs_color(g, v)) return 1;
    return 0;
}

DAG Validation

A complete DAG validator checks: (1) all edges are directed, (2) no self-loops, (3) no cycles.

C — Full DAG Validation
/* Comprehensive DAG validation */
void validate_full(DAG *g) {
    int ok = 1;
    printf("=== DAG Validation ===\n");

    /* Check 1: each edge column has exactly one +1 and one -1 */
    int e, v, pos, neg;
    for (e = 0; e < g->ecount; e++) {
        pos = neg = 0;
        for (v = 0; v < g->vcount; v++) {
            if (g->matrix[v][e] == +1) pos++;
            if (g->matrix[v][e] == -1) neg++;
        }
        if (pos != 1 || neg != 1) {
            printf("FAIL: Edge %d malformed!\n", e);
            ok = 0;
        }
    }

    /* Check 2: no self-loops (same vertex +1 and -1 in col) */
    for (e = 0; e < g->ecount; e++)
        for (v = 0; v < g->vcount; v++)
            if (g->matrix[v][e] == +1 &&
                g->matrix[v][e] == -1) {  /* impossible but explicit */
                printf("FAIL: Self-loop at vertex %s\n",
                       g->vname[v]);
                ok = 0;
            }

    /* Check 3: no cycles */
    if (detect_cycle(g)) {
        printf("FAIL: Cycle detected!\n"); ok = 0;
    }

    if (ok) printf("PASS: Valid DAG!\n");
}
08 — CONNECTIVITY OPERATIONS

Reachability Check

Can vertex u reach vertex v by following directed edges? Build a reachability matrix using transitive closure.

📡 Real-Life Analogy

Can information flow from department A to department Z in a company's communication chain? Reachability analysis answers this for all pairs at once.

C — Reachability Matrix (Transitive Closure)
/* reach[i][j] = 1 if vertex i can reach vertex j */
void build_reachability(DAG *g, int reach[][MAX_V]) {
    int i, j, e, u;

    /* Initialize: reach[i][i] = 1 (every node reaches itself) */
    for (i=0; i<g->vcount; i++)
        for (j=0; j<g->vcount; j++)
            reach[i][j] = (i==j) ? 1 : 0;

    /* Fill direct edges */
    for (e=0; e<g->ecount; e++) {
        int src=-1, dst=-1;
        for (i=0; i<g->vcount; i++) {
            if (g->matrix[i][e]==+1) src=i;
            if (g->matrix[i][e]==-1) dst=i;
        }
        if (src!=-1 && dst!=-1) reach[src][dst]=1;
    }

    /* Warshall's algorithm for transitive closure */
    int k;
    for (k=0; k<g->vcount; k++)
        for (i=0; i<g->vcount; i++)
            for (j=0; j<g->vcount; j++)
                if (reach[i][k] && reach[k][j])
                    reach[i][j] = 1;
}

    int ra=find_root(a), rb=find_root(b);

            if(g->matrix[v][e]==-1) dst=v;

/* Group tasks into parallel execution levels
   Level 0: no dependencies
   Level 1: depends only on level 0, etc.  */
void parallel_schedule(DAG *g) {
    int level[MAX_V], indeg[MAX_V];
    compute_indegree(g, indeg);
    int v, e, u, max_level = 0;

    for (v=0; v<g->vcount; v++) level[v] = 0;

    /* Process in Kahn's order, assign levels */
    int queue[MAX_V], front=0, rear=0;
    for (v=0; v<g->vcount; v++)
        if (indeg[v]==0) queue[rear++] = v;

    while (front < rear) {
        v = queue[front++];
        for (e=0; e<g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u=0; u<g->vcount; u++) {
                    if (g->matrix[u][e]==-1) {
                        if(level[v]+1 > level[u])
                            level[u] = level[v]+1;
                        indeg[u]--;
                        if(indeg[u]==0) queue[rear++]=u;
                    }
                }
            }
        }
        if(level[v] > max_level) max_level = level[v];
    }

    int l;
    for (l=0; l<=max_level; l++) {
        printf("Level %d [parallel]: ", l);
        for(v=0; v<g->vcount; v++)
            if(level[v]==l) printf("%s ", g->vname[v]);
        printf("\n");
    }
}

Critical Path Analysis

The critical path is the longest path through the DAG. It determines the minimum project duration. Tasks on the critical path have zero slack — any delay cascades to the end.

🏗️ Real-Life Analogy

In building construction, pouring concrete must cure for 7 days before framing. This is on the critical path. Painting walls (can happen while other things proceed) might not be critical.

C — Critical Path Analysis (Earliest Start Times)
/* Compute Earliest Start Time (EST) for each vertex.
   EST[v] = max(EST[u] + duration[edge u→v])    */
void critical_path(DAG *g) {
    int est[MAX_V], topo[MAX_V], tlen=0;
    int v, e, u;

    for (v=0; v<g->vcount; v++) est[v] = 0;
    topo_sort_dfs(g, topo, &tlen);

    int ti;
    for (ti=0; ti<tlen; ti++) {
        v = topo[ti];
        for (e=0; e<g->ecount; e++) {
            if (g->matrix[v][e] == +1) {
                for (u=0; u<g->vcount; u++) {
                    if (g->matrix[u][e] == -1) {
                        int t = est[v]+g->weight[e];
                        if (t > est[u]) est[u] = t;
                    }
                }
            }
        }
    }

    printf("Earliest Start Times:\n");
    for (v=0; v<g->vcount; v++)
        printf("  %s: %d\n", g->vname[v], est[v]);

    /* Critical path = path leading to max EST */
    int max_est=0;
    for(v=0;v<g->vcount;v++)
        if(est[v]>max_est) max_est=est[v];
    printf("Project duration: %d\n", max_est);
}

Critical Path Method (CPM)

CPM computes both Earliest Start Time (EST) and Latest Start Time (LST). Slack = LST - EST. Nodes with slack = 0 are on the critical path.

C — Full CPM (EST + LST + Slack)
void cpm(DAG *g) {
    int est[MAX_V], lst[MAX_V];
    int topo[MAX_V], tlen=0;
    int v, e, u, ti;

    topo_sort_dfs(g, topo, &tlen);

    /* Forward pass: compute EST */
    for(v=0;v<g->vcount;v++) est[v]=0;
    for(ti=0;ti<tlen;ti++) {
        v=topo[ti];
        for(e=0;e<g->ecount;e++) {
            if(g->matrix[v][e]==+1) {
                for(u=0;u<g->vcount;u++) {
                    if(g->matrix[u][e]==-1) {
                        int t=est[v]+g->weight[e];
                        if(t>est[u]) est[u]=t;
                    }
                }
            }
        }
    }
    int T = 0;
    for(v=0;v<g->vcount;v++) if(est[v]>T) T=est[v];

    /* Backward pass: compute LST */
    for(v=0;v<g->vcount;v++) lst[v]=T;
    for(ti=tlen-1;ti>=0;ti--) {
        v=topo[ti];
        for(e=0;e<g->ecount;e++) {
            if(g->matrix[v][e]==+1) {
                for(u=0;u<g->vcount;u++) {
                    if(g->matrix[u][e]==-1) {
                        int t=lst[u]-g->weight[e];
                        if(t<lst[v]) lst[v]=t;
                    }
                }
            }
        }
    }

    printf("%-8s %5s %5s %6s %s\n",
           "Vertex","EST","LST","Slack","Critical?");
    for(v=0;v<g->vcount;v++) {
        int slack = lst[v]-est[v];
        printf("%-8s %5d %5d %6d %s\n",
               g->vname[v], est[v], lst[v], slack,
               slack==0 ? "*** CRITICAL ***" : "");
    }
}

    for(e=0;e<g->ecount;e++) printf(" e%d", e);
    printf("\n");

    for(v=0;v<g->vcount;v++) {
        printf("%-8s", g->vname[v]);
        for(e=0;e<g->ecount;e++)
            printf(" %2d", g->matrix[v][e]);
        printf("\n");
    }
}

    }

/* Print dependency tree using ASCII art */
void print_dep_tree(DAG *g, int v,
                    char *prefix,
                    int is_last,
                    int vis[]) {
    printf("%s%s%s\n", prefix,
           is_last ? "└─ " : "├─ ", g->vname[v]);
    vis[v] = 1;

    /* Collect predecessors */
    int preds[MAX_V], np=0, e, u;
    for(e=0;e<g->ecount;e++)
        if(g->matrix[v][e]==-1)
            for(u=0;u<g->vcount;u++)
                if(g->matrix[u][e]==+1&&!vis[u])
                    preds[np++]=u;

    char new_prefix[200];
    int i;
    for(i=0;i<np;i++) {
        sprintf(new_prefix, "%s%s", prefix,
                is_last ? "   " : "│  ");
        print_dep_tree(g, preds[i], new_prefix,
                       i==np-1, vis);
    }
}

/* Generate full execution schedule with timing */
void execution_schedule(DAG *g) {
    int est[MAX_V], level[MAX_V];
    int indeg[MAX_V], queue[MAX_V];
    int front=0, rear=0, v, e, u;

    compute_indegree(g, indeg);
    for(v=0;v<g->vcount;v++) {
        est[v]=0; level[v]=0;
        if(indeg[v]==0) queue[rear++]=v;
    }

    while(front<rear) {
        v=queue[front++];
        for(e=0;e<g->ecount;e++) {
            if(g->matrix[v][e]==+1) {
                for(u=0;u<g->vcount;u++) {
                    if(g->matrix[u][e]==-1) {
                        int t=est[v]+g->weight[e];
                        if(t>est[u]) est[u]=t;
                        if(level[v]+1>level[u])
                            level[u]=level[v]+1;
                        indeg[u]--;
                        if(indeg[u]==0) queue[rear++]=u;
                    }
                }
            }
        }
    }

    printf("%-4s %-12s %5s %6s\n",
           "Lv","Task","EST","Status");
    printf("-------------------------------\n");
    int max_lv=0;
    for(v=0;v<g->vcount;v++)
        if(level[v]>max_lv) max_lv=level[v];

    int l;
    for(l=0;l<=max_lv;l++) {
        for(v=0;v<g->vcount;v++) {
            if(level[v]==l)
                printf("%4d %-12s %5d [READY]\n",
                       l, g->vname[v], est[v]);
        }
    }
}
