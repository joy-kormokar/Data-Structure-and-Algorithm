What is a Weighted Graph?

Beginner

A graph is a collection of nodes (vertices) connected by edges. A weighted graph adds a number (weight/cost) to each edge — representing distance, time, cost, or capacity.

🗺️
Real-life Analogy

Think of a map of cities (vertices) connected by roads (edges). The distance in km between each city is the weight. To find the shortest travel route, we need those weights!

Vertices (Nodes)

Points in the graph. Can represent cities, computers, people, states — anything that connects to something else.

Edges (Links)

Connections between vertices. Can be directed (one-way street) or undirected (two-way street).

Weights

A number on each edge — could be distance (km), cost ($), time (min), bandwidth (Mbps), or any measurable value.

Adjacency Matrix

A 2D array where cell [i][j] stores the weight of the edge from vertex i to vertex j. Zero (or ∞) means no edge.

Interactive Weighted Graph Explorer

C
#define MAX 10
#define INF 99999   /* represents no edge (infinity) */

/* A weighted graph using Adjacency Matrix */
struct Graph {
    int vertices;                  /* total number of vertices */
    int matrix[MAX][MAX];           /* weight stored at [i][j] */
    char label[MAX];               /* optional: vertex names A,B,C... */
};

/* Initialize an empty graph */
void initGraph(struct Graph *g, int v) {
    int i, j;
    g->vertices = v;
    for (i = 0; i < v; i++) {
        g->label[i] = 'A' + i;       /* name vertices A, B, C... */
        for (j = 0; j < v; j++) {
            g->matrix[i][j] = 0;     /* 0 means no edge */
        }
    }
}

Vertex Operations

Beginner

Vertices are the nodes of your graph. You can insert, delete, search, and update them. Each operation modifies the adjacency matrix accordingly.

🏙️
City Network Analogy

Adding a vertex is like building a new city. Initially it has no roads (edges) to other cities. Deleting a vertex is demolishing a city — all roads to/from it disappear too.

Insert Vertex

Adding a new vertex increases the matrix size by 1 row and 1 column, all initialized to 0 (no edges).

Vertex Operations Visualizer

Graph has 4 vertices. Try inserting or deleting a vertex.
C — Insert, Delete, Search, Update Vertex
/* INSERT VERTEX — adds a new vertex, extends matrix */
void insertVertex(struct Graph *g) {
    int v = g->vertices;
    if (v >= MAX) { printf("Graph is full!\n"); return; }
    g->label[v] = 'A' + v;
    for (int i = 0; i <= v; i++) {
        g->matrix[v][i] = 0;   /* new row: all zeros */
        g->matrix[i][v] = 0;   /* new column: all zeros */
    }
    g->vertices++;
    printf("Vertex %c inserted!\n", g->label[v]);
}

/* DELETE VERTEX — removes vertex and all its edges */
void deleteVertex(struct Graph *g, int u) {
    for (int i = u; i < g->vertices - 1; i++) {
        g->label[i] = g->label[i + 1];
        for (int j = 0; j < g->vertices; j++) {
            g->matrix[i][j] = g->matrix[i + 1][j];
            g->matrix[j][i] = g->matrix[j][i + 1];
        }
    }
    g->vertices--;
}

/* SEARCH VERTEX — check if vertex index is valid */
int searchVertex(struct Graph *g, int u) {
    return (u >= 0 && u < g->vertices);
}

/* UPDATE VERTEX — rename the label */
void updateVertex(struct Graph *g, int u, char newLabel) {
    if (searchVertex(g, u))
        g->label[u] = newLabel;
}

Traversal Operations

Beginner

Traversal means visiting every vertex in a systematic order. BFS explores level by level (nearest neighbours first), DFS goes as deep as possible before backtracking.

🌊
BFS = Ripples in a pond. DFS = Exploring a maze.

BFS spreads outward like ripples from a stone — visiting all friends, then friends-of-friends. DFS is like solving a maze — go as far as you can, hit a dead end, backtrack, try another path.

C — BFS on Weighted Graph
/* BREADTH-FIRST SEARCH using a simple array as queue */
void BFS(struct Graph *g, int start) {
    int visited[MAX] = {0};    /* track visited vertices */
    int queue[MAX], front = 0, rear = 0;

    visited[start] = 1;
    queue[rear++] = start;

    printf("BFS: ");
    while (front < rear) {
        int u = queue[front++];
        printf("%c ", g->label[u]);

        for (int v = 0; v < g->vertices; v++) {
            if (g->matrix[u][v] != 0 && !visited[v]) {
                visited[v] = 1;
                queue[rear++] = v;
            }
        }
    }
}

/* DEPTH-FIRST SEARCH (recursive) */
void DFS(struct Graph *g, int u, int visited[]) {
    visited[u] = 1;
    printf("%c ", g->label[u]);
    for (int v = 0; v < g->vertices; v++) {
        if (g->matrix[u][v] != 0 && !visited[v])
            DFS(g, v, visited);
    }
}

/* Weighted Traversal: also track cumulative cost */
void weightedBFS(struct Graph *g, int start) {
    int visited[MAX] = {0}, cost[MAX] = {0};
    int queue[MAX], front = 0, rear = 0;
    visited[start] = 1;
    queue[rear++] = start;
    while (front < rear) {
        int u = queue[front++];
        printf("%c(cost:%d) ", g->label[u], cost[u]);
        for (int v = 0; v < g->vertices; v++) {
            if (g->matrix[u][v] && !visited[v]) {
                visited[v] = 1;
                cost[v] = cost[u] + g->matrix[u][v];
                queue[rear++] = v;
            }
        }
    }
}

C — Find Path, All Paths, Shortest & Longest
/* FIND PATH — does a path exist from src to dst? (DFS) */
int findPath(struct Graph *g, int src, int dst, int visited[]) {
    if (src == dst) return 1;   /* reached destination */
    visited[src] = 1;
    for (int v = 0; v < g->vertices; v++) {
        if (g->matrix[src][v] && !visited[v])
            if (findPath(g, v, dst, visited)) return 1;
    }
    return 0;
}

/* ALL PATHS — print every path from src to dst */
void allPaths(struct Graph *g, int src, int dst,
              int path[], int pLen, int visited[]) {
    path[pLen++] = src;
    visited[src] = 1;
    if (src == dst) {
        for (int i = 0; i < pLen; i++)
            printf("%c%s", 'A'+path[i], i<pLen-1?"→":"\n");
    } else {
        for (int v = 0; v < g->vertices; v++)
            if (g->matrix[src][v] && !visited[v])
                allPaths(g, v, dst, path, pLen, visited);
    }
    visited[src] = 0;   /* backtrack */
}

/* MINIMUM COST PATH — track cumulative weight */
int minCost;
void minCostPath(struct Graph *g, int src, int dst,
                 int visited[], int curCost) {
    if (src == dst) {
        if (curCost < minCost) minCost = curCost;
        return;
    }
    visited[src] = 1;
    for (int v = 0; v < g->vertices; v++)
        if (g->matrix[src][v] && !visited[v])
            minCostPath(g, v, dst, visited, curCost + g->matrix[src][v]);
    visited[src] = 0;
}

Dijkstra's Algorithm

Intermediate

Dijkstra finds the shortest path from one source vertex to all other vertices. It always picks the unvisited vertex with the smallest known distance — greedy approach. Works only with non-negative weights.

📍
Spreading Rumor Analogy

Imagine spreading a rumor starting from you. The rumor spreads to your closest friend first, then their closest friends — always reaching the "cheapest" person next. That's Dijkstra!

Dijkstra's Algorithm — Step by Step

Click "Run Dijkstra" to find shortest paths from source vertex.
Time: O(V²) with matrix Space: O(V) Negative weights: ❌ No Use case: GPS, Maps, Network routing
C — Dijkstra's Algorithm
void dijkstra(struct Graph *g, int src) {
    int dist[MAX], visited[MAX], parent[MAX];
    int i, v, u, minDist, n = g->vertices;

    /* Step 1: Initialize all distances to infinity */
    for (i = 0; i < n; i++) {
        dist[i] = INF;
        visited[i] = 0;
        parent[i] = -1;
    }
    dist[src] = 0;   /* distance to self is 0 */

    for (int count = 0; count < n - 1; count++) {
        /* Step 2: Pick unvisited vertex with minimum distance */
        minDist = INF; u = -1;
        for (v = 0; v < n; v++)
            if (!visited[v] && dist[v] < minDist) {
                minDist = dist[v]; u = v;
            }
        if (u == -1) break;
        visited[u] = 1;

        /* Step 3: Relax neighbours */
        for (v = 0; v < n; v++) {
            if (g->matrix[u][v] && !visited[v] &&
                dist[u] + g->matrix[u][v] < dist[v]) {
                dist[v] = dist[u] + g->matrix[u][v];
                parent[v] = u;
            }
        }
    }

    /* Print shortest distances */
    for (v = 0; v < n; v++)
        printf("%c to %c: %d\n", g->label[src], g->label[v], dist[v]);
}

Bellman-Ford Algorithm

Intermediate

Bellman-Ford handles negative edge weights and detects negative cycles. It relaxes all edges V-1 times. Slower than Dijkstra but more general.

💳
Currency Exchange Analogy

In finance, some currency conversions can have "negative cost" (you gain money). Bellman-Ford handles this. If converting currencies in a loop profits you, that's a negative cycle!

Bellman-Ford Step Visualizer

Bellman-Ford handles negative weights. Click Run to start.
Time: O(V × E) Negative weights: ✅ Yes Detects negative cycles: ✅ Yes
C — Bellman-Ford Algorithm
void bellmanFord(struct Graph *g, int src) {
    int dist[MAX], i, u, v, n = g->vertices;

    for (i = 0; i < n; i++) dist[i] = INF;
    dist[src] = 0;

    /* Relax all edges V-1 times */
    for (int iter = 0; iter < n - 1; iter++) {
        for (u = 0; u < n; u++) {
            for (v = 0; v < n; v++) {
                if (g->matrix[u][v] && dist[u] != INF &&
                    dist[u] + g->matrix[u][v] < dist[v]) {
                    dist[v] = dist[u] + g->matrix[u][v];
                }
            }
        }
    }

    /* Check for negative cycle (V-th relaxation still changes) */
    for (u = 0; u < n; u++)
        for (v = 0; v < n; v++)
            if (g->matrix[u][v] && dist[u] != INF &&
                dist[u] + g->matrix[u][v] < dist[v])
                printf("Negative cycle detected!\n");
}

Floyd-Warshall Algorithm

Intermediate

Floyd-Warshall finds shortest paths between ALL pairs of vertices in one pass. It considers every vertex as a possible intermediate step.

✈️
Flight Connections Analogy

Is it cheaper to fly A→C directly, or via a connecting flight through B? Floyd-Warshall checks every possible connecting city and fills in the cheapest cost for every city pair.

Floyd-Warshall Distance Matrix

Click Run to compute all-pairs shortest paths.
Time: O(V³) Space: O(V²) Negative weights: ✅ Yes All pairs: ✅ Yes
C — Floyd-Warshall
void floydWarshall(struct Graph *g) {
    int dist[MAX][MAX], i, j, k, n = g->vertices;

    /* Initialize dist with the adjacency matrix */
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++) {
            if (i == j) dist[i][j] = 0;
            else if (g->matrix[i][j]) dist[i][j] = g->matrix[i][j];
            else dist[i][j] = INF;
        }

    /* Try every vertex k as an intermediate */
    for (k = 0; k < n; k++)
        for (i = 0; i < n; i++)
            for (j = 0; j < n; j++)
                if (dist[i][k] != INF && dist[k][j] != INF &&
                    dist[i][k] + dist[k][j] < dist[i][j])
                    dist[i][j] = dist[i][k] + dist[k][j];
}

A* Search Algorithm

Advanced

A* combines Dijkstra with a heuristic (estimated distance to goal) to find the shortest path more efficiently. It avoids exploring paths that are clearly going in the wrong direction.

🎯
Smart GPS Analogy

A smart GPS doesn't explore roads heading away from your destination. It uses your phone's GPS to estimate how far you are from the goal and prioritizes paths that get you closer. That's the heuristic!

A* Search — f(n) = g(n) + h(n)

A* uses f = g (cost so far) + h (estimated remaining cost). Click Run.
C — A* Search (simplified)
/* Heuristic: estimate of remaining cost (must be admissible — never overestimates) */
int heuristic(int u, int goal) {
    /* Simple example: use pre-defined estimates */
    int h[] = {6, 4, 2, 3, 0};   /* h[i] = estimate from i to goal */
    return h[u];
}

void aStar(struct Graph *g, int src, int goal) {
    int gCost[MAX], fCost[MAX], visited[MAX] = {0};
    int n = g->vertices, u, v;

    for (int i = 0; i < n; i++) {
        gCost[i] = INF;
        fCost[i] = INF;
    }
    gCost[src] = 0;
    fCost[src] = heuristic(src, goal);

    while (1) {
        /* Pick unvisited vertex with lowest fCost */
        int minF = INF; u = -1;
        for (v = 0; v < n; v++)
            if (!visited[v] && fCost[v] < minF) { minF = fCost[v]; u = v; }
        if (u == -1 || u == goal) break;
        visited[u] = 1;

        for (v = 0; v < n; v++) {
            if (g->matrix[u][v] && !visited[v]) {
                int newG = gCost[u] + g->matrix[u][v];
                if (newG < gCost[v]) {
                    gCost[v] = newG;
                    fCost[v] = newG + heuristic(v, goal);
                }
            }
        }
    }
    printf("A* shortest cost: %d\n", gCost[goal]);
}

C — Connected Check, Components, Isolation
/* CONNECTED CHECK — graph is connected if BFS visits all vertices */
int isConnected(struct Graph *g) {
    int visited[MAX] = {0}, queue[MAX], f = 0, r = 0, count = 0;
    visited[0] = 1; queue[r++] = 0;
    while (f < r) {
        int u = queue[f++]; count++;
        for (int v = 0; v < g->vertices; v++)
            if (g->matrix[u][v] && !visited[v]) { visited[v]=1; queue[r++]=v; }
    }
    return count == g->vertices;
}

/* COMPONENTS DETECTION — find all disconnected components */
int findComponents(struct Graph *g) {
    int visited[MAX] = {0}, components = 0;
    for (int i = 0; i < g->vertices; i++) {
        if (!visited[i]) {
            components++;
            /* BFS from this unvisited vertex */
            int q[MAX], f = 0, r = 0;
            visited[i] = 1; q[r++] = i;
            while (f < r) {
                int u = q[f++];
                for (int v = 0; v < g->vertices; v++)
                    if (g->matrix[u][v] && !visited[v]) { visited[v]=1; q[r++]=v; }
            }
        }
    }
    return components;
}

/* ISOLATION CHECK — vertex with no edges */
int isIsolated(struct Graph *g, int u) {
    for (int v = 0; v < g->vertices; v++)
        if (g->matrix[u][v] != 0) return 0;
    return 1;
}

Cycle Operations

Intermediate

A cycle is a path that starts and ends at the same vertex. Detecting cycles is crucial for avoiding infinite loops in routing and checking if a graph is a tree.

🔄
Traffic Roundabout Analogy

A cycle in a road network is like a roundabout — you can keep going in circles. In a delivery network, a cycle means packages could be routed in circles, wasting fuel (cycle cost analysis).

Cycle Detection Visualizer

Click "Detect Cycle" to check for cycles using DFS back-edge detection.
C — Cycle Detection & Weighted Cycle Cost
/* DETECT CYCLE using DFS (undirected graph) */
int dfsHasCycle(struct Graph *g, int u, int parent, int visited[]) {
    visited[u] = 1;
    for (int v = 0; v < g->vertices; v++) {
        if (g->matrix[u][v] == 0) continue;
        if (!visited[v]) {
            if (dfsHasCycle(g, v, u, visited)) return 1;
        } else if (v != parent) {
            return 1;   /* back edge found — cycle! */
        }
    }
    return 0;
}

int hasCycle(struct Graph *g) {
    int visited[MAX] = {0};
    for (int i = 0; i < g->vertices; i++)
        if (!visited[i] && dfsHasCycle(g, i, -1, visited))
            return 1;
    return 0;
}

/* WEIGHTED CYCLE COST — find cycle and sum its edge weights */
int cycleCost(struct Graph *g, int start) {
    int path[MAX], pLen = 0, cost = 0, visited[MAX] = {0};
    path[pLen++] = start;
    visited[start] = 1;
    int u = start;
    while (1) {
        int moved = 0;
        for (int v = 0; v < g->vertices; v++) {
            if (g->matrix[u][v] && !visited[v]) {
                cost += g->matrix[u][v];
                path[pLen++] = v;
                visited[v] = 1;
                u = v; moved = 1; break;
            }
        }
        if (!moved) { /* try to close cycle back to start */
            if (g->matrix[u][start]) cost += g->matrix[u][start];
            break;
        }
    }
    return cost;
}

Degree Operations

Beginner

The degree of a vertex is the number of edges connected to it. Weighted degree sums all edge weights. Total edge cost sums all weights in the graph.

📡
Social Network Analogy

Degree = how many friends you have. Weighted degree = total "closeness" with all friends combined (e.g., hours spent together). A celebrity has high degree; a close-knit friend group has high weighted degree.

Degree Analysis Dashboard

Click an operation to analyze vertex degrees.
C — Degree Analysis
/* DEGREE — count edges of vertex u */
int degree(struct Graph *g, int u) {
    int deg = 0;
    for (int v = 0; v < g->vertices; v++)
        if (g->matrix[u][v] != 0) deg++;
    return deg;
}

/* WEIGHTED DEGREE — sum of all edge weights for vertex u */
int weightedDegree(struct Graph *g, int u) {
    int wDeg = 0;
    for (int v = 0; v < g->vertices; v++)
        wDeg += g->matrix[u][v];
    return wDeg;
}

/* TOTAL EDGE COST — sum of all weights in the graph */
int totalEdgeCost(struct Graph *g) {
    int total = 0;
    for (int i = 0; i < g->vertices; i++)
        for (int j = i + 1; j < g->vertices; j++)
            total += g->matrix[i][j];   /* count each edge once */
    return total;
}

/* DEGREE ANALYSIS — print all vertices with their degrees */
void degreeAnalysis(struct Graph *g) {
    printf("%-8s %-8s %s\n", "Vertex", "Degree", "Wtd Degree");
    for (int i = 0; i < g->vertices; i++)
        printf("%-8c %-8d %d\n", g->label[i], degree(g,i), weightedDegree(g,i));
}

Minimum Spanning Tree (MST)

Intermediate

An MST connects all vertices using the minimum total edge weight, with no cycles. Prim's builds the MST vertex by vertex; Kruskal's adds edges in sorted order.

💡
Electrical Grid Analogy

Imagine connecting all houses in a city with electrical cable at minimum cost. You don't need redundant connections (no cycles). MST gives you the cheapest network that reaches every house!

MST Visualizer — Prim's & Kruskal's

Choose Prim's or Kruskal's to find the Minimum Spanning Tree.
C — Prim's Algorithm
void prim(struct Graph *g) {
    int inMST[MAX] = {0}, key[MAX], parent[MAX];
    int n = g->vertices, u, v, totalCost = 0;

    for (int i = 0; i < n; i++) { key[i] = INF; parent[i] = -1; }
    key[0] = 0;   /* start from vertex 0 */

    for (int count = 0; count < n; count++) {
        /* Pick minimum key vertex not yet in MST */
        int minKey = INF; u = -1;
        for (v = 0; v < n; v++)
            if (!inMST[v] && key[v] < minKey) { minKey = key[v]; u = v; }
        if (u == -1) break;
        inMST[u] = 1;
        totalCost += key[u];

        /* Update keys of adjacent vertices */
        for (v = 0; v < n; v++)
            if (g->matrix[u][v] && !inMST[v] && g->matrix[u][v] < key[v]) {
                key[v] = g->matrix[u][v];
                parent[v] = u;
            }
    }
    printf("MST Total Cost: %d\n", totalCost);
}
C — Kruskal's Algorithm (Union-Find)
/* UNION-FIND for cycle detection */
int parent_uf[MAX];
int find(int x) { return parent_uf[x]==x ? x : (parent_uf[x]=find(parent_uf[x])); }
void unite(int x, int y) { parent_uf[find(x)] = find(y); }

/* Edge list for Kruskal */
struct Edge { int u, v, w; };

int cmpEdge(const void*a, const void*b) {
    return ((struct Edge*)a)->w - ((struct Edge*)b)->w;
}

void kruskal(struct Graph *g) {
    struct Edge edges[MAX*MAX]; int eCount = 0;
    for (int i=0; i<g->vertices; i++)
        for (int j=i+1; j<g->vertices; j++)
            if (g->matrix[i][j]) { edges[eCount].u=i; edges[eCount].v=j; edges[eCount++].w=g->matrix[i][j]; }
    qsort(edges, eCount, sizeof(struct Edge), cmpEdge);
    for (int i=0; i<g->vertices; i++) parent_uf[i] = i;
    int total = 0;
    for (int i=0; i<eCount; i++) {
        if (find(edges[i].u) != find(edges[i].v)) {
            unite(edges[i].u, edges[i].v);
            total += edges[i].w;
            printf("%c-%c: %d\n", 'A'+edges[i].u, 'A'+edges[i].v, edges[i].w);
        }
    }
    printf("Total MST Cost: %d\n", total);
}

/* Greedy Nearest Neighbour for TSP-like route optimization */
void routeOptimization(struct Graph *g, int start) {
    int visited[MAX] = {0}, route[MAX], rLen = 0;
    int cur = start, totalCost = 0, n = g->vertices;

    route[rLen++] = cur;
    visited[cur] = 1;

    while (rLen < n) {
        /* Pick nearest unvisited neighbour */
        int minW = INF, next = -1;
        for (int v = 0; v < n; v++) {
            if (!visited[v] && g->matrix[cur][v] > 0 &&
                g->matrix[cur][v] < minW) {
                minW = g->matrix[cur][v]; next = v;
            }
        }
        if (next == -1) break;   /* no reachable vertex */
        totalCost += minW;
        route[rLen++] = next;
        visited[next] = 1;
        cur = next;
    }
    /* Return to start */
    totalCost += g->matrix[cur][start];
    printf("Optimized route cost: %d\n", totalCost);
}

/* Resource Allocation: distribute capacity proportional to edge weights */
void resourceAllocation(struct Graph *g, int src, int totalResource) {
    int total = weightedDegree(g, src);
    for (int v = 0; v < g->vertices; v++)
        if (g->matrix[src][v])
            printf("%c gets %d units\n", g->label[v],
                   totalResource * g->matrix[src][v] / total);
}

/* GRAPH COLORING — assign colors so no two adjacent vertices share a color */
void graphColoring(struct Graph *g) {
    int color[MAX], n = g->vertices, u, v;
    for (int i = 0; i < n; i++) color[i] = -1;
    color[0] = 0;   /* first vertex gets color 0 */

    for (u = 1; u < n; u++) {
        int used[MAX] = {0};
        for (v = 0; v < n; v++)
            if (g->matrix[u][v] && color[v] != -1)
                used[color[v]] = 1;   /* mark neighbour colors as used */
        int c = 0;
        while (used[c]) c++;   /* find smallest unused color */
        color[u] = c;
    }
    for (u = 0; u < n; u++)
        printf("%c → Color %d\n", g->label[u], color[u]);
}

/* BIPARTITE CHECK — 2-color with BFS */
int isBipartite(struct Graph *g) {
    int color[MAX], q[MAX], f=0, r=0, n=g->vertices, u, v;
    for (int i=0; i<n; i++) color[i]=-1;
    color[0]=0; q[r++]=0;
    while (f<r) {
        u=q[f++];
        for (v=0; v<n; v++) {
            if (g->matrix[u][v]) {
                if (color[v]==-1) { color[v]=1-color[u]; q[r++]=v; }
                else if (color[v]==color[u]) return 0; /* same color! */
            }
        }
    }
    return 1;
}
C — Network Flow (Ford-Fulkerson) & SCC (Kosaraju's)
/* NETWORK FLOW — Ford-Fulkerson: max flow from src to sink */
int bfsFlow(int rg[][MAX], int src, int sink, int parent[], int n) {
    int vis[MAX]={0}, q[MAX], f=0, r=0;
    vis[src]=1; q[r++]=src; parent[src]=-1;
    while (f<r) {
        int u=q[f++];
        for (int v=0; v<n; v++)
            if (!vis[v]&&rg[u][v]>0) { parent[v]=u; vis[v]=1; q[r++]=v; }
    }
    return vis[sink];
}
int maxFlow(struct Graph *g, int src, int sink) {
    int rg[MAX][MAX], parent[MAX], flow=0, u, v, n=g->vertices;
    for(u=0;u<n;u++) for(v=0;v<n;v++) rg[u][v]=g->matrix[u][v];
    while(bfsFlow(rg,src,sink,parent,n)) {
        int pathFlow=INF;
        for(v=sink;v!=src;v=parent[v]) { u=parent[v]; if(rg[u][v]<pathFlow)pathFlow=rg[u][v]; }
        for(v=sink;v!=src;v=parent[v]) { u=parent[v]; rg[u][v]-=pathFlow; rg[v][u]+=pathFlow; }
        flow+=pathFlow;
    }
    return flow;
}

/* SCC (Kosaraju's) — two-pass DFS on directed graph */
void dfsFinish(int g[][MAX], int u, int vis[], int stk[], int *top, int n) {
    vis[u]=1;
    for(int v=0;v<n;v++) if(g[u][v]&&!vis[v]) dfsFinish(g,v,vis,stk,top,n);
    stk[(*top)++]=u;
}
