01
Initialization Operations
// create persistent skip list · initialize header · set max level · configure persistence
📖Theory Explanation
Before any data can be stored, a Persistent Skip List must be initialized. This sets up the skeleton of the structure.

A header node acts as the entry point at every level. It doesn't hold real data — it's just a sentinel (guard) that helps traversal start cleanly from the top.

Maximum Level (MAX_LEVEL) is the cap on how many "express lanes" the skip list can have. A typical choice is log₂(max_elements) — for 1024 elements, that's 10 levels.

Persistence Strategy (Copy-on-Write) means: when you modify a node, you don't change it in place. Instead, you create a copy of the affected path and link it to a new version root. Old versions remain untouched.
🗂Main & Sub Operations
  • Create Persistent Skip List
  • Initialize Header Node
  • Set Maximum Level
  • Define Persistence Strategy
  • Allocate Header Node
  • Initialize Forward Pointers (all NULL)
  • Set Current Level = 0
  • Initialize Version Metadata (version_id=0)
  • Configure Persistence Mode (Copy-on-Write)
// skip list initialization — header node with MAX_LEVEL forward pointers
C — initialization 
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define MAX_LEVEL 6   /* Maximum number of levels in skip list */
#define PROBABILITY 0.5 /* Probability for level generation */

/* A single node in the persistent skip list */
typedef struct Node {
    int key;
    int value;
    int level;                       /* how many forward pointers this node has */
    struct Node *forward[MAX_LEVEL]; /* array of forward pointers per level */
} Node;

/* Represents one snapshot (version) of the skip list */
typedef struct Version {
    int version_id;    /* unique identifier for this version */
    Node *header;      /* entry point (header) for this version */
    int current_level; /* highest active level in this version */
    int node_count;    /* number of elements in this version */
} Version;

/* The persistent skip list container — holds all versions */
typedef struct PersistentSkipList {
    Version *versions[1000]; /* array of all versions */
    int total_versions;       /* how many versions exist */
} PersistentSkipList;

/* Allocate and initialize a new node */
Node* createNode(int key, int value, int level) {
    Node *n = (Node*) malloc(sizeof(Node));
    n->key   = key;
    n->value = value;
    n->level = level;
    for (int i = 0; i < MAX_LEVEL; i++)
        n->forward[i] = NULL; /* all forward pointers start as NULL */
    return n;
}

/* Create and return a new Persistent Skip List (version 0) */
PersistentSkipList* initPersistentSkipList() {
    srand(time(NULL)); /* seed random number generator */

    PersistentSkipList *psl = (PersistentSkipList*)
        malloc(sizeof(PersistentSkipList));
    psl->total_versions = 0;

    /* Create version 0 */
    Version *v0 = (Version*) malloc(sizeof(Version));
    v0->version_id    = 0;
    v0->current_level = 0;
    v0->node_count    = 0;
    v0->header = createNode(-1, -1, MAX_LEVEL); /* header sentinel */

    psl->versions[0] = v0;
    psl->total_versions = 1;

    printf("Persistent Skip List initialized. Version 0 created.\n");
    return psl;
}

int main() {
    PersistentSkipList *psl = initPersistentSkipList();
    printf("Total versions: %d\n", psl->total_versions);
    return 0;
}
🏗️
Real-Life Analogy — Building Foundation
Initialization is like laying the foundation of a multi-story building. The header node is the lobby — every visitor (search, insert, delete) starts here. MAX_LEVEL is the maximum number of floors you're allowed to build. Version metadata is like the building permit — it records which version of the blueprint is currently active.
02
Version Management Operations
// create new version · access previous version · maintain version history
📖Theory Explanation
Version management is the heart of persistence. Every time you modify a skip list, instead of overwriting data, you create a new version.

Each version has its own header/root pointer, which leads to a shared-node structure. Nodes that weren't changed are shared between versions — no copying needed!

Copy Modified Paths: Only the nodes along the modification path get new copies. Unaffected branches of the skip list remain shared, saving memory dramatically.

Version Timeline: A linear chain of versions (v0 → v1 → v2 → ...) where each version points back to its parent, enabling rollback and historical queries.
🗂Main & Sub Operations

  • Access Previous Version
  • Maintain Version History
  • Copy Modified Paths (only changed nodes)
  • Assign Version ID (auto-increment)
  • Store Version References in array
  • Link Version Roots (new header)
  • Track Version Timeline (parent pointer)
// version timeline — shared nodes between versions
C — version management 
/* Copy a node (for copy-on-write during modifications) */
Node* copyNode(Node *original) {
    Node *copy = (Node*) malloc(sizeof(Node));
    *copy = *original; /* shallow copy: copies key, value, all forward pointers */
    return copy;
}

/* Create a new version by copying the latest version's header */
Version* createNewVersion(PersistentSkipList *psl) {
    int latest = psl->total_versions - 1;
    Version *prev = psl->versions[latest];

    Version *newV = (Version*) malloc(sizeof(Version));
    newV->version_id    = prev->version_id + 1; /* assign next ID */
    newV->current_level = prev->current_level;
    newV->node_count    = prev->node_count;

    /* Copy header to allow independent modifications */
    newV->header = copyNode(prev->header);

    psl->versions[psl->total_versions] = newV;
    psl->total_versions++;

    printf("New version %d created (parent: version %d)\n",
           newV->version_id, prev->version_id);
    return newV;
}

/* Access any past version by its ID */
Version* accessVersion(PersistentSkipList *psl, int version_id) {
    if (version_id < 0 || version_id >= psl->total_versions) {
        printf("Error: Version %d does not exist!\n", version_id);
        return NULL;
    }
    printf("Accessing version %d (nodes: %d)\n",
           version_id, psl->versions[version_id]->node_count);
    return psl->versions[version_id];
}

/* Print all versions and their metadata */
void printVersionHistory(PersistentSkipList *psl) {
    printf("\n=== Version History ===\n");
    for (int i = 0; i < psl->total_versions; i++) {
        Version *v = psl->versions[i];
        printf("  Version %d | Nodes: %d | Level: %d\n",
               v->version_id, v->node_count, v->current_level);
    }
}
📷
Real-Life Analogy — Git Version Control
Version management is exactly like Git commits. Each commit (version) is a snapshot of your codebase. Files that didn't change are shared references — Git doesn't re-copy every file on every commit. Only changed files get new copies. You can always checkout any past commit (version) to see what the code looked like then.
03
Insertion Operations
// insert element · insert key-value pair · copy-on-write path
📖Theory Explanation
Inserting into a persistent skip list follows these steps:

1. Search Insert Position: Start from the highest level of the current version's header and move forward while the next node's key is smaller.

2. Generate Random Level: Use a coin-flip (probability 0.5) to decide how many levels the new node should span. Higher levels = faster future lookups.

3. Copy Affected Nodes: Since we can't modify old nodes (immutability!), every node on the update path gets a new copy in the new version.

4. Update Forward Pointers: Link the new node into the copied path. Old version pointers remain unchanged, preserving past versions.

  • Copy Affected Nodes (CoW)
  • Generate Random Level
  • Create New Node

  • Preserve Old Version Links
  • Adjust Current Level if needed

// insertion animation — click step to advance
C — insertion with copy-on-write 
/* Generate a random level for a new node */
int randomLevel() {
    int level = 1;
    /* flip a coin: while heads, go up one more level */
    while ((double)rand() / RAND_MAX < PROBABILITY && level < MAX_LEVEL)
        level++;
    return level;
}

/* Insert key-value into a new version of the persistent skip list */
void insert(PersistentSkipList *psl, int key, int value) {
    /* Step 1: Create a new version */
    Version *newV = createNewVersion(psl);
    Node *header  = newV->header;

    /* update[i] will hold the node whose forward[i] needs updating */
    Node *update[MAX_LEVEL];
    Node *current = header;

    /* Step 2: Traverse top-down to find insert position */
    for (int i = newV->current_level - 1; i >= 0; i--) {
        /* move forward while next key is smaller than target key */
        while (current->forward[i] != NULL &&
               current->forward[i]->key < key)
            current = current->forward[i];
        update[i] = current; /* remember the predecessor at this level */
    }

    /* Step 3: Generate random level for new node */
    int newLevel = randomLevel();

    /* If new level exceeds current max, extend update[] with header */
    if (newLevel > newV->current_level) {
        for (int i = newV->current_level; i < newLevel; i++)
            update[i] = header;
        newV->current_level = newLevel;
    }

    /* Step 4: Create the new node */
    Node *newNode = createNode(key, value, newLevel);

    /* Step 5: Copy affected nodes and update forward pointers */
    for (int i = 0; i < newLevel; i++) {
        /* Copy the predecessor node if not header (CoW) */
        Node *copiedPred = copyNode(update[i]);

        /* Wire new node into the list */
        newNode->forward[i]    = copiedPred->forward[i];
        copiedPred->forward[i] = newNode;

        /* If predecessor is the header, update the version's header */
        if (update[i] == header)
            newV->header = copiedPred;
    }

    newV->node_count++;
    printf("Inserted key=%d, value=%d at level=%d in version %d\n",
           key, value, newLevel, newV->version_id);
}
📮
Real-Life Analogy — Filing Cabinet with Carbon Copies
Imagine a filing cabinet where you must never change existing documents. To add a new file, you photocopy only the folders along the path from the front to the insertion spot. The new folder is inserted in your copies; the old folders on other paths are simply referenced as-is. The original cabinet drawer (old version) is preserved untouched.
04
Deletion Operations
// delete element · copy affected path · preserve previous structure
📖Theory Explanation
Deletion in a persistent skip list is similar to insertion — we never destroy any existing nodes. Instead:

1. Search Target Node: Traverse top-down to find the node with the target key and collect predecessor nodes.

2. Copy Affected Paths: Every predecessor node is copied (CoW). In the new version, the copied predecessors skip over the deleted node.

3. Logical Deletion: The old version still "sees" the deleted node through its original pointers. Only the new version's copied predecessors bypass it.

4. Reduce Current Level: If the deleted node was the only node at the highest level, the new version's level is decremented.
🗂Main & Sub Operations
  • Delete Element
  • Delete Key
  • Search Target Node
  • Traverse Levels (top-down)
  • Copy Affected Paths (CoW)
  • Remove Forward Links in new copies
  • Preserve Previous Structure (old version intact)
  • Delete Logical Reference
  • Reduce Current Level if needed
  • Decrease Node Count
// deletion visualization — old version preserved
C — persistent deletion 
/* Delete a key from the skip list — creates a new version */
int deleteKey(PersistentSkipList *psl, int key) {
    Version *newV  = createNewVersion(psl);
    Node *header   = newV->header;
    Node *update[MAX_LEVEL];
    Node *current  = header;

    /* Step 1: Find predecessors at each level */
    for (int i = newV->current_level - 1; i >= 0; i--) {
        while (current->forward[i] != NULL &&
               current->forward[i]->key < key)
            current = current->forward[i];
        update[i] = current;
    }

    /* Step 2: Check if the key actually exists at level 0 */
    Node *target = (update[0]->forward[0] != NULL &&
                    update[0]->forward[0]->key == key)
                   ? update[0]->forward[0] : NULL;

    if (target == NULL) {
        printf("Key %d not found!\n", key);
        return -1; /* key doesn't exist */
    }

    /* Step 3: Copy affected predecessors and bypass the target node */
    for (int i = 0; i < newV->current_level; i++) {
        if (update[i]->forward[i] != target)
            break; /* this level doesn't touch target */

        Node *copiedPred = copyNode(update[i]);
        /* skip over target: point to what target pointed to */
        copiedPred->forward[i] = target->forward[i];

        if (update[i] == header)
            newV->header = copiedPred;
    }

    /* Step 4: Remove empty top levels in new version */
    while (newV->current_level > 1 &&
           newV->header->forward[newV->current_level - 1] == NULL)
        newV->current_level--;

    newV->node_count--;
    /* NOTE: we do NOT free(target) — old version still references it! */
    printf("Deleted key %d in new version %d. Old version preserved.\n",
           key, newV->version_id);
    return 0;
}
🗑️
Real-Life Analogy — Recycle Bin
Persistent deletion is like moving a file to the Recycle Bin instead of permanently erasing it. In the new folder view (new version), the file is gone. But if you open the Recycle Bin (old version), it's still there. The file isn't physically destroyed — it's just not visible in the new directory listing.

Search Operations
// search key · search by version · search value
📖Theory Explanation
Searching is the most frequent operation and the reason skip lists are fast.

Algorithm:
— Start at the header of the chosen version
— Begin at the highest level (fast lane)
— Move forward as long as the next node's key < target
— Drop one level when you can't move forward
— Repeat until level 0, then check if key matches

Search by Version: You pick which version's header to start from. This lets you search the past without any time travel overhead — it's just a pointer dereference!

Average complexity: O(log n) because each level halves the search space.

  • Start from Highest Level
  • Move Forward in Level (while next.key < target)
  • Drop to Lower Level
  • Compare Keys
  • Detect Key Found (level 0 check)
  • Return Node / Data

C — search by key and version 
/* Search for a key in a specific version of the skip list */
int searchVersion(PersistentSkipList *psl, int version_id, int key) {
    Version *v = accessVersion(psl, version_id);
    if (v == NULL) return -1; /* invalid version */

    Node *current = v->header;

    /* Step 1: Start at the highest level of this version */
    for (int i = v->current_level - 1; i >= 0; i--) {
        /* Step 2: Move forward while next key is less than target */
        while (current->forward[i] != NULL &&
               current->forward[i]->key < key) {
            current = current->forward[i];
            printf("  Level %d: moved to key %d\n", i, current->key);
        }
        /* Step 3: Drop down one level */
        printf("  Dropping to level %d\n", i - 1);
    }

    /* Step 4: Check if we found the key at level 0 */
    Node *candidate = current->forward[0];
    if (candidate != NULL && candidate->key == key) {
        printf("Found key=%d, value=%d in version %d\n",
               key, candidate->value, version_id);
        return candidate->value; /* found! */
    }

    printf("Key %d NOT found in version %d\n", key, version_id);
    return -1; /* not found */
}

/* Search in the latest version */
int search(PersistentSkipList *psl, int key) {
    return searchVersion(psl, psl->total_versions - 1, key);
}
🔍
Real-Life Analogy — Express Train + Local Train
Searching a skip list is like taking an express train. You start on the fastest express line (top level), zoom past most stations. When you overshoot, you switch to a slower local train (lower level) for finer navigation. You alternate between express and local trains until you reach your destination station (key). Much faster than walking the entire track (O(n))!

Historical Access Operations
// access old version · compare versions · retrieve historical data
📖Theory Explanation
Historical access is what makes a skip list persistent rather than ephemeral.

Access Old Version: Simply retrieve the version's header pointer from the versions array and traverse the structure from there. The old structure is frozen — it hasn't changed.

Compare Versions: Walk two version structures in parallel (like merging two sorted lists) and detect:
— Nodes only in version A (added in B)
— Nodes only in version B (deleted in B)
— Nodes in both but with different values (updated)

Detect Modified Nodes: Use pointer equality — if two versions share the same node pointer, the node is unmodified. Different pointer = a copied/changed node.
🗂Main & Sub Operations
  • Access Old Version
  • Compare Versions
  • Retrieve Version Root (O(1) lookup)
  • Traverse Historical Structure
  • Compare Node States
  • Detect Modified Nodes (pointer comparison)
  • Access Historical Data
C — historical access and version comparison 
/* Print all elements visible in a given version */
void printVersion(PersistentSkipList *psl, int version_id) {
    Version *v = accessVersion(psl, version_id);
    if (!v) return;

    printf("\n[Version %d] Elements at level 0:\n", version_id);
    Node *current = v->header->forward[0]; /* start after sentinel */
    while (current != NULL) {
        printf("  key=%-4d  value=%d\n", current->key, current->value);
        current = current->forward[0];
    }
}

/* Compare two versions and print differences */
void compareVersions(PersistentSkipList *psl, int vA, int vB) {
    Version *A = accessVersion(psl, vA);
    Version *B = accessVersion(psl, vB);
    if (!A || !B) return;

    Node *a = A->header->forward[0];
    Node *b = B->header->forward[0];

    printf("\n=== Diff: Version %d vs Version %d ===\n", vA, vB);

    while (a != NULL || b != NULL) {
        if (a == b) {           /* same pointer = shared, unmodified */
            printf("  = key=%d (shared, unchanged)\n", a->key);
            a = a->forward[0];
            b = b->forward[0];
        } else if (b == NULL || (a != NULL && a->key < b->key)) {
            printf("  - key=%d (removed in v%d)\n", a->key, vB);
            a = a->forward[0];
        } else if (a == NULL || b->key < a->key) {
            printf("  + key=%d value=%d (added in v%d)\n",
                   b->key, b->value, vB);
            b = b->forward[0];
        } else {               /* same key, different pointer = updated */
            printf("  ~ key=%d: value %d → %d (modified)\n",
                   a->key, a->value, b->value);
            a = a->forward[0]; b = b->forward[0];
        }
    }
}

You can visit what a website looked like in 2005 — the old pages are still there, frozen in time. The website's current version changed, but the archive preserved every historical state. Comparing versions is like using a diff tool on two archived snapshots.

Update Operations
// update existing value · replace data · preserve old data
📖Theory Explanation
Updating a value in a persistent skip list requires a full Copy-on-Write path copy, even though only the value changes (not the key or structure).

Why copy the path? Because the node containing the key is immutable — you can't change its value in-place (that would corrupt the old version). So you:

1. Search for the target node
2. Copy every node on the path from header to target
3. In the new copy of the target node, update the value
4. Rebuild forward pointer chain through the copies
5. New version sees updated value; old version sees original

This is identical to a delete + insert but more memory-efficient since the structure/level remains the same.
🗂Main & Sub Operations
  • Update Existing Value
  • Replace Data
  • Search Target Node (by key)
  • Copy Modified Nodes along path
  • Modify Stored Value in the copy
  • Create New Version
  • Preserve Old Data (old version unchanged)
C — persistent value update 
/* Update the value of an existing key (creates a new version) */
int updateValue(PersistentSkipList *psl, int key, int newValue) {
    Version *newV  = createNewVersion(psl);
    Node *header   = newV->header;
    Node *update[MAX_LEVEL];
    Node *current  = header;

    /* Step 1: Find predecessors at all levels (same as search) */
    for (int i = newV->current_level - 1; i >= 0; i--) {
        while (current->forward[i] != NULL &&
               current->forward[i]->key < key)
            current = current->forward[i];
        update[i] = current;
    }

    /* Step 2: Confirm key exists */
    Node *target = update[0]->forward[0];
    if (!target || target->key != key) {
        printf("Key %d not found for update!\n", key);
        return -1;
    }

    /* Step 3: Create a copy of the target node with updated value */
    Node *updatedNode = copyNode(target);
    updatedNode->value = newValue; /* only the value changes */

    /* Step 4: Copy predecessors and rewire to updated node */
    for (int i = 0; i < target->level; i++) {
        if (update[i]->forward[i] != target) break;
        Node *cp = copyNode(update[i]);
        cp->forward[i] = updatedNode;  /* point to updated node, not old */
        if (update[i] == header) newV->header = cp;
    }

    /* Old version still has target with old value — untouched */
    printf("Updated key=%d: old value=%d, new value=%d (version %d)\n",
           key, target->value, newValue, newV->version_id);
    return 0;
}
✏️
Real-Life Analogy — Google Docs Revision History
When you edit a Google Doc, the old text doesn't disappear — it's saved in revision history. You can scroll back to see "what this paragraph said 3 days ago." Updating a value in a persistent skip list works the same way: the new version shows the new value, but the old version's node is permanently preserved in history.

Traversal Operations
// traverse entire list · traverse specific version · traverse specific level
📖Theory Explanation
Traversal visits nodes sequentially using forward pointers.

Level 0 Traversal: The bottom level (level 0) contains all nodes in sorted order. Traversing it gives you all elements of a version like a regular linked list.

Specific Level Traversal: Traversing level k only visits nodes that were promoted to that level — useful for debugging or visualizing the skip list's structure.

Version-wise Traversal: By picking a specific version's header, you can traverse any past state of the skip list. Different versions may share many of the same node pointers (unmodified nodes), but they might differ in which nodes their forward pointers chain through.

  • Print Keys
  • Print Values

C — traversal operations 
/* Traverse all nodes at level 0 of a version (full sorted list) */
void traverseVersion(PersistentSkipList *psl, int version_id) {
    Version *v = accessVersion(psl, version_id);
    if (!v) return;

    printf("Version %d → [", version_id);
    Node *curr = v->header->forward[0];
    while (curr) {
        printf("%d", curr->key);
        if (curr->forward[0]) printf(" → ");
        curr = curr->forward[0];
    }
    printf("] NULL\n");
}

/* Traverse a specific level of a version */
void traverseLevel(PersistentSkipList *psl, int version_id, int level) {
    Version *v = accessVersion(psl, version_id);
    if (!v || level >= v->current_level) {
        printf("Level %d not active in version %d\n", level, version_id);
        return;
    }

    printf("Version %d Level %d → [", version_id, level);
    Node *curr = v->header->forward[level];
    while (curr) {
        printf("%d", curr->key);
        if (curr->forward[level]) printf(" → ");
        curr = curr->forward[level];
    }
    printf("] NULL\n");
}

/* Display all levels of all versions (full structure dump) */
void displayAll(PersistentSkipList *psl) {
    for (int v = 0; v < psl->total_versions; v++) {
        printf("\n--- Version %d (level=%d, nodes=%d) ---\n",
               v, psl->versions[v]->current_level,
               psl->versions[v]->node_count);
        for (int lv = psl->versions[v]->current_level - 1; lv >= 0; lv--)
            traverseLevel(psl, v, lv);
    }
}

Level Management Operations
// manage levels · adjust maximum level · maintain balance
📖Theory Explanation
Levels in a skip list determine its performance characteristics. More levels = potentially faster search (fewer nodes per level), but more memory.

Increase Current Level: Happens automatically during insertion when a new node's random level exceeds the current max. New header forward pointers at the new levels are set.

Decrease Current Level: Happens during deletion when the highest level's only node is removed. The level counter is decremented until a non-empty level is found.

Remove Empty Levels: An empty level wastes traversal time (you start high and immediately drop). Cleaning them keeps search efficient.

Balanced Levels: With probability 0.5 per level, the expected number of nodes at level k is n/2ᵏ — naturally balanced!
🗂Main & Sub Operations
  • Manage Levels
  • Adjust Maximum Level
  • Increase Current Level (during insert)
  • Decrease Current Level (during delete)
  • Remove Empty Levels
  • Maintain Balanced Levels (probabilistic)

Persistence Maintenance Operations
// copy-on-write · share unmodified nodes · immutable links · version dependencies
📖Theory Explanation
Persistence maintenance ensures the correctness and efficiency of the persistent structure over time.

Copy-on-Write (CoW): The fundamental rule — never modify a node directly. Always create a copy first. CoW ensures old versions are never corrupted.

Share Unmodified Nodes: Nodes that don't lie on the modification path are shared between old and new versions via pointer sharing. This keeps memory usage at O(log n) per operation instead of O(n).

Maintain Immutable Links: Once a node's forward pointer is set, it never changes. This is the invariant that makes historical access safe.

Track Version Dependencies: Some advanced systems track which versions reference which nodes, enabling garbage collection of unreachable old versions.
🗂Main & Sub Operations
  • Maintain Persistence
  • Preserve Structural Integrity
  • Copy-on-Write Handling
  • Share Unmodified Nodes
  • Maintain Immutable Links
  • Track Version Dependencies
// copy-on-write — shared vs copied nodes between v1 and v2
C — copy-on-write and node sharing 
/*
 * Copy-on-Write principle:
 * Before modifying a node, always copy it first.
 * Old pointer stays unchanged → old version intact.
 * New pointer points to copy → new version has modification.
 */

/* Count total shared nodes between two versions */
int countSharedNodes(PersistentSkipList *psl, int vA, int vB) {
    Version *A = psl->versions[vA];
    Version *B = psl->versions[vB];
    Node *a   = A->header->forward[0];
    Node *b   = B->header->forward[0];
    int shared = 0;

    while (a && b) {
        if (a == b) { /* identical pointer = same node shared */
            shared++;
            a = a->forward[0];
            b = b->forward[0];
        } else if (a->key < b->key) a = a->forward[0];
        else                        b = b->forward[0];
    }
    printf("Shared nodes between v%d and v%d: %d\n", vA, vB, shared);
    return shared;
}

/* Verify immutability: check no node in old version was modified
   (key invariant of persistence) */
int verifyImmutability(PersistentSkipList *psl, int version_id) {
    /* In a correct persistent implementation, all nodes reachable
       from old version headers are guaranteed unmodified.
       Here we just report the version's node count for validation. */
    Version *v = psl->versions[version_id];
    int count  = 0;
    Node *curr  = v->header->forward[0];
    while (curr) { count++; curr = curr->forward[0]; }
    printf("Version %d: recorded=%d, actual=%d → %s\n",
           version_id, v->node_count, count,
           (count == v->node_count) ? "INTACT" : "CORRUPTED!");
    return (count == v->node_count);
}

Only the mutated segments are different. The original DNA (old version) remains in the parent cell, completely unchanged — only the new cell has the modification.

Pointer Management Operations
// maintain forward pointers · update links · redirect references
📖Theory Explanation
Forward pointers are the veins of a skip list — they connect nodes and enable fast traversal across levels.

Update Pointer Links: After copying a node (CoW), the new copy's forward pointers must be carefully set to point to either new nodes (for the modification path) or old shared nodes (for unaffected branches).

Preserve Old References: The original (pre-copy) node's pointers must never change. This is the critical rule — if even one old pointer is modified, the old version becomes corrupted.

Initialize Pointer Arrays: Every node has MAX_LEVEL forward pointer slots. All unused slots (above the node's level) must be NULL to prevent traversal errors.

  • Update Pointer Links (in copied nodes only)
  • Redirect References (after insert/delete)
  • Preserve Old References (original never changed)
  • Initialize Pointer Arrays (all NULL at creation)
C — pointer management 
/*
 * POINTER MANAGEMENT RULES:
 * 1. Never modify forward[i] of an existing (old) node.
 * 2. Always copy a node before changing its forward pointers.
 * 3. Initialize all forward[i] to NULL when creating a node.
 * 4. Only update forward[i] for levels 0 ... node->level-1
 */

/* Safely redirect a forward pointer in a NEW (copied) node only */
void redirectForward(Node *copiedNode, int level, Node *newTarget) {
    if (level < 0 || level >= MAX_LEVEL) {
        printf("Error: invalid level %d\n", level);
        return;
    }
    printf("  Redirecting forward[%d] of node(key=%d) → key=%d\n",
           level, copiedNode->key,
           newTarget ? newTarget->key : -1);
    copiedNode->forward[level] = newTarget;
}

/* Validate that all pointer arrays in a version are consistent */
void validatePointers(Version *v) {
    Node *curr = v->header;
    int  errors = 0;

    while (curr) {
        /* Slots above the node's level must be NULL */
        for (int i = curr->level; i < MAX_LEVEL; i++) {
            if (curr->forward[i] != NULL) {
                printf("WARN: node key=%d has non-NULL forward[%d]!\n",
                       curr->key, i);
                errors++;
            }
        }
        curr = curr->forward[0];
    }
    printf("Pointer validation: %d errors found.\n", errors);
}

/* Print all forward pointers of a node (for debugging) */
void printNodePointers(Node *n) {
    printf("Node(key=%d, level=%d) forward pointers:\n", n->key, n->level);
    for (int i = 0; i < n->level; i++)
        printf("  forward[%d] → key=%d\n", i,
               n->forward[i] ? n->forward[i]->key : -1);
}

Memory Management Operations
// allocate · deallocate · garbage collection · prevent memory leak
📖Theory Explanation
Memory management in a persistent skip list is trickier than a regular skip list because nodes are shared between versions.

Allocate Memory: Each new node and version requires a malloc call. Nodes are created during insertion and copy-on-write.

Deallocate Memory: You can ONLY free a node when no version references it anymore. Freeing a shared node would corrupt all versions that still point to it.

Garbage Collection: To safely free old nodes, use reference counting — each node has a counter of how many versions reference it. When count reaches 0, the node can be freed.

Free Unused Versions: If a version is no longer needed, free the version metadata. But first ensure no other version needs the nodes it uniquely owns.
🗂Main & Sub Operations
  • Allocate Memory (malloc per node)
  • Deallocate Memory (only when safe)
  • Create Node (malloc + initialize)
  • Free Unused Versions (metadata only)
  • Garbage Collection (ref-counting)
  • Prevent Memory Leak (track all allocs)
C — memory management with reference counting 
/* Extended node with reference count for safe deallocation */
typedef struct NodeGC {
    int key;
    int value;
    int level;
    int ref_count;              /* how many versions reference this node */
    struct NodeGC *forward[MAX_LEVEL];
} NodeGC;

/* Increment reference count when a version starts using a node */
void retainNode(NodeGC *n) {
    if (n) { n->ref_count++; printf("  retain key=%d (ref=%d)\n", n->key, n->ref_count); }
}

/* Decrement reference count; free node when count reaches 0 */
void releaseNode(NodeGC *n) {
    if (!n) return;
    n->ref_count--;
    printf("  release key=%d (ref=%d)", n->key, n->ref_count);
    if (n->ref_count == 0) {
        printf(" → FREED\n");
        free(n); /* safe to free — nobody references this node */
    } else {
        printf(" → still in use\n");
    }
}

/* Free an entire version (only its metadata, not shared nodes!) */
void freeVersion(Version *v) {
    /* In a GC system, release each node referenced by this version */
    /* For simplicity, here we only free the version metadata */
    free(v->header); /* header was uniquely allocated per version */
    free(v);
    printf("Version metadata freed.\n");
}

/* Free all versions and the persistent skip list container */
void freePSL(PersistentSkipList *psl) {

       Here we only free version metadata as a simplified example. */

        free(psl->versions[i]);
    }

}

Analytical Operations
// analyze structure · version efficiency · shared nodes · memory usage
📖Theory Explanation
Analytical operations help you evaluate and optimize a persistent skip list.

Count Total Versions: The total_versions field gives this in O(1).

Measure Search Complexity: Run searches and count comparisons. In a well-balanced skip list, this should be approximately 2 × log₂(n).

Analyze Shared Nodes: Compare node pointers across versions. High sharing = good memory efficiency. Low sharing = too much copying.

Evaluate Memory Usage: Count total live nodes across all versions. With CoW, total memory ≈ n + v × log(n) where v = number of versions and n = elements. Without persistence, it would be v × n.

C — analytical operations 
/* Count unique nodes reachable from all versions (total live nodes) */
int countTotalUniqueNodes(PersistentSkipList *psl) {
    /* Use a simple pointer set via sorted array for uniqueness check */
    Node *seen[10000]; int seenCount = 0;

    for (int v = 0; v < psl->total_versions; v++) {
        Node *curr = psl->versions[v]->header->forward[0];
        while (curr) {
            int found = 0;
            for (int s = 0; s < seenCount; s++)
                if (seen[s] == curr) { found = 1; break; }
            if (!found) seen[seenCount++] = curr;
            curr = curr->forward[0];
        }
    }
    printf("Total unique nodes across all versions: %d\n", seenCount);
    return seenCount;
}

/* Measure search complexity by counting comparisons */
int searchWithCount(Version *v, int key) {
    int comparisons = 0;
    Node *curr = v->header;
    for (int i = v->current_level - 1; i >= 0; i--) {
        while (curr->forward[i] && curr->forward[i]->key < key) {
            comparisons++;
            curr = curr->forward[i];
        }
        comparisons++;
    }
    printf("Search for %d: %d comparisons (log2(%d) ≈ %.1f)\n",
           key, comparisons, v->node_count,
           v->node_count > 0 ? log2(v->node_count) : 0);
    return comparisons;
}

/* Print full analytics report */
void analyticsReport(PersistentSkipList *psl) {
    int totalV    = psl->total_versions;
    int totalNodes = countTotalUniqueNodes(psl);
    int latest     = psl->versions[totalV-1]->node_count;

    printf("\n=== Analytics Report ===\n");
    printf("  Total versions:         %d\n", totalV);
    printf("  Latest version nodes:   %d\n", latest);
    printf("  Total unique live nodes: %d\n", totalNodes);
    printf("  Memory savings vs naive: %.1f%%\n",
           totalV * latest > 0
           ? (1.0 - (double)totalNodes / (totalV * latest)) * 100
           : 0.0);
}

Advanced Structural Operations
// merge versions · split skip list · snapshot · rollback
📖Theory Explanation
These advanced operations combine multiple persistence features.

Merge Versions: Combine the elements of two different versions into a new version. Useful for merging branches (like git merge). Requires a merge-sort-like traversal of both structures.

Split Skip List: Create two new versions — one with keys < pivot, one with keys ≥ pivot. Nodes can be shared where ranges don't overlap.

Snapshot Creation: A lightweight operation — simply note the current version ID as a named snapshot. No data copying needed since all versions are already persistent!

Rollback Operation: To "undo" to a past state, simply redirect the "current version" pointer to an earlier version. O(1) operation!

/* Snapshot: record a named checkpoint (just a version ID) */
typedef struct { char name[64]; int version_id; } Snapshot;
Snapshot snapshots[100]; int snapCount = 0;

void takeSnapshot(PersistentSkipList *psl, const char *name) {
    int vid = psl->total_versions - 1;
    snprintf(snapshots[snapCount].name, 64, "%s", name);
    snapshots[snapCount].version_id = vid;
    snapCount++;
    printf("Snapshot '%s' saved at version %d\n", name, vid);
}

/* Rollback: set the "current" pointer to an earlier version */
void rollback(PersistentSkipList *psl, int target_version) {
    if (target_version < 0 || target_version >= psl->total_versions) {
        printf("Invalid rollback target!\n"); return;
    }
    /* In a real system, you'd set a "current_version" pointer.
       Here, we create a new version that copies the target. */
    Version *target = psl->versions[target_version];
    Version *restored = (Version*)malloc(sizeof(Version));
    restored->version_id    = psl->total_versions;
    restored->current_level = target->current_level;
    restored->node_count    = target->node_count;
    restored->header = copyNode(target->header); /* share nodes with target */
    psl->versions[psl->total_versions++] = restored;
    printf("Rolled back: new version %d mirrors version %d\n",
           restored->version_id, target_version);
}

/* Merge two versions (simplified: re-insert all keys from both) */
void mergeVersions(PersistentSkipList *psl, int vA, int vB) {
    Version *A = accessVersion(psl, vA);
    Version *B = accessVersion(psl, vB);
    if (!A || !B) return;

    printf("Merging version %d and version %d:\n", vA, vB);
    Node *a = A->header->forward[0];
    Node *b = B->header->forward[0];

    /* Merge-sort style: insert smaller key first into new version */
    while (a || b) {
        if (!b || (a && a->key <= b->key)) {
            insert(psl, a->key, a->value); a = a->forward[0];
        } else {
            insert(psl, b->key, b->value); b = b->forward[0];
        }
    }
    printf("Merge complete → new version %d\n", psl->total_versions-1);
}
