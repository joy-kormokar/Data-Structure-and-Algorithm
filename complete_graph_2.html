Complete Graph & Adjacency List
A complete graph (Kn) is a graph where every vertex is connected to every other vertex. An adjacency list stores each vertex alongside a list of its neighbours — the most memory-efficient way to represent sparse-to-dense graphs.
🌐
Interactive Complete Graph Visualizer
Vertices:
K₄ has 4 vertices and 6 edges. Formula: n(n-1)/2 = 4×3/2 = 6

Adjacency List Structure in C: An array of linked list heads. Index i stores a pointer to the head of the list of all vertices adjacent to vertex i.

// Adjacency List Node
struct Node {
    int data;          // neighbor vertex id
    struct Node *next; // pointer to next neighbor
};

// Graph structure
struct Graph {
    int numVertices;            // total vertices
    struct Node **adjList;   // array of linked list heads
};

// Create a graph with n vertices
struct Graph* createGraph(int n) {
    struct Graph* g = malloc(sizeof(struct Graph));
    g->numVertices = n;
    g->adjList = malloc(n * sizeof(struct Node*));
    for (int i = 0; i < n; i++)
        g->adjList[i] = NULL; // each list starts empty
    return g;
}

Insert Vertex
Adding a new vertex means expanding the adjacency list array by one. In a complete graph, we also connect it to all existing vertices automatically.

void insertVertex(struct Graph* g) {
    int n = g->numVertices;

    /* Step 1: Grow the adjList array by 1 */
    g->adjList = realloc(g->adjList,
                  (n + 1) * sizeof(struct Node*));
    /* Step 2: New vertex starts with empty list */
    g->adjList[n] = NULL;
    g->numVertices++;

    /* Step 3: Auto-connect to all existing vertices */
    for (int i = 0; i < n; i++) {
        addEdge(g, i, n); // connect old → new
        addEdge(g, n, i); // connect new → old
    }
    printf("Vertex %d inserted.\n", n);
}

Like adding a new person to a fully connected team chat — they instantly get added to every existing conversation thread.

Vertex Operations
Delete Vertex
Deleting a vertex requires removing all its edges (from its own list and from all neighbours' lists), then shifting the array down.
🗑
Delete Vertex Visualizer

Select a vertex and click Delete.
void deleteVertex(struct Graph* g, int v) {
    int n = g->numVertices;

    /* Free the vertex's own adjacency list */
    freeList(g->adjList[v]);

    /* Remove v from all other adjacency lists */
    for (int i = 0; i < n; i++) {
        if (i != v)
            g->adjList[i] = removeFromList(g->adjList[i], v);
    }

    /* Shift array: fill gap left by v */
    for (int i = v; i < n - 1; i++)
        g->adjList[i] = g->adjList[i + 1];

    g->numVertices--;
    printf("Vertex %d deleted.\n", v);
}

Like removing a person from a group chat — their messages vanish and they are removed from every other member's contact list.

Vertex Operations
Update Vertex
Updating a vertex ID means renaming it everywhere — in its own list and in all neighbours' lists. All references must be updated to maintain consistency.
✏️
Update Vertex

In a typical integer-keyed adjacency list, "updating" a vertex means renaming vertex oldID to newID. The adjList array index is the primary ID, so we swap entries and walk all neighbour lists to update references.

void updateVertex(struct Graph* g, int oldID, int newID) {
    int n = g->numVertices;
    if (!searchVertex(g, oldID) || !searchVertex(g, newID)) {
        printf("Invalid vertex IDs.\n");
        return;
    }
    /* Swap the two adjList entries */
    struct Node* temp = g->adjList[oldID];
    g->adjList[oldID] = g->adjList[newID];
    g->adjList[newID] = temp;

    /* Update all neighbour references */
    for (int i = 0; i < n; i++) {
        struct Node* cur = g->adjList[i];
        while (cur) {
            if (cur->data == oldID) cur->data = newID;
            else if (cur->data == newID) cur->data = oldID;
            cur = cur->next;
        }
    }
    printf("Swapped vertices %d and %d.\n", oldID, newID);
}
🔄 Analogy

Like renaming a city on a map — you must update every road sign that mentions the old name.

Edge Operations
Insert Edge
Adding an edge means prepending to the neighbour's linked list. For an undirected graph, we add in both directions.
🔗

/* Create a new linked list node */
struct Node* newNode(int data) {
    struct Node* node = malloc(sizeof(struct Node));
    node->data = data;
    node->next = NULL;
    return node;
}

/* Add undirected edge (u, v) */
void addEdge(struct Graph* g, int u, int v) {
    /* Prepend v to u's list */
    struct Node* node = newNode(v);
    node->next = g->adjList[u];
    g->adjList[u] = node;

    /* Prepend u to v's list (undirected) */
    node = newNode(u);
    node->next = g->adjList[v];
    g->adjList[v] = node;
}
🤝 Analogy

Like two people adding each other as friends — the connection is mutual and appears in both their contact lists.

Delete Edge
Deleting an edge removes the target node from both endpoints' adjacency lists.
✂️
Delete Edge
struct Node* removeNode(struct Node* head, int val) {
    struct Node* cur = head, *prev = NULL;
    while (cur) {
        if (cur->data == val) {
            if (prev) prev->next = cur->next;
            else head = cur->next;
            free(cur);
            return head;
        }
        prev = cur; cur = cur->next;
    }
    return head;
}

void deleteEdge(struct Graph* g, int u, int v) {
    g->adjList[u] = removeNode(g->adjList[u], v);
    g->adjList[v] = removeNode(g->adjList[v], u);
    printf("Edge (%d,%d) deleted.\n", u, v);
}

Like unfriending someone — they disappear from your list and you disappear from theirs.

Search Edge
Walk the adjacency list of vertex u and check if v appears in it. O(degree) time.
🔎
Search Edge
int searchEdge(struct Graph* g, int u, int v) {
    struct Node* cur = g->adjList[u];
    while (cur) {
        if (cur->data == v) {
            printf("Edge (%d,%d) exists.\n", u, v);
            return 1;
        }
        cur = cur->next;
    }
    printf("Edge (%d,%d) NOT found.\n", u, v);
    return 0;
}

In a complete graph Kₙ every edge exists, so this always returns 1 for valid vertices. Still useful for incomplete/partial graphs during construction.

Update Edge
In a weighted graph, updating an edge means changing its weight. We walk both neighbour lists and update the weight field.
⚖️
Weighted Edge Update
/* Weighted Node */
struct WNode {
    int dest;
    int weight;
    struct WNode* next;
};

void updateEdge(struct WNode** adjList, int u, int v,
                int newWeight) {
    /* Update u -> v weight */
    struct WNode* cur = adjList[u];
    while (cur) {
        if (cur->dest == v) { cur->weight = newWeight; break; }
        cur = cur->next;
    }
    /* Update v -> u weight (undirected) */
    cur = adjList[v];
    while (cur) {
        if (cur->dest == u) { cur->weight = newWeight; break; }
        cur = cur->next;
    }
}

int countEdges(struct Graph* g) {
    int count = 0;
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* cur = g->adjList[i];
        while (cur) { count++; cur = cur->next; }
    }
    return count / 2; // each edge counted twice
}

void displayEdges(struct Graph* g) {
    printf("Edges:\n");
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* cur = g->adjList[i];
        while (cur) {
            if (cur->data > i) // print each edge once
                printf("  (%d, %d)\n", i, cur->data);
            cur = cur->next;
        }
    }
}

Traversal Operations
Breadth First Search (BFS)
BFS explores level by level, using a queue. Start from source, visit all neighbours, then their neighbours, and so on.
🌊
BFS Visualizer
Start:
BFS Order: (click Run BFS)

BFS uses a queue (FIFO). Mark source visited → enqueue → dequeue front → visit all unvisited neighbours → enqueue them → repeat.

void BFS(struct Graph* g, int start) {
    int n = g->numVertices;
    int visited[n], queue[n];
    int front = 0, rear = 0;

    /* Step 1: Initialize visited array */
    for (int i = 0; i < n; i++) visited[i] = 0;

    /* Step 2: Mark start visited & enqueue */
    visited[start] = 1;
    queue[rear++] = start;

    while (front < rear) {
        /* Step 3: Dequeue front vertex */
        int v = queue[front++];
        printf("%d ", v);

        /* Step 4: Traverse adjacent nodes */
        struct Node* cur = g->adjList[v];
        while (cur) {
            if (!visited[cur->data]) {
                visited[cur->data] = 1;
                queue[rear++] = cur->data;
            }
            cur = cur->next;
        }
    }
}
🏄 Analogy

Like broadcasting news — first tell your immediate friends (level 1), then they tell their friends (level 2), and so on. BFS = layer-by-layer spread.

DFS dives deep into one branch before backtracking. Implemented recursively or with an explicit stack.

void DFS_recursive(struct Graph* g, int v, int visited[]) {
    /* Mark visited & print */
    visited[v] = 1;
    printf("%d ", v);

    /* Recurse on unvisited neighbours */
    struct Node* cur = g->adjList[v];
    while (cur) {
        if (!visited[cur->data])
            DFS_recursive(g, cur->data, visited);
        cur = cur->next; // backtracking happens here
    }
}
void DFS_stack(struct Graph* g, int start) {
    int n = g->numVertices;
    int visited[n], stack[n], top = -1;
    for (int i = 0; i < n; i++) visited[i] = 0;

    stack[++top] = start; // push start

    while (top >= 0) {
        int v = stack[top--]; // pop
        if (!visited[v]) {
            visited[v] = 1;
            printf("%d ", v);
        }
        struct Node* cur = g->adjList[v];
        while (cur) {
            if (!visited[cur->data])
                stack[++top] = cur->data;
            cur = cur->next;
        }
    }
}
🕳️ Analogy

Like exploring a maze — you go as deep as you can down one path, hit a dead end, backtrack, and try the next branch.

Completeness Operations
Check & Validate Completeness
A graph is complete if every vertex has degree n−1 and total edges = n(n−1)/2.
✅
Completeness Checker
Click a button to check completeness.
Required edges = n(n−1)/2  |  Each vertex must have degree = n−1
/* Count length of a linked list */
int listLen(struct Node* head) {
    int c = 0;
    while (head) { c++; head = head->next; }
    return c;
}

int isCompleteGraph(struct Graph* g) {
    int n = g->numVertices;
    int required = n * (n - 1) / 2;

    /* Validate degree of each vertex */
    for (int i = 0; i < n; i++) {
        if (listLen(g->adjList[i]) != n - 1) {
            printf("Vertex %d missing edges!\n", i);
            return 0; // not complete
        }
    }
    /* Compare actual vs required edge count */
    if (countEdges(g) != required) return 0;

    printf("Graph is complete K_%d\n", n);
    return 1;
}

void autoComplete(struct Graph* g) {
    int n = g->numVertices, added = 0;
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            /* Detect missing edge */
            if (!searchEdge(g, i, j)) {
                addEdge(g, i, j); // insert missing edge
                added++;
            }
        }
    }
    printf("%d edges added. Graph is now K_%d.\n", added, n);
}

Path Operations
Find Path & All Paths
Find whether a path exists between two vertices, and enumerate all possible paths using DFS with backtracking.
🛤️
Path Finding
/* Find if path exists: src → dest */
int findPath(struct Graph* g, int src, int dest,
             int visited[]) {
    if (src == dest) return 1; // reached!
    visited[src] = 1;
    struct Node* cur = g->adjList[src];
    while (cur) {
        if (!visited[cur->data])
            if (findPath(g, cur->data, dest, visited))
                return 1;
        cur = cur->next;
    }
    return 0;
}

/* Print all paths from src to dest */
void allPaths(struct Graph* g, int src, int dest,
              int path[], int pLen, int visited[]) {
    path[pLen++] = src;
    visited[src] = 1;

    if (src == dest) {
        for(int i=0; i<pLen; i++)
            printf("%d%s", path[i], i<pLen-1?"->":"\n");
    } else {
        struct Node* cur = g->adjList[src];
        while (cur) {
            if (!visited[cur->data])
                allPaths(g, cur->data, dest,
                          path, pLen, visited);
            cur = cur->next;
        }
    }
    /* Backtrack */
    visited[src] = 0;
}
🗺️ Analogy

Like planning all routes from City A to City B on a road map — findPath tells you if any route exists; allPaths lists every route.

Shortest Path (BFS-based)
In an unweighted graph, BFS naturally finds the shortest path (fewest edges). Track predecessors to reconstruct the path.
⚡
Shortest Path Visualizer

In Kₙ the shortest path between any two vertices is always 1 (direct edge).
void shortestPath(struct Graph* g, int src, int dest) {
    int n = g->numVertices;
    int visited[n], dist[n], parent[n], queue[n];
    int front = 0, rear = 0;

    for (int i = 0; i < n; i++) {
        visited[i] = 0; dist[i] = -1; parent[i] = -1;
    }

    visited[src] = 1; dist[src] = 0;
    queue[rear++] = src;

    while (front < rear) {
        int v = queue[front++];
        struct Node* cur = g->adjList[v];
        while (cur) {
            if (!visited[cur->data]) {
                visited[cur->data] = 1;
                dist[cur->data] = dist[v] + 1;
                parent[cur->data] = v;
                queue[rear++] = cur->data;
            }
            cur = cur->next;
        }
    }
    printf("Distance: %d\n", dist[dest]);
    /* Trace back using parent array */
}
Path Operations
Hamiltonian Path Check
A Hamiltonian path visits every vertex exactly once. In Kₙ a Hamiltonian path always exists. We verify this with backtracking DFS.
🚩
Hamiltonian Path

In a complete graph Kₙ (n ≥ 1), a Hamiltonian path always exists because every vertex is reachable from every other vertex directly. The check still demonstrates the backtracking technique.

int hamiltonianPath(struct Graph* g, int v,
                      int visited[], int path[], int pos) {
    int n = g->numVertices;
    if (pos == n) { // all vertices visited
        printPath(path, n); return 1;
    }

    struct Node* cur = g->adjList[v];
    while (cur) {
        if (!visited[cur->data]) {
            visited[cur->data] = 1;
            path[pos] = cur->data;
            if (hamiltonianPath(g, cur->data,
                    visited, path, pos+1))
                return 1;
            /* Backtrack */
            visited[cur->data] = 0;
        }
        cur = cur->next;
    }
    return 0;
}
🏁 Analogy

Like a salesperson who visits every city exactly once — no city can be visited twice, no city can be skipped.

/* Detect cycle using DFS (returns 1 if cycle found) */
int detectCycle(struct Graph* g, int v, int parent,
                int visited[]) {
    visited[v] = 1;
    struct Node* cur = g->adjList[v];
    while (cur) {
        if (!visited[cur->data]) {
            if (detectCycle(g, cur->data, v, visited))
                return 1;
        } else if (cur->data != parent) {
            return 1; // back-edge = cycle found!
        }
        cur = cur->next;
    }
    return 0;
}

/* Check Euler Cycle: all vertices must have even degree */
int hasEulerCycle(struct Graph* g) {
    for (int i = 0; i < g->numVertices; i++)
        if (listLen(g->adjList[i]) % 2 != 0) return 0;
    return 1; // all even degree → Euler cycle exists
}
🎡 Analogy

Hamiltonian Cycle: Visit every city exactly once and return home. Euler Cycle: Traverse every road exactly once and return home (like the Königsberg bridge problem).

int inDegree(struct Graph* g, int v) {

/* Clique Detection: a clique is a complete subgraph */
/* In Kn, the entire graph is a clique */
int isClique(struct Graph* g, int verts[], int k) {
    for(int i=0; i<k; i++)
        for(int j=i+1; j<k; j++)
            if(!searchEdge(g, verts[i], verts[j]))
                return 0;
    return 1;
}

/* Symmetry: adjList[u] contains v ↔ adjList[v] contains u */
int isSymmetric(struct Graph* g) {
    for(int i=0; i<g->numVertices; i++) {
        struct Node* cur = g->adjList[i];
        while(cur) {
            if(!searchEdge(g, cur->data, i)) return 0;
            cur = cur->next;
        }
    }
    return 1;
}

/* Greedy Graph Coloring */
void graphColoring(struct Graph* g) {
    int n = g->numVertices;
    int color[n];
    int available[n];

    color[0] = 0; // first vertex gets color 0
    for(int i=1; i<n; i++) color[i] = -1;
    for(int i=0; i<n; i++) available[i] = 0;

    for(int v=1; v<n; v++) {
        /* Mark colors used by neighbors */
        struct Node* cur = g->adjList[v];
        while(cur) {
            if(color[cur->data] != -1)
                available[color[cur->data]] = 1;
            cur = cur->next;
        }
        /* Assign first available color */
        int c;
        for(c=0; c<n; c++) if(!available[c]) break;
        color[v] = c;
        /* Reset available array */
        cur = g->adjList[v];
        while(cur) {
            if(color[cur->data] != -1)
                available[color[cur->data]] = 0;
            cur = cur->next;
        }
    }
    for(int i=0;i<n;i++)
        printf("Vertex %d → Color %d\n", i, color[i]);
}

Traveling Salesman Problem (TSP): Find the minimum-cost Hamiltonian cycle. On Kₙ this is finding the shortest tour through all cities. Brute force: try all (n−1)! permutations.

/* TSP brute force on weighted complete graph */
int minCost = INT_MAX;

void tsp(int weight[][10], int n, int visited[],
         int cur, int count, int cost, int start) {
    if(count == n && weight[cur][start]) {
        /* Completed a tour — check if cheapest */
        if(cost + weight[cur][start] < minCost)
            minCost = cost + weight[cur][start];
        return;
    }
    for(int i=0; i<n; i++) {
        if(!visited[i] && weight[cur][i]) {
            visited[i] = 1;
            tsp(weight, n, visited, i, count+1,
                cost+weight[cur][i], start);
            visited[i] = 0; // backtrack
        }
    }
}
/* Complexity: O((n-1)!) — exponential */

A complete subgraph (clique) of size k means k vertices all mutually connected. Finding the maximum clique is NP-hard in general but trivial in Kₙ (the whole graph).

/* Check if subset of k vertices forms a complete subgraph */
int isCompleteSubgraph(struct Graph* g,
                        int subset[], int k) {
    for(int i=0; i<k; i++)
        for(int j=i+1; j<k; j++)
            if(!searchEdge(g, subset[i], subset[j]))
                return 0; // missing edge → not complete
    return 1;
}

/* isEmpty: true if no vertices */
int isEmpty(struct Graph* g) {
    return g->numVertices == 0;

/* Clear: free all lists, reset numVertices */
void clearGraph(struct Graph* g) {
    for(int i=0; i<g->numVertices; i++)
        freeList(g->adjList[i]);
    free(g->adjList);
    g->adjList = NULL;
    g->numVertices = 0;
}

/* Copy graph: deep copy all adjacency lists */
struct Graph* copyGraph(struct Graph* src) {
    struct Graph* dst = createGraph(src->numVertices);
    for(int i=0; i<src->numVertices; i++) {
        struct Node* cur = src->adjList[i];
        while(cur) {
            appendToList(&dst->adjList[i], cur->data);
            cur = cur->next;
        }
    }
    return dst;
}

/* Reset visited array to all zeros */
void resetVisited(int visited[], int n) {
    for(int i=0; i<n; i++) visited[i] = 0;
}

void displayFullGraph(struct Graph* g) {
    printf("\n=== Graph (K%d) ===\n", g->numVertices);
    displayVertices(g);
    displayGraph(g);
    printf("Edges: %d | Complete: %s\n",
        countEdges(g),
        isCompleteGraph(g) ? "YES" : "NO");
}
