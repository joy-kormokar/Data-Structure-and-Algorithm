Introduction to Disconnected Graphs
SECTION 01 · FOUNDATION
📖 What is a Graph?

A Graph is a data structure made of vertices (nodes) and edges (connections) between them. Think of it as a network.

A Disconnected Graph is one where not all vertices are reachable from every other vertex. It has two or more separate groups (components) with no path between them.

🔴 Connected vs 🟢 Disconnected Graph
A connected graph: every node can reach every other node.
🌍 Real-Life Analogy
Islands in an ocean: Each island is a component. People on the same island can travel to each other. But islands with no bridge or boat route between them are disconnected — even though they all exist on the same planet (graph).
📌 Key Terms
TermMeaning
VertexA node/point in the graph (e.g., city, person)
EdgeA connection between two vertices (e.g., road, friendship)
ComponentA group of vertices all connected to each other
Isolated VertexA vertex with no edges at all
AdjacencyTwo vertices are adjacent if there is an edge between them

Adjacency List Representation
SECTION 02 · REPRESENTATION
📖 Theory

An Adjacency List stores the graph as an array of linked lists. Each index in the array represents a vertex. The linked list at each index stores all the neighbors of that vertex.

Why Adjacency List for Disconnected Graphs? It is memory-efficient — it only stores existing edges, not all possible connections like a matrix would. Perfect when the graph is sparse (few edges).

  • Space complexity: O(V + E)
  • Add edge: O(1)
  • Check edge between u and v: O(degree)
Adjacency List Structure
Hover over nodes to see connections.
C — Adjacency List Structure 
#include <stdio.h>
#include <stdlib.h>

#define MAX 100

/* Each node in the adjacency linked list */
struct Node {
    int vertex;
    struct Node* next;
};

/* Graph structure: array of linked list heads */
struct Graph {
    int numVertices;
    struct Node* adjList[MAX];
};

/* Create a new graph with V vertices */
struct Graph* createGraph(int V) {
    struct Graph* graph = (struct Graph*) malloc(sizeof(struct Graph));
    graph->numVertices = V;
    for (int i = 0; i < V; i++) {
        graph->adjList[i] = NULL; /* No edges yet */
    }
    return graph;
}

/* Create a new node */
struct Node* createNode(int v) {
    struct Node* newNode = (struct Node*) malloc(sizeof(struct Node));
    newNode->vertex = v;
    newNode->next = NULL;
    return newNode;
}
🌍 Real-Life Analogy
School attendance register: Each row = one student (vertex). The row lists all their friends in class (neighbors). If a student has no friends, their row is empty (isolated vertex). Students in different sections who don't interact = different components.

Delete Vertex
VERTEX OPERATIONS
📖 Theory

Deleting a vertex means: (1) Remove all edges from other vertices pointing to it. (2) Free the vertex's own adjacency list. (3) Shift array indices if needed or mark as deleted. This can be complex in a fixed-array implementation.

C — Delete Vertex
void deleteVertex(struct Graph* graph, int v) {
    if (v < 0 || v >= graph->numVertices) {
        printf("Vertex not found!\n");
        return;
    }

    /* Step 1: Free the vertex's own adjacency list */
    struct Node* curr = graph->adjList[v];
    while (curr != NULL) {
        struct Node* temp = curr;
        curr = curr->next;
        free(temp);
    }
    graph->adjList[v] = NULL;

    /* Step 2: Remove all edges pointing TO vertex v */
    for (int i = 0; i < graph->numVertices; i++) {
        if (i == v) continue;
        struct Node** ptr = &graph->adjList[i];
        while (*ptr != NULL) {
            if ((*ptr)->vertex == v) {
                struct Node* del = *ptr;
                *ptr = (*ptr)->next;
                free(del);
            } else ptr = &(*ptr)->next;
        }
    }
    printf("Vertex %d deleted.\n", v);
}
🌍 Analogy
Like removing a student who transferred schools. You erase their name from the register AND remove them from every other student's "friend list".

Insert Edge
SECTION 04 · EDGE OPERATIONS
📖 Theory

Adding an edge (u, v) means: prepend vertex v to u's adjacency list. For an undirected graph, also prepend u to v's adjacency list. This connects two previously separate or same-component vertices.

Edge Insertion Visualization
Click "Add Edge" to connect two nodes.
C — Insert Edge (Undirected)
void insertEdge(struct Graph* graph, int u, int v) {
    /* Add v to u's list */
    struct Node* newNode = createNode(v);
    newNode->next = graph->adjList[u];
    graph->adjList[u] = newNode;

    /* Add u to v's list (undirected graph) */
    newNode = createNode(u);
    newNode->next = graph->adjList[v];
    graph->adjList[v] = newNode;

    printf("Edge (%d-%d) inserted.\n", u, v);
}
🌍 Analogy
Building a road between two cities. Once the road exists, travel is possible in both directions (undirected). If it's a one-way road, it's a directed edge.

Delete Edge
EDGE OPERATIONS
C — Delete Edge
void deleteEdge(struct Graph* graph, int u, int v) {
    /* Remove v from u's list */
    struct Node** ptr = &graph->adjList[u];
    while (*ptr != NULL) {
        if ((*ptr)->vertex == v) {
            struct Node* del = *ptr;
            *ptr = (*ptr)->next;
            free(del);
            break;
        }
        ptr = &(*ptr)->next;
    }

    /* Remove u from v's list (undirected) */
    ptr = &graph->adjList[v];
    while (*ptr != NULL) {
        if ((*ptr)->vertex == u) {
            struct Node* del = *ptr;
            *ptr = (*ptr)->next;
            free(del);
            break;
        }
        ptr = &(*ptr)->next;
    }
    printf("Edge (%d-%d) deleted.\n", u, v);
}

int searchEdge(struct Graph* graph, int u, int v) {

    while (curr != NULL) {
        if (curr->vertex == v) {
            printf("Edge (%d-%d) exists!\n", u, v);

    printf("Edge (%d-%d) not found.\n", u, v);
    return 0;
}

Breadth First Search (BFS)
SECTION 05 · TRAVERSAL OPERATIONS
📖 Theory — Multi-Component BFS

BFS explores all nodes level by level using a queue. In a disconnected graph, a single BFS call may not visit all nodes. We must loop over all vertices and restart BFS from any unvisited node to cover all components.

  • Queue Initialization: Start with the source node
  • Visit Node: Print/process it
  • Mark Visited: Avoid re-visiting
  • Traverse Adjacent Nodes: Add unvisited neighbors to queue
  • Multi-Component BFS: Repeat from every unvisited start

C — Multi-Component BFS
#include <stdio.h>
#include <stdlib.h>

int queue[MAX], front = 0, rear = 0;
int visited[MAX];

void enqueue(int v) { queue[rear++] = v; }
int  dequeue()      { return queue[front++]; }
int  isEmpty()      { return front == rear; }

/* BFS from a single source */
void BFS(struct Graph* graph, int start) {
    visited[start] = 1;
    enqueue(start);

    while (!isEmpty()) {
        int v = dequeue();
        printf("%d ", v);  /* Visit node */

        struct Node* curr = graph->adjList[v];
        while (curr != NULL) {
            if (!visited[curr->vertex]) {
                visited[curr->vertex] = 1;
                enqueue(curr->vertex);
            }
            curr = curr->next;
        }
    }
}

/* Multi-Component BFS — handles disconnected graph */
void BFSAll(struct Graph* graph) {
    for (int i = 0; i < graph->numVertices; i++)
        visited[i] = 0;  /* reset visited */

    for (int i = 0; i < graph->numVertices; i++) {
        if (!visited[i]) {
            printf("\nComponent starting at %d: ", i);
            front = rear = 0; /* reset queue */
            BFS(graph, i);
        }
    }
}
🌍 Analogy
Flooding water on a map: Pour water at a city. It spreads level by level through roads. When it stops, pour water at the next dry city (new component). Repeat until every city is flooded (visited).

Depth First Search (DFS)
TRAVERSAL OPERATIONS
📖 Theory — Multi-Component DFS

DFS explores as deep as possible before backtracking. Uses recursion (implicit stack) or explicit stack. For disconnected graphs, like BFS, we loop over all vertices and restart DFS from unvisited ones.

  • Recursive DFS: Visit → recurse on neighbors → backtrack
  • Stack DFS: Push start, pop and visit, push unvisited neighbors
  • Backtracking: When all neighbors visited, go back to previous node
  • Multi-Component DFS: Restart from every unvisited vertex

Click "Step DFS" to begin depth-first traversal.
C — Recursive DFS + Multi-Component
int visited[MAX];

/* Recursive DFS from one vertex */
void DFS(struct Graph* graph, int v) {
    visited[v] = 1;
    printf("%d ", v);  /* Visit node */

    struct Node* curr = graph->adjList[v];
    while (curr != NULL) {
        if (!visited[curr->vertex])
            DFS(graph, curr->vertex); /* Recurse */
        curr = curr->next;
    }
    /* Backtracking happens automatically here */
}

/* Multi-Component DFS */
void DFSAll(struct Graph* graph) {
    for (int i = 0; i < graph->numVertices; i++)
        visited[i] = 0;

    for (int i = 0; i < graph->numVertices; i++) {
        if (!visited[i]) {
            printf("\nComponent %d: ", i);
            DFS(graph, i);
        }
    }
}

/* Stack-based DFS (iterative) */
void DFSStack(struct Graph* graph, int start) {
    int stack[MAX], top = -1;
    stack[++top] = start;
    visited[start] = 1;

    while (top >= 0) {
        int v = stack[top--]; /* Pop */
        printf("%d ", v);

        struct Node* curr = graph->adjList[v];
        while (curr != NULL) {
            if (!visited[curr->vertex]) {
                visited[curr->vertex] = 1;
                stack[++top] = curr->vertex; /* Push */
            }
            curr = curr->next;
        }
    }
}
🌍 Analogy
Exploring a cave system: Walk as deep as you can into one tunnel. When it dead-ends, backtrack and try another tunnel. When the entire cave system (component) is explored, move to the next separate cave system.

Check Disconnected Graph
SECTION 06 · CONNECTIVITY OPERATIONS
📖 Theory

A graph is disconnected if running BFS/DFS from any single vertex does NOT visit all other vertices. Count visited nodes and compare to total — if less, it's disconnected.

C — Check if Graph is Disconnected
int isDisconnected(struct Graph* graph) {
    int visited[MAX] = {0};
    int visitedCount = 0;

    /* Run DFS from vertex 0 */
    DFS(graph, 0);

    /* Count how many were visited */
    for (int i = 0; i < graph->numVertices; i++)
        if (visited[i]) visitedCount++;

    if (visitedCount < graph->numVertices) {
        printf("Graph IS disconnected!\n");
        return 1;
    }
    printf("Graph is connected.\n");
    return 0;
}

Count & Find Connected Components
CONNECTIVITY OPERATIONS
Component Detection

C — Count & Find Connected Components
int componentId[MAX]; /* which component each vertex belongs to */

void DFSComponent(struct Graph* g, int v, int compNum) {
    componentId[v] = compNum;
    struct Node* curr = g->adjList[v];
    while (curr != NULL) {
        if (componentId[curr->vertex] == -1)
            DFSComponent(g, curr->vertex, compNum);
        curr = curr->next;
    }
}

int countComponents(struct Graph* g) {
    for (int i = 0; i < g->numVertices; i++)
        componentId[i] = -1; /* unvisited */

    int count = 0;
    for (int i = 0; i < g->numVertices; i++) {
        if (componentId[i] == -1) {
            DFSComponent(g, i, count);
            count++;
        }
    }
    printf("Total connected components: %d\n", count);
    return count;
}

void findComponents(struct Graph* g) {
    int total = countComponents(g);
    for (int c = 0; c < total; c++) {
        printf("Component %d: ", c);
        for (int i = 0; i < g->numVertices; i++)
            if (componentId[i] == c)
                printf("%d ", i);
        printf("\n");
    }
}
🌍 Analogy
Coloring countries on a map with different colors for each isolated continent. Each color group = one connected component.

Detect Isolated Vertices
CONNECTIVITY OPERATIONS
📖 Theory

An isolated vertex has degree zero — no edges connect it to any other vertex. Its adjacency list is NULL. In disconnected graphs, isolated vertices form their own component of size 1.

void detectIsolatedVertices(struct Graph* g) {
    int found = 0;

        if (g->adjList[i] == NULL) {
            printf("%d ", i);
            found = 1;
        }

    if (!found) printf("None");
    printf("\n");
}

A person with absolutely no friends in a social network. Their friend list is completely empty — they are isolated.

Reachability Check & Connectivity Restoration
CONNECTIVITY OPERATIONS
📖 Theory

Reachability: Can vertex u reach vertex v? Run DFS/BFS from u and check if v gets visited. If not, they are in different components.

Connectivity Restoration: Add edges between components to make the graph connected. Minimum edges needed = (number of components − 1).

C — Reachability Check
int isReachable(struct Graph* g, int src, int dest) {
    int visited[MAX] = {0};
    /* Use BFS from src */
    int queue[MAX], front = 0, rear = 0;
    queue[rear++] = src;
    visited[src] = 1;

    while (front != rear) {
        int v = queue[front++];
        if (v == dest) {
            printf("%d can reach %d\n", src, dest);
            return 1;
        }
        struct Node* curr = g->adjList[v];
        while (curr) {
            if (!visited[curr->vertex]) {
                visited[curr->vertex] = 1;
                queue[rear++] = curr->vertex;
            }
            curr = curr->next;
        }
    }
    printf("%d CANNOT reach %d (different component)\n", src, dest);
    return 0;
}

/* Connectivity Restoration: connect all components */
void restoreConnectivity(struct Graph* g) {
    int prevComp = -1;
    for (int i = 0; i < g->numVertices; i++) {
        if (componentId[i] != prevComp && prevComp != -1) {
            /* Find the first vertex of previous component */
            int prev = 0;
            while (componentId[prev] != prevComp) prev++;
            insertEdge(g, prev, i); /* bridge the gap */
            printf("Added bridge edge: %d - %d\n", prev, i);
        }
        prevComp = componentId[i];
    }
}

Weak & Strong Connectivity Check
CONNECTIVITY OPERATIONS
📖 Theory

Weakly Connected: A directed graph is weakly connected if treating all directed edges as undirected makes it connected.

Strongly Connected: Every vertex can reach every other vertex via directed paths. For undirected graphs, "strongly connected" = "connected".

C — Weak Connectivity
/* Convert directed to undirected,
   then check if connected */
int isWeaklyConnected(struct Graph* g) {
    /* Add reverse edges temporarily */
    struct Graph* ug = createGraph(g->numVertices);
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* c = g->adjList[i];
        while (c) {
            insertEdge(ug, i, c->vertex);
            c = c->next;
        }
    }
    return !isDisconnected(ug);
}
C — Strong Connectivity
/* Kosaraju's basic check */
int isStronglyConnected(struct Graph* g) {
    /* Step 1: DFS from vertex 0 */
    for (int i=0; i<g->numVertices; i++)
        visited[i] = 0;
    DFS(g, 0);
    for (int i=0; i<g->numVertices; i++)
        if (!visited[i]) return 0;

    /* Step 2: DFS on reversed graph */
    struct Graph* rev = reverseGraph(g);
    for (int i=0; i<g->numVertices; i++)
        visited[i] = 0;
    DFS(rev, 0);
    for (int i=0; i<g->numVertices; i++)
        if (!visited[i]) return 0;
    return 1;
}

Label, Merge & Separate Components
SECTION 07 · COMPONENT OPERATIONS
📖 Theory

Label Components: Assign a unique ID to each component (done in countComponents above).

Merge Components: Add an edge between two different components — they become one.

Separate Components: Remove a bridge edge — the component splits into two.

Traverse Each Component: Use component IDs to iterate only within a specific group.

C — Merge & Separate Components
/* Merge: simply add an edge between the two components */
void mergeComponents(struct Graph* g, int u, int v) {
    if (componentId[u] == componentId[v]) {
        printf("Already in same component!\n");
        return;
    }
    insertEdge(g, u, v);  /* Bridge the two components */
    printf("Merged component %d and %d\n", componentId[u], componentId[v]);
    countComponents(g);   /* Recompute component IDs */
}

/* Separate: delete the edge linking two components */
void separateComponents(struct Graph* g, int u, int v) {
    deleteEdge(g, u, v);
    printf("Separated at edge (%d-%d)\n", u, v);
    countComponents(g);   /* Recompute */
}

/* Traverse only one specific component */
void traverseComponent(struct Graph* g, int compNum) {
    printf("Component %d: ", compNum);
    for (int i = 0; i < g->numVertices; i++)
        if (componentId[i] == compNum)
            printf("%d ", i);
    printf("\n");
}
🌍 Analogy
Countries & Borders: Merging = two countries sign a travel treaty (new road). Separating = closing a border crossing (delete edge). Each country is a component!

Path Operations
SECTION 08 · PATH OPERATIONS
📖 Theory

Path operations determine routes between vertices. Key operations:

  • Find Path: Any path from u to v (DFS-based)
  • Shortest Path: Minimum hops from u to v (BFS-based)
  • All Paths: All distinct paths from u to v
  • Reachable Path Check: Same as reachability
  • Unreachable Node Detection: Vertices not reachable from source

⚠️ In a disconnected graph, paths between different components are impossible.

C — Find Path (DFS) + Shortest Path (BFS)
int path[MAX], pathLen = 0;

/* Find ANY path from src to dest using DFS */
int findPath(struct Graph* g, int curr, int dest, int* vis) {
    vis[curr] = 1;
    path[pathLen++] = curr;
    if (curr == dest) return 1;

    struct Node* node = g->adjList[curr];
    while (node) {
        if (!vis[node->vertex])
            if (findPath(g, node->vertex, dest, vis)) return 1;
        node = node->next;
    }
    pathLen--; /* Backtrack */
    return 0;
}

/* Shortest path using BFS */
void shortestPath(struct Graph* g, int src, int dest) {
    int dist[MAX], parent[MAX];
    for (int i=0; i<g->numVertices; i++) { dist[i]=-1; parent[i]=-1; }
    int queue[MAX], front=0, rear=0;

    dist[src] = 0;
    queue[rear++] = src;

    while (front != rear) {
        int v = queue[front++];
        struct Node* curr = g->adjList[v];
        while (curr) {
            if (dist[curr->vertex] == -1) {
                dist[curr->vertex] = dist[v] + 1;
                parent[curr->vertex] = v;
                queue[rear++] = curr->vertex;
            }
            curr = curr->next;
        }
    }
    if (dist[dest] == -1)
        printf("No path (different component!)\n");
    else
        printf("Shortest distance: %d\n", dist[dest]);
}

/* Find unreachable nodes from source */
void unreachableNodes(struct Graph* g, int src) {
    int vis[MAX] = {0};
    DFS(g, src);
    printf("Unreachable from %d: ", src);
    for (int i = 0; i < g->numVertices; i++)
        if (!vis[i]) printf("%d ", i);
    printf("\n");
}
🌍 Analogy
GPS Navigation: "Find path" = any route from A to B. "Shortest path" = fastest route. "Unreachable" = destination on an island with no bridge — GPS says "No route found!"

Cycle Operations
SECTION 09 · CYCLE OPERATIONS
📖 Theory

A cycle is a path that starts and ends at the same vertex. In disconnected graphs, a cycle might exist in one component but not another.

  • Detect Cycle: DFS — if we visit a node that's already in the current path, a cycle exists
  • Count Cycles: More complex — requires back edge analysis
  • Check Cyclic Components: Identify which components have cycles
  • Remove Cycle: Delete back edges (edges that form cycles)
C — Detect & Remove Cycles
/* Cycle detection using DFS (undirected graph) */
int detectCycleDFS(struct Graph* g, int v, int* vis, int parent) {
    vis[v] = 1;
    struct Node* curr = g->adjList[v];
    while (curr != NULL) {
        if (!vis[curr->vertex]) {
            if (detectCycleDFS(g, curr->vertex, vis, v))
                return 1; /* Cycle found */
        } else if (curr->vertex != parent) {
            printf("Cycle detected! Back edge: %d-%d\n", v, curr->vertex);
            return 1; /* Back edge = cycle */
        }
        curr = curr->next;
    }
    return 0;
}

/* Check all components for cycles */
void checkAllCycles(struct Graph* g) {
    int vis[MAX] = {0};
    for (int i = 0; i < g->numVertices; i++) {
        if (!vis[i]) {
            if (detectCycleDFS(g, i, vis, -1))
                printf("Component of %d has a cycle!\n", i);
            else
                printf("Component of %d is acyclic (tree/forest)\n", i);
        }
    }
}

/* Remove cycle: delete the back edge */
void removeCycle(struct Graph* g, int u, int v) {
    deleteEdge(g, u, v);
    printf("Cycle removed by deleting edge %d-%d\n", u, v);
}
🌍 Analogy
Road network: A cycle = a ring road where you can drive in a loop. Removing a cycle = closing one road so traffic can only travel one way through the area.

Degree Operations
SECTION 10 · DEGREE OPERATIONS
📖 Theory

The degree of a vertex = number of edges connected to it.

  • Undirected graph: degree = length of adjacency list
  • Directed graph: in-degree (edges coming in) + out-degree (edges going out)
  • Isolated vertex: degree = 0
C — Degree Operations
/* Out-degree: length of adjacency list */
int outDegree(struct Graph* g, int v) {
    int count = 0;
    struct Node* curr = g->adjList[v];
    while (curr) { count++; curr = curr->next; }
    return count;
}

/* In-degree: count appearances in other lists */
int inDegree(struct Graph* g, int v) {
    int count = 0;
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* curr = g->adjList[i];
        while (curr) {
            if (curr->vertex == v) count++;
            curr = curr->next;
        }
    }
    return count;
}

/* Max and Min degree across all vertices */
void maxMinDegree(struct Graph* g) {
    int maxD = 0, minD = g->numVertices;
    int maxV = 0, minV = 0;
    for (int i = 0; i < g->numVertices; i++) {
        int d = outDegree(g, i);
        if (d > maxD) { maxD = d; maxV = i; }
        if (d < minD) { minD = d; minV = i; }
    }
    printf("Max degree: vertex %d with degree %d\n", maxV, maxD);
    printf("Min degree: vertex %d with degree %d\n", minV, minD);
}

/* Display all degrees */
void displayDegrees(struct Graph* g) {
    for (int i = 0; i < g->numVertices; i++)
        printf("Vertex %d: degree = %d\n", i, outDegree(g, i));
}
🌍 Analogy
Social media followers: In-degree = number of followers. Out-degree = number of people you follow. A celebrity (hub vertex) has very high in-degree. A hermit (isolated) has degree 0.

Structural Operations
SECTION 11 · STRUCTURAL OPERATIONS
📖 Theory
  • Graph Size: Number of edges |E|
  • Graph Order: Number of vertices |V|
  • Density: ratio of actual edges to max possible edges
  • Sparse/Dense: Sparse if few edges relative to vertices; Dense if many
  • Forest Validation: A disconnected acyclic graph is a Forest (collection of Trees)
  • Component Analysis: Combine all the above per component
C — Structural Operations
int graphOrder(struct Graph* g) { return g->numVertices; }
int graphSize(struct Graph* g)  { return countEdges(g); }

/* Density = 2*E / (V*(V-1)) for undirected graph */
float graphDensity(struct Graph* g) {
    int V = g->numVertices;
    int E = countEdges(g);
    if (V <= 1) return 0.0f;
    float density = (float)(2 * E) / (V * (V - 1));
    printf("Density: %.3f\n", density);
    return density;
}

void sparseOrDense(struct Graph* g) {
    float d = graphDensity(g);
    if (d < 0.5f)
        printf("Graph is SPARSE (density < 0.5)\n");
    else
        printf("Graph is DENSE (density >= 0.5)\n");
}

/* Forest = disconnected acyclic graph
   A forest has exactly V - #components edges */
int isForest(struct Graph* g) {
    int E = countEdges(g);
    int C = countComponents(g);
    if (E == g->numVertices - C) {
        printf("Graph is a FOREST!\n");
        return 1;
    }
    printf("Not a forest.\n");
    return 0;
}
🌍 Analogy
Forest analogy: A forest = multiple separate trees. Each tree = one connected component with no cycles. A disconnected acyclic graph IS literally called a Forest in graph theory!

Union-Find / Disjoint Set Union (DSU)
SECTION 12 · ADVANCED ALGORITHMS
📖 Theory

Union-Find (DSU) is a specialized data structure for tracking which component each vertex belongs to. It supports two operations:

  • Find(x): Which component is x in? (returns root/representative)
  • Union(x, y): Merge the components of x and y

Optimizations: Path Compression (flatten tree on Find) + Union by Rank (smaller tree under larger). Time: near O(1) amortized — O(α(n)) where α is inverse Ackermann.

Union-Find Visualization
Each node starts as its own component. Click Union to merge.
C — Union-Find with Path Compression + Rank
int parent[MAX], rank_uf[MAX];

/* Initialize: each vertex is its own component */
void initDSU(int n) {
    for (int i = 0; i < n; i++) {
        parent[i] = i;     /* self-parent */
        rank_uf[i] = 0;   /* all same rank */
    }
}

/* Find root with path compression */
int find(int x) {
    if (parent[x] != x)
        parent[x] = find(parent[x]); /* compress */
    return parent[x];
}

/* Union by rank */
void unionSets(int x, int y) {
    int rx = find(x), ry = find(y);
    if (rx == ry) { printf("Same set!\n"); return; }

    if (rank_uf[rx] < rank_uf[ry])     parent[rx] = ry;
    else if (rank_uf[rx] > rank_uf[ry]) parent[ry] = rx;
    else { parent[ry] = rx; rank_uf[rx]++; }
    printf("Merged sets of %d and %d\n", x, y);
}

/* Count components using DSU */
int countComponentsDSU(int n) {
    int count = 0;
    for (int i = 0; i < n; i++)
        if (find(i) == i) count++; /* root = component rep */
    printf("Components: %d\n", count);
    return count;
}
🌍 Analogy
Groups in a class project: Initially everyone is in their own group. Union(A, B) = merge A's group with B's group. Find(X) = "Who is X's group leader?" Path compression = remembering the group leader directly instead of asking the whole chain.

Strongly Connected Components (Kosaraju's)
ADVANCED ALGORITHMS
📖 Theory — Kosaraju's Algorithm

An SCC is a maximal group where every vertex can reach every other. Kosaraju's Algorithm uses two DFS passes:

  1. DFS on original graph → record finish times (push to stack)
  2. Transpose (reverse) the graph
  3. DFS on transposed graph in reverse finish order → each DFS = one SCC

Time: O(V + E)

C — Kosaraju's SCC Algorithm
int stack[MAX], stackTop = -1;
int visited[MAX];

/* Step 1: DFS and push finish times */
void dfsFinish(struct Graph* g, int v) {
    visited[v] = 1;
    struct Node* curr = g->adjList[v];
    while (curr) {
        if (!visited[curr->vertex])
            dfsFinish(g, curr->vertex);
        curr = curr->next;
    }
    stack[++stackTop] = v; /* push when done */
}

/* Step 2: Transpose the graph */
struct Graph* transpose(struct Graph* g) {
    struct Graph* t = createGraph(g->numVertices);
    for (int v = 0; v < g->numVertices; v++) {
        struct Node* curr = g->adjList[v];
        while (curr) {
            /* Reverse edge: curr->v becomes v->curr */
            struct Node* n = createNode(v);
            n->next = t->adjList[curr->vertex];
            t->adjList[curr->vertex] = n;
            curr = curr->next;
        }
    }
    return t;
}

/* Step 3: Kosaraju's main function */
void kosarajuSCC(struct Graph* g) {
    for (int i=0; i<g->numVertices; i++) visited[i]=0;
    for (int i=0; i<g->numVertices; i++)
        if (!visited[i]) dfsFinish(g, i);

    struct Graph* t = transpose(g);
    for (int i=0; i<g->numVertices; i++) visited[i]=0;

    printf("Strongly Connected Components:\n");
    while (stackTop >= 0) {
        int v = stack[stackTop--];
        if (!visited[v]) {
            printf("SCC: ");
            DFS(t, v);
            printf("\n");
        }
    }
}
🌍 Analogy
One-way streets: An SCC is a group of intersections where you can drive from any intersection to any other using one-way roads. Think of roundabouts within a city district.

Graph Coloring
ADVANCED ALGORITHMS
📖 Theory

Graph coloring assigns colors to vertices so that no two adjacent vertices share the same color. The minimum colors needed = chromatic number χ(G). For disconnected graphs, each component is colored independently.

Greedy Coloring: For each vertex, assign the smallest color not used by its neighbors.

Graph Coloring Visualization
Click "Color Graph" to apply greedy coloring.
C — Greedy Graph Coloring
int color[MAX];

void greedyColoring(struct Graph* g) {
    for (int i=0; i<g->numVertices; i++) color[i] = -1;

    color[0] = 0; /* First vertex gets color 0 */

    for (int v = 1; v < g->numVertices; v++) {
        int available[MAX];
        for (int i=0; i<g->numVertices; i++) available[i] = 1;

        /* Mark colors used by neighbors */
        struct Node* curr = g->adjList[v];
        while (curr) {
            if (color[curr->vertex] != -1)
                available[color[curr->vertex]] = 0;
            curr = curr->next;
        }

        /* Assign smallest available color */
        int c;
        for (c = 0; c < g->numVertices; c++)
            if (available[c]) break;
        color[v] = c;
    }

    for (int i=0; i<g->numVertices; i++)
        printf("Vertex %d -> Color %d\n", i, color[i]);
}
🌍 Analogy
Exam scheduling: Courses that share students can't be in the same time slot. Color = time slot. Two adjacent courses (sharing students) get different colors (different time slots). Minimize colors = minimize exam days!

Spanning Forest
ADVANCED ALGORITHMS
📖 Theory

A Spanning Forest of a disconnected graph contains a Spanning Tree for each connected component. It uses exactly V − C edges (V = vertices, C = components) with no cycles, and connects all vertices within each component.

Built using DFS/BFS — edges traversed (excluding back-edges) form the spanning forest.

C — Build Spanning Forest using DFS
/* Edge list to store spanning forest edges */
struct Edge { int u, v; };
struct Edge forestEdges[MAX];
int forestSize = 0;

void dfsBuildForest(struct Graph* g, int v, int* vis) {
    vis[v] = 1;
    struct Node* curr = g->adjList[v];
    while (curr) {
        if (!vis[curr->vertex]) {
            /* Tree edge -> part of spanning forest */
            forestEdges[forestSize].u = v;
            forestEdges[forestSize].v = curr->vertex;
            forestSize++;
            dfsBuildForest(g, curr->vertex, vis);
        }
        curr = curr->next;
    }
}

void buildSpanningForest(struct Graph* g) {
    int vis[MAX] = {0};
    forestSize = 0;
    for (int i=0; i<g->numVertices; i++)
        if (!vis[i]) dfsBuildForest(g, i, vis);

    printf("Spanning Forest Edges (%d total):\n", forestSize);
    for (int i=0; i<forestSize; i++)
        printf("  %d - %d\n", forestEdges[i].u, forestEdges[i].v);
}
🌍 Analogy
Minimum roads to connect all cities: Each island (component) builds its own minimum road network (spanning tree) to connect all internal cities. The entire set of island networks = spanning forest.

Utility Operations
SECTION 13 · UTILITY OPERATIONS
📖 Theory

Essential utility functions to manage the graph's lifecycle:

  • Is Empty: No vertices in the graph
  • Clear Graph: Remove all edges but keep vertices
  • Copy Graph: Deep copy — duplicate all nodes and lists
  • Display Graph: Print full adjacency list
  • Reset Visited Array: Set all visited[] entries to 0
C — All Utility Operations
/* Is graph empty? */
int isEmpty(struct Graph* g) {
    return g->numVertices == 0;
}

/* Clear all edges, keep vertices */
void clearGraph(struct Graph* g) {
    for (int i = 0; i < g->numVertices; i++) {
        struct Node* curr = g->adjList[i];
        while (curr) {
            struct Node* temp = curr;
            curr = curr->next;
            free(temp);
        }
        g->adjList[i] = NULL;
    }
    printf("Graph cleared.\n");
}

/* Deep copy of a graph */
struct Graph* copyGraph(struct Graph* src) {
    struct Graph* dest = createGraph(src->numVertices);
    for (int i = 0; i < src->numVertices; i++) {
        struct Node* curr = src->adjList[i];
        /* Build list in reverse to preserve order */
        struct Node* tail = NULL;
        while (curr) {
            struct Node* n = createNode(curr->vertex);
            if (!dest->adjList[i]) dest->adjList[i] = n;
            else tail->next = n;
            tail = n;
            curr = curr->next;
        }
    }
    return dest;
}

/* Reset visited array */
void resetVisited(int* vis, int n) {
    for (int i = 0; i < n; i++) vis[i] = 0;
}

/* Display complete adjacency list */
void displayGraph(struct Graph* g) {
    printf("=== Graph (V=%d) ===\n", g->numVertices);
    for (int i = 0; i < g->numVertices; i++) {
        printf("[%d] ->", i);
        struct Node* curr = g->adjList[i];
        while (curr) {
            printf(" %d ->", curr->vertex);
            curr = curr->next;
        }
        printf(" NULL\n");
    }
}

#include <stdio.h>
#include <stdlib.h>
#define MAX 100

struct Node { int vertex; struct Node* next; };
struct Graph { int V; struct Node* adj[MAX]; };
int vis[MAX];

struct Graph* createGraph(int v) {
    struct Graph* g = malloc(sizeof(struct Graph));
    g->V = v;
    for(int i=0;i<v;i++) g->adj[i]=NULL;
    return g;
}
void addEdge(struct Graph* g, int u, int v) {
    struct Node* n = malloc(sizeof(struct Node));
    n->vertex=v; n->next=g->adj[u]; g->adj[u]=n;
    n = malloc(sizeof(struct Node));
    n->vertex=u; n->next=g->adj[v]; g->adj[v]=n;
}
void dfs(struct Graph* g, int v) {
    vis[v]=1; printf("%d ",v);
    for(struct Node* c=g->adj[v]; c; c=c->next)
        if(!vis[c->vertex]) dfs(g,c->vertex);
}
int main() {
    struct Graph* g = createGraph(7);
    /* Component 1: 0-1-2 */
    addEdge(g, 0, 1); addEdge(g, 1, 2);
    /* Component 2: 3-4 */
    addEdge(g, 3, 4);
    /* Component 3: 5-6 */
    addEdge(g, 5, 6);
    /* Vertex 7 is isolated (no edges) */

    printf("BFS/DFS of disconnected graph:\n");
    for(int i=0; i<g->V; i++) vis[i]=0;
    int comp=0;
    for(int i=0; i<g->V; i++) {
        if(!vis[i]) {
            printf("\nComponent %d: ", ++comp);
            dfs(g, i);
        }
    }
    printf("\nTotal components: %d\n", comp);
    return 0;
}
