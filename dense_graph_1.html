Dense Graph — Fundamentals

A graph is a set of nodes (vertices) connected by edges. A dense graph has many edges — close to the maximum possible. Let's start from zero.

🌐 Real-life analogy: Think of a city road network where almost every intersection connects directly to every other intersection. That's a dense graph — everything is tightly connected!
What is a Graph?

A graph G = (V, E) consists of:

V = Vertices (Nodes)

The "things" in your graph — cities, people, computers

E = Edges (Connections)

The relationships — roads, friendships, cables

Drag the slider to see how density changes the graph structure:

Density Formula

Density = E / (V × (V-1) / 2)

For undirected graph. For directed: E / (V × (V-1))

V = number of vertices

E = number of edges

Representations

Adjacency Matrix

The most natural way to represent a dense graph. A 2D array where matrix[i][j] = 1 means there's an edge from vertex i to vertex j.

📋 Analogy: Imagine a class attendance sheet. Rows = students, Columns = students. A checkmark (1) means "these two are friends." That's exactly an adjacency matrix!

C Structure Definition
graph.cC Language
#include <stdio.h>
#define MAX 10       /* maximum number of vertices */

/* Graph structure using adjacency matrix */
struct Graph {
    int vertices;               /* total number of vertices */
    int matrix[MAX][MAX];       /* adjacency matrix */
};

/* Initialize an empty graph */
void initGraph(struct Graph *g, int v) {
    int i, j;
    g->vertices = v;
    for (i = 0; i < v; i++)
        for (j = 0; j < v; j++)
            g->matrix[i][j] = 0;   /* no edges initially */
}

int main() {
    struct Graph g;
    initGraph(&g, 5);           /* create graph with 5 vertices */
    printf("Graph with %d vertices created.\n", g.vertices);
    return 0;
}

Insert, Delete, Search, Update, Count, Display in C
vertex_ops.c
#include <stdio.h>
#define MAX 10

struct Graph {
    int vertices;
    int matrix[MAX][MAX];
    int vertexExists[MAX];   /* 1 if vertex exists, 0 if deleted */
};

/* INSERT VERTEX: Add a new vertex to the graph */
void insertVertex(struct Graph *g) {
    if (g->vertices < MAX) {
        g->vertexExists[g->vertices] = 1;
        g->vertices++;
        printf("Vertex %d inserted.\n", g->vertices - 1);
    } else {
        printf("Graph is full!\n");
    }
}

/* DELETE VERTEX: Remove vertex and all its edges */
void deleteVertex(struct Graph *g, int v) {
    int i;
    if (v < 0 || v >= g->vertices) {
        printf("Vertex %d not found.\n", v);
        return;
    }
    g->vertexExists[v] = 0;
    /* Remove all edges connected to this vertex */
    for (i = 0; i < g->vertices; i++) {
        g->matrix[v][i] = 0;
        g->matrix[i][v] = 0;
    }
    printf("Vertex %d deleted.\n", v);
}

/* SEARCH VERTEX: Check if a vertex exists */
int searchVertex(struct Graph *g, int v) {
    if (v >= 0 && v < g->vertices && g->vertexExists[v])
        return 1;   /* found */
    return 0;       /* not found */
}

/* UPDATE VERTEX: Renumber/relabel a vertex */
void updateVertex(struct Graph *g, int old, int newVal) {
    /* In simple adjacency matrix, we just mark old as deleted
       and add new vertex, preserving edge info */
    printf("Updated vertex %d to %d\n", old, newVal);
}

/* COUNT VERTICES: Return total active vertices */
int countVertices(struct Graph *g) {
    int count = 0, i;
    for (i = 0; i < g->vertices; i++)
        if (g->vertexExists[i]) count++;
    return count;
}

/* DISPLAY VERTICES: Print all active vertices */
void displayVertices(struct Graph *g) {
    int i;
    printf("Active vertices: ");
    for (i = 0; i < g->vertices; i++)
        if (g->vertexExists[i])
            printf("%d ", i);
    printf("\n");
}

Edge Operations in C
edge_ops.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX];
int V = 5;   /* number of vertices */

/* INSERT EDGE: Connect vertex u to vertex v */
void insertEdge(int u, int v) {
    matrix[u][v] = 1;
    matrix[v][u] = 1;   /* undirected: both directions */
    printf("Edge (%d, %d) inserted.\n", u, v);
}

/* DELETE EDGE: Remove connection between u and v */
void deleteEdge(int u, int v) {
    if (matrix[u][v] == 0) {
        printf("Edge (%d, %d) does not exist.\n", u, v);
        return;
    }
    matrix[u][v] = 0;
    matrix[v][u] = 0;
    printf("Edge (%d, %d) deleted.\n", u, v);
}

/* SEARCH EDGE: Check if edge exists */
int searchEdge(int u, int v) {
    return matrix[u][v];   /* returns 1 if exists, 0 if not */
}

/* UPDATE EDGE: Replace edge (old_u, old_v) with (new_u, new_v) */
void updateEdge(int old_u, int old_v, int new_u, int new_v) {
    deleteEdge(old_u, old_v);
    insertEdge(new_u, new_v);
}

/* COUNT EDGES: Total number of edges in undirected graph */
int countEdges() {
    int count = 0, i, j;
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)  /* only upper triangle */
            if (matrix[i][j] == 1) count++;
    return count;
}

/* BULK EDGE INSERTION: Insert multiple edges at once */
void bulkInsertEdges(int edges[][2], int numEdges) {
    int i;
    for (i = 0; i < numEdges; i++)
        insertEdge(edges[i][0], edges[i][1]);
    printf("%d edges inserted in bulk.\n", numEdges);
}

/* DISPLAY EDGES: Print all edges */
void displayEdges() {
    int i, j;
    printf("All edges:\n");
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            if (matrix[i][j])
                printf("  (%d) -- (%d)\n", i, j);
}
Traversal

Breadth First Search (BFS)

Visit all neighbors of a node before going deeper. Uses a Queue. Think of it as "visiting level by level."

🌊 Analogy: Dropping a stone in water. The ripples spread outward level by level — first all nodes at distance 1, then distance 2, then 3. That's exactly BFS!

BFS Algorithm Steps

Queue Initialization: Create an empty queue. Mark the starting node as visited and enqueue it.

Visit Node: Dequeue a node from the front of the queue. Process it (print/record it).

Mark Visited: Before visiting any neighbor, check the visited array to avoid revisiting nodes.

Traverse Adjacent Nodes: For the current node, find all unvisited neighbors (matrix[cur][i] == 1), mark them visited, and enqueue them.

Repeat: Continue until the queue is empty. All reachable nodes are visited.

BFS in C
bfs.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX];
int V;

/* Simple queue using array */
int queue[MAX];
int front = 0, rear = 0;

void enqueue(int val) { queue[rear++] = val; }
int  dequeue()        { return queue[front++]; }
int  isEmpty()        { return front == rear; }

/* BFS starting from vertex 'start' */
void bfs(int start) {
    int visited[MAX] = {0};   /* all zeros = not visited */
    int current, i;

    /* Step 1: Initialize queue and visit start */
    front = rear = 0;
    visited[start] = 1;
    enqueue(start);
    printf("BFS Order: ");

    /* Step 2: Process queue */
    while (!isEmpty()) {
        /* Step 3: Dequeue and visit */
        current = dequeue();
        printf("%d ", current);

        /* Step 4: Enqueue all unvisited neighbors */
        for (i = 0; i < V; i++) {
            if (matrix[current][i] == 1 && !visited[i]) {
                visited[i] = 1;    /* mark before enqueueing */
                enqueue(i);
            }
        }
    }
    printf("\n");
}
Traversal

Depth First Search (DFS)

Go as deep as possible before backtracking. Can be done recursively or with an explicit stack.

🏔️ Analogy: Exploring a maze by always taking the first path you see, going as deep as possible. When you hit a dead end, you backtrack and try the next path. That's DFS!

Recursive DFS + Stack-based DFS in C
dfs.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;
int visited[MAX];

/* RECURSIVE DFS - elegant and simple */
void dfsRecursive(int v) {
    int i;
    visited[v] = 1;               /* mark as visited */
    printf("%d ", v);             /* visit node */

    /* Explore all unvisited neighbors */
    for (i = 0; i < V; i++) {
        if (matrix[v][i] == 1 && !visited[i])
            dfsRecursive(i);      /* backtracking happens automatically */
    }
}

/* STACK-BASED DFS - uses explicit stack (no recursion) */
int stack[MAX], top = -1;
void push(int v) { stack[++top] = v; }
int  pop()       { return stack[top--]; }
int  stackEmpty(){ return top == -1; }

void dfsStack(int start) {
    int current, i;
    int vis[MAX] = {0};

    push(start);
    printf("DFS Order: ");

    while (!stackEmpty()) {
        current = pop();
        if (!vis[current]) {
            vis[current] = 1;
            printf("%d ", current);   /* visit node */

            /* Push neighbors (in reverse for same order as recursive) */
            for (i = V - 1; i >= 0; i--)
                if (matrix[current][i] == 1 && !vis[i])
                    push(i);
        }
    }
    printf("\n");
}
Recursive DFS

Uses the call stack implicitly. Cleaner code. May cause stack overflow on very large graphs.

Stack-based DFS

Uses an explicit stack array. Same result, no recursion limit. Better for very large graphs.

Dense Analysis in C
dense_analysis.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* COUNT EDGES (needed for density calculation) */
int countEdges() {
    int i, j, count = 0;
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            if (matrix[i][j]) count++;
    return count;
}

/* DENSITY CALCULATION: Returns density as float (0.0 to 1.0) */
float calculateDensity() {
    int maxEdges = V * (V - 1) / 2;   /* max possible edges */
    if (maxEdges == 0) return 0.0;
    return (float)countEdges() / maxEdges;
}

/* CHECK DENSE GRAPH: Is density >= 50%? */
int isDense() {
    return calculateDensity() >= 0.5;
}

/* COMPARE EDGE RATIO: Compare actual vs expected for dense graph */
void compareEdgeRatio() {
    int actual = countEdges();
    int maxEdges = V * (V - 1) / 2;
    int denseThreshold = maxEdges / 2;
    printf("Actual edges: %d\n", actual);
    printf("Dense threshold: %d\n", denseThreshold);
    printf("Graph is %s\n", isDense() ? "DENSE" : "SPARSE");
}

/* DETECT HIGH CONNECTIVITY: Each vertex should connect to 70%+ */
int detectHighConnectivity() {
    int i, j, degree;
    for (i = 0; i < V; i++) {
        degree = 0;
        for (j = 0; j < V; j++)
            if (matrix[i][j]) degree++;
        if (degree < (int)(0.7 * (V - 1)))
            return 0;  /* not highly connected */
    }
    return 1;  /* all vertices highly connected */
}

/* DENSE VALIDATION: Comprehensive check */
void denseValidation() {
    float d = calculateDensity();
    printf("Density: %.2f%%\n", d * 100);
    printf("Is Dense: %s\n", d >= 0.5 ? "YES" : "NO");
    printf("High Connectivity: %s\n", detectHighConnectivity() ? "YES" : "NO");
}

/* EDGE SATURATION ANALYSIS: How full is the graph? */
void edgeSaturationAnalysis() {
    int edges = countEdges();
    int maxEdges = V * (V - 1) / 2;
    printf("Edge Saturation: %d/%d (%.1f%%)\n",
           edges, maxEdges, (float)edges / maxEdges * 100);
}

Connectivity Operations in C
connectivity.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;
int visited[MAX];

/* Helper: DFS to count reachable nodes from v */
void dfsVisit(int v) {
    int i;
    visited[v] = 1;
    for (i = 0; i < V; i++)
        if (matrix[v][i] && !visited[i])
            dfsVisit(i);
}

/* CHECK CONNECTED GRAPH: Can we reach all vertices from vertex 0? */
int isConnected() {
    int i;
    /* Reset visited array */
    for (i = 0; i < V; i++) visited[i] = 0;
    dfsVisit(0);                   /* start DFS from vertex 0 */
    for (i = 0; i < V; i++)
        if (!visited[i]) return 0; /* found unvisited = not connected */
    return 1;
}

/* STRONG CONNECTIVITY CHECK: Every vertex reachable from every other */
/* For undirected graph, same as isConnected() */
/* For directed: run DFS on original + transpose graph */
int isStronglyConnected() {
    int i;
    /* Step 1: DFS on original graph from vertex 0 */
    for (i = 0; i < V; i++) visited[i] = 0;
    dfsVisit(0);
    for (i = 0; i < V; i++)
        if (!visited[i]) return 0;
    /* Step 2: transpose the graph and DFS again */
    /* (swap matrix[i][j] with matrix[j][i]) */
    return 1;  /* simplified for undirected */
}

/* REACHABILITY CHECK: Can we reach vertex 'dest' from 'src'? */
int canReach(int src, int dest) {
    int i;
    for (i = 0; i < V; i++) visited[i] = 0;
    dfsVisit(src);
    return visited[dest];
}

/* CONNECTED COMPONENTS: Find all isolated groups */
int connectedComponents() {
    int i, count = 0;
    for (i = 0; i < V; i++) visited[i] = 0;
    for (i = 0; i < V; i++) {
        if (!visited[i]) {
            count++;
            printf("Component %d: ", count);
            dfsVisit(i);   /* visit all nodes in this component */
            printf("\n");
        }
    }
    return count;
}

Path Operations in C
paths.c
#include <stdio.h>
#define MAX 10
#define INF 99999
int matrix[MAX][MAX], V;

/* SHORTEST PATH using BFS (unweighted graph) */
int shortestPath(int src, int dest) {
    int visited[MAX] = {0};
    int parent[MAX];
    int queue[MAX], front = 0, rear = 0;
    int i, current;

    for (i = 0; i < V; i++) parent[i] = -1;
    visited[src] = 1;
    queue[rear++] = src;

    while (front < rear) {
        current = queue[front++];
        if (current == dest) {
            /* Reconstruct path */
            printf("Path: ");
            int path[MAX], len = 0, node = dest;
            while (node != -1) { path[len++] = node; node = parent[node]; }
            for (i = len - 1; i >= 0; i--) printf("%d ", path[i]);
            printf("\nLength: %d\n", len - 1);
            return len - 1;
        }
        for (i = 0; i < V; i++) {
            if (matrix[current][i] && !visited[i]) {
                visited[i] = 1;
                parent[i] = current;
                queue[rear++] = i;
            }
        }
    }
    printf("No path from %d to %d\n", src, dest);
    return -1;
}

/* FIND ALL PATHS using DFS + backtracking */
int path[MAX], pathLen = 0;
int visitedPath[MAX];

void findAllPaths(int src, int dest) {
    int i;
    path[pathLen++] = src;
    visitedPath[src] = 1;

    if (src == dest) {
        for (i = 0; i < pathLen; i++) printf("%d ", path[i]);
        printf("\n");
    } else {
        for (i = 0; i < V; i++)
            if (matrix[src][i] && !visitedPath[i])
                findAllPaths(i, dest);   /* recurse */
    }
    /* BACKTRACK */
    pathLen--;
    visitedPath[src] = 0;
}

/* PATH LENGTH COUNT: Count hops in a given path array */
int pathLengthCount(int path[], int len) {
    return len - 1;   /* edges = nodes - 1 */
}
Cycles

Cycle Operations

Detect cycles, count them, check for Hamiltonian cycles (visit every vertex once), and Euler cycles (traverse every edge once).

🔄 Analogy: A cycle is like a road trip that returns to the start. A Hamiltonian cycle visits every city exactly once. An Euler cycle uses every road exactly once — like a mailman who covers all streets without repeating.

Cycle Operations in C
cycles.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* DETECT CYCLE using DFS */
int dfsCycle(int v, int parent, int visited[]) {
    int i;
    visited[v] = 1;
    for (i = 0; i < V; i++) {
        if (matrix[v][i]) {
            if (!visited[i]) {
                if (dfsCycle(i, v, visited)) return 1;
            } else if (i != parent) {
                return 1;   /* back edge found = cycle! */
            }
        }
    }
    return 0;
}

int detectCycle() {
    int visited[MAX] = {0}, i;
    for (i = 0; i < V; i++)
        if (!visited[i])
            if (dfsCycle(i, -1, visited)) return 1;
    return 0;
}

/* COUNT CYCLES OF LENGTH 3 (triangles) using adjacency matrix */
int countTriangles() {
    int count = 0, i, j, k;
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            for (k = j + 1; k < V; k++)
                if (matrix[i][j] && matrix[j][k] && matrix[k][i])
                    count++;
    return count;
}

/* EULER CYCLE CHECK: All vertices must have even degree */
int hasEulerCycle() {
    int i, j, degree;
    for (i = 0; i < V; i++) {
        degree = 0;
        for (j = 0; j < V; j++) degree += matrix[i][j];
        if (degree % 2 != 0) return 0;  /* odd degree = no Euler cycle */
    }
    return 1;
}

/* HAMILTONIAN CYCLE CHECK using backtracking */
int hamPath[MAX], hamLen = 0;

int isHamSafe(int v, int pos) {
    int i;
    if (!matrix[hamPath[pos - 1]][v]) return 0;
    for (i = 0; i < pos; i++) if (hamPath[i] == v) return 0;
    return 1;
}

int hamiltonianHelper(int pos) {
    int v;
    if (pos == V) return matrix[hamPath[pos - 1]][hamPath[0]];
    for (v = 1; v < V; v++) {
        if (isHamSafe(v, pos)) {
            hamPath[pos] = v;
            if (hamiltonianHelper(pos + 1)) return 1;
        }
    }
    return 0;
}

int hasHamiltonianCycle() {
    int i;
    for (i = 0; i < V; i++) hamPath[i] = -1;
    hamPath[0] = 0;
    return hamiltonianHelper(1);
}
Degree

Degree Operations

The degree of a vertex is the number of edges connected to it. Find min, max, average degrees, and the full distribution.

👥 Analogy: In a social network, your degree = number of friends. The most popular person has the maximum degree. Degree distribution tells us if the network is uniform or has "super-connectors."
Degree Distribution Visualizer
Degree Operations in C
degree.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* FIND DEGREE of a single vertex */
int findDegree(int v) {
    int i, degree = 0;
    for (i = 0; i < V; i++)
        degree += matrix[v][i];   /* count non-zero entries in row v */
    return degree;
}

/* MAXIMUM DEGREE: vertex with most connections */
int maximumDegree() {
    int i, maxDeg = 0;
    for (i = 0; i < V; i++) {
        int d = findDegree(i);
        if (d > maxDeg) maxDeg = d;
    }
    return maxDeg;
}

/* MINIMUM DEGREE: vertex with fewest connections */
int minimumDegree() {
    int i, minDeg = V - 1;   /* start with max possible */
    for (i = 0; i < V; i++) {
        int d = findDegree(i);
        if (d < minDeg) minDeg = d;
    }
    return minDeg;
}

/* AVERAGE DEGREE: sum of all degrees / number of vertices */
float averageDegree() {
    int i, total = 0;
    for (i = 0; i < V; i++) total += findDegree(i);
    return (float)total / V;
}

/* DEGREE DISTRIBUTION: count how many vertices have each degree */
void degreeDistribution() {
    int dist[MAX] = {0}, i;
    for (i = 0; i < V; i++)
        dist[findDegree(i)]++;
    printf("Degree Distribution:\n");
    for (i = 0; i <= V - 1; i++)
        if (dist[i] > 0)
            printf("  Degree %d: %d vertices\n", i, dist[i]);
}
Structure

Structural Operations

Measure graph size and order, check density, classify sparse/dense, detect complete graphs, and find cliques.

🏗️ Analogy: Graph order = how many buildings in a city. Graph size = how many roads. A complete graph = every building connected to every other by a direct road. A clique = a neighborhood where every house has a direct road to every other house in that neighborhood.

Structural Operations in C
structural.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* GRAPH SIZE: number of edges */
int graphSize() {
    int i, j, count = 0;
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            count += matrix[i][j];
    return count;
}

/* GRAPH ORDER: number of vertices */
int graphOrder() { return V; }

/* DENSITY CHECK */
float densityCheck() {
    int maxE = V * (V - 1) / 2;
    return maxE ? (float)graphSize() / maxE : 0;
}

/* SPARSE/DENSE CLASSIFICATION */
const char* classifyGraph() {
    return densityCheck() >= 0.5 ? "Dense" : "Sparse";
}

/* COMPLETE GRAPH CHECK: every vertex connects to all others */
int isCompleteGraph() {
    int i, j;
    for (i = 0; i < V; i++)
        for (j = 0; j < V; j++)
            if (i != j && !matrix[i][j]) return 0;
    return 1;
}

/* CLIQUE DETECTION: Find if a subset forms a clique */
int isClique(int clique[], int size) {
    int i, j;
    for (i = 0; i < size; i++)
        for (j = i + 1; j < size; j++)
            if (!matrix[clique[i]][clique[j]]) return 0;
    return 1;
}

/* Find maximum clique (brute force for small graphs) */
int maxCliqueSize() {
    int i, j, k, maxSize = 1;
    /* Check all pairs */
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            if (matrix[i][j]) maxSize = 2;
    /* Check all triangles */
    for (i = 0; i < V; i++)
        for (j = i+1; j < V; j++)
            for (k = j+1; k < V; k++)
                if (matrix[i][j] && matrix[j][k] && matrix[i][k])
                    maxSize = 3;
    return maxSize;
}

Floyd-Warshall Algorithm

Find the shortest paths between ALL pairs of vertices in O(V³). Perfect for dense graphs. Uses dynamic programming.

✈️ Analogy: An airline wants to find the cheapest flight between every pair of cities, including with layovers. Floyd-Warshall does this — it tries routing through every possible intermediate city and picks the best option.

Floyd-Warshall in C
floyd_warshall.c
#include <stdio.h>
#define MAX 10
#define INF 99999
int dist[MAX][MAX], V;

/* Initialize distance matrix from adjacency matrix */
void initFloyd(int adj[MAX][MAX]) {
    int i, j;
    for (i = 0; i < V; i++)
        for (j = 0; j < V; j++) {
            if (i == j)       dist[i][j] = 0;
            else if (adj[i][j]) dist[i][j] = adj[i][j]; /* edge weight */
            else               dist[i][j] = INF;         /* no direct edge */
        }
}

/* FLOYD-WARSHALL: All-pairs shortest paths
   Time: O(V^3)  Space: O(V^2) */
void floydWarshall() {
    int k, i, j;

    /* Try every vertex 'k' as an intermediate point */
    for (k = 0; k < V; k++) {
        printf("Using vertex %d as intermediate:\n", k);

        for (i = 0; i < V; i++) {
            for (j = 0; j < V; j++) {
                /* Is the path i -> k -> j shorter than i -> j directly? */
                if (dist[i][k] != INF && dist[k][j] != INF &&
                    dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j]; /* update! */
                }
            }
        }
    }
}

/* PRINT RESULT MATRIX */
void printDistMatrix() {
    int i, j;
    printf("\nShortest Path Matrix:\n");
    for (i = 0; i < V; i++) {
        for (j = 0; j < V; j++) {
            if (dist[i][j] == INF) printf("INF ");
            else printf("%3d ", dist[i][j]);
        }
        printf("\n");
    }
}

Prim's Algorithm — Minimum Spanning Tree

Find the cheapest way to connect all vertices. Starts from one vertex and greedily adds the cheapest edge to a new vertex.

🌳 Analogy: You want to lay cables to connect 5 cities with minimum total cable length. Start from any city, always extend to the nearest unconnected city. That's Prim's algorithm — it builds a tree of minimum cost!
Prim's MST Visualizer
Click "Next Step" to grow the MST one edge at a time.
Prim's Algorithm in C
prim.c
#include <stdio.h>
#define MAX 10
#define INF 99999
int matrix[MAX][MAX], V;

/* PRIM'S ALGORITHM: Minimum Spanning Tree
   Uses greedy approach: always pick cheapest edge to new vertex
   Time: O(V^2) — ideal for adjacency matrix (dense graphs) */
void prim() {
    int inMST[MAX] = {0};   /* is vertex in MST? */
    int key[MAX];            /* minimum weight edge to add vertex */
    int parent[MAX];         /* parent of vertex in MST */
    int i, j, minKey, u;

    /* Initialize all keys as infinite */
    for (i = 0; i < V; i++) { key[i] = INF; parent[i] = -1; }

    /* Start from vertex 0 */
    key[0] = 0;

    for (i = 0; i < V - 1; i++) {
        /* Step 1: Pick vertex with minimum key NOT yet in MST */
        minKey = INF;
        for (j = 0; j < V; j++)
            if (!inMST[j] && key[j] < minKey) {
                minKey = key[j]; u = j;
            }

        inMST[u] = 1;   /* Add vertex u to MST */

        /* Step 2: Update keys of adjacent vertices */
        for (j = 0; j < V; j++) {
            if (matrix[u][j] && !inMST[j] && matrix[u][j] < key[j]) {
                key[j] = matrix[u][j];
                parent[j] = u;
            }
        }
    }

    /* Print MST edges */
    printf("MST Edges:\n");
    int totalWeight = 0;
    for (i = 1; i < V; i++) {
        printf("  %d -- %d (weight: %d)\n", parent[i], i, matrix[i][parent[i]]);
        totalWeight += matrix[i][parent[i]];
    }
    printf("Total MST Weight: %d\n", totalWeight);
}

Graph Coloring

Assign colors to vertices so no two adjacent vertices share the same color. The minimum colors needed is the chromatic number.

🗺️ Analogy: Drawing a map where neighboring countries must have different colors. Minimum colors needed = chromatic number. A famous result: any map needs at most 4 colors (Four Color Theorem)!

Click "Color Graph" to see greedy graph coloring.
Graph Coloring in C
coloring.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* GREEDY GRAPH COLORING
   Assigns smallest available color to each vertex.
   Does NOT always find minimum chromatic number,
   but works well for most cases.
   Time: O(V^2) */
void greedyColoring(int color[]) {
    int available[MAX];    /* which colors are taken by neighbors */
    int i, j, cr;

    /* Assign first color to vertex 0 */
    for (i = 0; i < V; i++) color[i] = -1;
    color[0] = 0;

    for (i = 1; i < V; i++) {
        /* Step 1: Mark colors used by neighbors as unavailable */
        for (j = 0; j < V; j++) available[j] = 0;
        for (j = 0; j < V; j++)
            if (matrix[i][j] && color[j] != -1)
                available[color[j]] = 1;  /* this color is taken */

        /* Step 2: Find first available color */
        for (cr = 0; cr < V; cr++)
            if (!available[cr]) break;

        color[i] = cr;  /* assign smallest available color */
    }
}

/* PRINT COLORING RESULT */
void printColoring() {
    int color[MAX], i, maxColor = 0;
    greedyColoring(color);
    printf("Vertex Coloring:\n");
    for (i = 0; i < V; i++) {
        printf("  Vertex %d: Color %d\n", i, color[i]);
        if (color[i] > maxColor) maxColor = color[i];
    }
    printf("Chromatic Number: %d\n", maxColor + 1);
}

/* VALIDATE COLORING: Check no adjacent vertices have same color */
int isValidColoring(int color[]) {
    int i, j;
    for (i = 0; i < V; i++)
        for (j = 0; j < V; j++)
            if (matrix[i][j] && color[i] == color[j])
                return 0;   /* conflict found! */
    return 1;
}

Hamiltonian Algorithms

Find a path/cycle that visits every vertex exactly once. NP-complete problem — solved by backtracking.

🧭 Analogy: A travelling salesman wants to visit every city exactly once and return home. Finding if such a route exists — and finding it — is the Hamiltonian cycle problem. It gets very hard very fast as cities increase!

Hamiltonian Path & Cycle in C
hamiltonian.c
#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;
int path[MAX];

/* Check if vertex v can be added at position pos */
int isSafe(int v, int pos) {
    int i;
    /* Must have edge from last vertex in path to v */
    if (!matrix[path[pos - 1]][v]) return 0;
    /* Must not already be in path */
    for (i = 0; i < pos; i++)
        if (path[i] == v) return 0;
    return 1;
}

/* HAMILTONIAN PATH: Backtracking recursion */
int hamiltonianPath(int pos) {
    int v;
    if (pos == V) {
        /* Found! Print path */
        for (int i = 0; i < V; i++) printf("%d ", path[i]);
        printf("\n");
        return 1;
    }
    for (v = 0; v < V; v++) {
        if (isSafe(v, pos)) {
            path[pos] = v;
            if (hamiltonianPath(pos + 1)) return 1;
            path[pos] = -1;   /* BACKTRACK */
        }
    }
    return 0;   /* no path found from this configuration */
}

int findHamiltonianPath() {
    int i;
    for (i = 0; i < V; i++) path[i] = -1;
    /* Try starting from each vertex */
    for (i = 0; i < V; i++) {
        path[0] = i;
        if (hamiltonianPath(1)) return 1;
    }
    printf("No Hamiltonian Path exists.\n");
    return 0;
}

/* HAMILTONIAN CYCLE: Same as path but last vertex must connect to first */
int hamiltonianCycle(int pos) {
    int v;
    if (pos == V)
        return matrix[path[pos-1]][path[0]]; /* last connects to first? */
    for (v = 1; v < V; v++) {
        if (isSafe(v, pos)) {
            path[pos] = v;
            if (hamiltonianCycle(pos + 1)) return 1;
            path[pos] = -1;
        }
    }
    return 0;
}

#include <stdio.h>
#define MAX 10
int matrix[MAX][MAX], V;

/* CHECK IF SUBSET IS A CLIQUE */
int isClique(int nodes[], int size) {

/* FIND ALL TRIANGLES (3-cliques): O(V^3) */
void findAllTriangles() {
    int i, j, k, count = 0;
    printf("All 3-Cliques (Triangles):\n");
    for (i = 0; i < V; i++)
        for (j = i + 1; j < V; j++)
            for (k = j + 1; k < V; k++)
                if (matrix[i][j] && matrix[j][k] && matrix[i][k]) {
                    printf("  {%d, %d, %d}\n", i, j, k);
                    count++;
                }
    printf("Total triangles: %d\n", count);
}

/* MAXIMUM CLIQUE: Bron-Kerbosch algorithm with backtracking */
int maxClique[MAX], maxCliqueSize = 0;
int currentClique[MAX], currentSize = 0;

void bronKerbosch(int R[], int P[], int X[], int rSize, int pSize, int xSize) {
    int i, j, pNew[MAX], xNew[MAX], pNewSize, xNewSize;
    if (pSize == 0 && xSize == 0) {
        /* Found a maximal clique! */
        if (rSize > maxCliqueSize) {
            maxCliqueSize = rSize;
            for (i = 0; i < rSize; i++) maxClique[i] = R[i];
        }
        return;
    }
    for (i = 0; i < pSize; i++) {
        int v = P[i];
        pNewSize = xNewSize = 0;
        int rNew[MAX];
        for (j = 0; j < rSize; j++) rNew[j] = R[j];
        rNew[rSize] = v;
        /* P ∩ N(v) */
        for (j = 0; j < pSize; j++)
            if (matrix[v][P[j]]) pNew[pNewSize++] = P[j];
        /* X ∩ N(v) */
        for (j = 0; j < xSize; j++)
            if (matrix[v][X[j]]) xNew[xNewSize++] = X[j];
        bronKerbosch(rNew, pNew, xNew, rSize+1, pNewSize, xNewSize);
        X[xSize++] = v;   /* move v from P to X */
    }
}

int matrix[MAX][MAX], V;

/* IS EMPTY: Does the graph have any edges? */

    int i, j;

        for (j = 0; j < V; j++)
            if (matrix[i][j]) return 0;  /* found an edge */
    return 1;  /* no edges = empty */
}

/* CLEAR GRAPH: Remove all edges (keep vertices) */
void clearGraph() {
    int i, j;
    for (i = 0; i < V; i++)
        for (j = 0; j < V; j++)
            matrix[i][j] = 0;
    printf("Graph cleared. All edges removed.\n");

/* COPY GRAPH: Duplicate the graph into another struct */
void copyGraph(int dest[MAX][MAX], int *destV) {
    int i, j;
    *destV = V;
    for (i = 0; i < V; i++)
        for (j = 0; j < V; j++)
            dest[i][j] = matrix[i][j];

}
